32 Foam::autoPtr<Foam::dynamicFvMesh> Foam::dynamicFvMesh::New(const IOobject& io)
33 {
34  // Note: - do not register the dictionary since dynamicFvMeshes themselves
35  // do this.
36  // - defaultRegion (region0) gets loaded from constant, other ones
37  // get loaded from constant/<regionname>. Normally we'd use
38  // polyMesh::dbDir() but we haven't got a polyMesh yet ...
39  IOdictionary dict
40  (
41  IOobject
42  (
43  "dynamicMeshDict",
44  io.time().constant(),
45  (io.name() == polyMesh::defaultRegion ? "" : io.name()),
46  io.db(),
47  IOobject::MUST_READ_IF_MODIFIED,
48  IOobject::NO_WRITE,
49  false
50  )
51  );
52 
53  const word dynamicFvMeshTypeName(dict.lookup("dynamicFvMesh"));
54 
55  Info<< "Selecting dynamicFvMesh " << dynamicFvMeshTypeName << endl;
56 
57  const_cast<Time&>(io.time()).libs().open
58  (
59  dict,
60  "dynamicFvMeshLibs",
61  IOobjectConstructorTablePtr_
62  );
63 
64  if (!IOobjectConstructorTablePtr_)
65  {
66  FatalErrorIn
67  (
68  "dynamicFvMesh::New(const IOobject&)"
69  ) << "dynamicFvMesh table is empty"
70  << exit(FatalError);
71  }
72 
73  IOobjectConstructorTable::iterator cstrIter =
74  IOobjectConstructorTablePtr_->find(dynamicFvMeshTypeName);
75 
76  if (cstrIter == IOobjectConstructorTablePtr_->end())
77  {
78  FatalErrorIn
79  (
80  "dynamicFvMesh::New(const IOobject&)"
81  ) << "Unknown dynamicFvMesh type "
82  << dynamicFvMeshTypeName << nl << nl
83  << "Valid dynamicFvMesh types are :" << endl
84  << IOobjectConstructorTablePtr_->sortedToc()
85  << exit(FatalError);
86  }
87 
88  return autoPtr<dynamicFvMesh>(cstrIter()(io));
89 }
