37 Foam::word Foam::Time::findInstance
38 (
39  const fileName& dir,
40  const word& name,
41  const IOobject::readOption rOpt,
42  const word& stopInstance
43 ) const
44 {
45  // Note: if name is empty, just check the directory itself
46 
47 
48  const fileName tPath(path());
49  const fileName dirPath(tPath/timeName()/dir);
50 
51  // check the current time directory
52  if
53  (
54  name.empty()
55  ? isDir(dirPath)
56  :
57  (
58  isFile(dirPath/name)
59  && IOobject(name, timeName(), dir, *this).headerOk()
60  )
61  )
62  {
63  if (debug)
64  {
65  Info<< "Time::findInstance"
66  "(const fileName&, const word&"
67  ", const IOobject::readOption, const word&)"
68  << " : found \"" << name
69  << "\" in " << timeName()/dir
70  << endl;
71  }
72 
73  return timeName();
74  }
75 
76  // Search back through the time directories to find the time
77  // closest to and lower than current time
78 
79  instantList ts = times();
80  label instanceI;
81 
82  for (instanceI = ts.size()-1; instanceI >= 0; --instanceI)
83  {
84  if (ts[instanceI].value() <= timeOutputValue())
85  {
86  break;
87  }
88  }
89 
90  // continue searching from here
91  for (; instanceI >= 0; --instanceI)
92  {
93  if
94  (
95  name.empty()
96  ? isDir(tPath/ts[instanceI].name()/dir)
97  :
98  (
99  isFile(tPath/ts[instanceI].name()/dir/name)
100  && IOobject(name, ts[instanceI].name(), dir, *this).headerOk()
101  )
102  )
103  {
104  if (debug)
105  {
106  Info<< "Time::findInstance"
107  "(const fileName&, const word&"
108  ", const IOobject::readOption, const word&)"
109  << " : found \"" << name
110  << "\" in " << ts[instanceI].name()/dir
111  << endl;
112  }
113 
114  return ts[instanceI].name();
115  }
116 
117  // Check if hit minimum instance
118  if (ts[instanceI].name() == stopInstance)
119  {
120  if (debug)
121  {
122  Info<< "Time::findInstance"
123  "(const fileName&, const word&"
124  ", const IOobject::readOption, const word&)"
125  << " : hit stopInstance " << stopInstance
126  << endl;
127  }
128 
129  if
130  (
131  rOpt == IOobject::MUST_READ
132  || rOpt == IOobject::MUST_READ_IF_MODIFIED
133  )
134  {
135  if (name.empty())
136  {
137  FatalErrorIn
138  (
139  "Time::findInstance"
140  "(const fileName&, const word&"
141  ", const IOobject::readOption, const word&)"
142  ) << "Cannot find directory "
143  << dir << " in times " << timeName()
144  << " down to " << stopInstance
145  << exit(FatalError);
146  }
147  else
148  {
149  FatalErrorIn
150  (
151  "Time::findInstance"
152  "(const fileName&, const word&"
153  ", const IOobject::readOption, const word&)"
154  ) << "Cannot find file \"" << name << "\" in directory "
155  << dir << " in times " << timeName()
156  << " down to " << stopInstance
157  << exit(FatalError);
158  }
159  }
160 
161  return ts[instanceI].name();
162  }
163  }
164 
165 
166  // not in any of the time directories, try constant
167 
168  // Note. This needs to be a hard-coded constant, rather than the
169  // constant function of the time, because the latter points to
170  // the case constant directory in parallel cases
171 
172  if
173  (
174  name.empty()
175  ? isDir(tPath/constant()/dir)
176  :
177  (
178  isFile(tPath/constant()/dir/name)
179  && IOobject(name, constant(), dir, *this).headerOk()
180  )
181  )
182  {
183  if (debug)
184  {
185  Info<< "Time::findInstance"
186  "(const fileName&, const word&"
187  ", const IOobject::readOption, const word&)"
188  << " : found \"" << name
189  << "\" in " << constant()/dir
190  << endl;
191  }
192 
193  return constant();
194  }
195 
196  if (rOpt == IOobject::MUST_READ || rOpt == IOobject::MUST_READ_IF_MODIFIED)
197  {
198  FatalErrorIn
199  (
200  "Time::findInstance"
201  "(const fileName&, const word&"
202  ", const IOobject::readOption, const word&)"
203  ) << "Cannot find file \"" << name << "\" in directory "
204  << dir << " in times " << timeName()
205  << " down to " << constant()
206  << exit(FatalError);
207  }
208 
209  return constant();
210 }
