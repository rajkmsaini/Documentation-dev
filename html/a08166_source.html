446 bool Foam::Time::writeObject
447 (
448  IOstream::streamFormat fmt,
449  IOstream::versionNumber ver,
450  IOstream::compressionType cmp
451 ) const
452 {
453  if (writeTime())
454  {
455  bool writeOK = writeTimeDict();
456 
457  if (writeOK)
458  {
459  writeOK = objectRegistry::writeObject(fmt, ver, cmp);
460  }
461 
462  if (writeOK)
463  {
464  // Does the writeTime trigger purging?
465  if (writeTime_ && purgeWrite_)
466  {
467  previousWriteTimes_.push(timeName());
468 
469  while (previousWriteTimes_.size() > purgeWrite_)
470  {
471  rmDir(objectRegistry::path(previousWriteTimes_.pop()));
472  }
473  }
474  }
475 
476  return writeOK;
477  }
478  else
479  {
480  return false;
