74 scalar azizChen::unscaledEnergy(const scalar r) const
75 {
76  scalar x = r/rm_;
77 
78  scalar F = 1.0;
79 
80  if (x < D_)
81  {
82  F = exp(-pow(((D_ / x) - 1.0),2));
83  }
84 
85  return
86  epsilon_
87  *(
88  A_ * Foam::pow(x, gamma_)*exp(-alpha_*x)
89  - (
90  (C6_/ Foam::pow(x, 6))
91  + (C8_/ Foam::pow(x, 8))
92  + (C10_/ Foam::pow(x, 10))
93  )
94  *F
95  );
96 }
