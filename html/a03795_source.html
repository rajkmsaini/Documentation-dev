70 Foam::tmp<Foam::volVectorField> Foam::wallLubricationModels::Frank::Fi() const
71 {
72  volVectorField Ur(pair_.Ur());
73 
74  const volVectorField& n(nWall());
75  const volScalarField& y(yWall());
76 
77  volScalarField Eo(pair_.Eo());
78  volScalarField yTilde(y/(Cwc_*pair_.dispersed().d()));
79 
80  return zeroGradWalls
81  (
82  (
83  pos(Eo - 1.0)*neg(Eo - 5.0)*exp(-0.933*Eo + 0.179)
84  + pos(Eo - 5.0)*neg(Eo - 33.0)*(0.00599*Eo - 0.0187)
85  + pos(Eo - 33.0)*0.179
86  )
87  *max
88  (
89  dimensionedScalar("zero", dimless/dimLength, 0.0),
90  (1.0 - yTilde)/(Cwd_*y*pow(yTilde, p_ - 1.0))
91  )
92  *pair_.continuous().rho()
93  *magSqr(Ur - (Ur & n)*n)
94  *n
95  );
96 }
