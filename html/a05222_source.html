100 void Foam::waveDisplacementPointPatchVectorField::updateCoeffs()
101 {
102  if (this->updated())
103  {
104  return;
105  }
106 
107  const polyMesh& mesh = this->internalField().mesh()();
108  const Time& t = mesh.time();
109 
110  const scalarField points( waveNumber_ & patch().localPoints());
111 
112  Field<vector>::operator=
113  (
114  amplitude_*cos(omega_*t.value() - points)
115  );
116 
117  fixedValuePointPatchField<vector>::updateCoeffs();
118 }
