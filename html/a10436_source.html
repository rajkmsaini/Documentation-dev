62 template<class ThermoType>
63 const ThermoType& Foam::homogeneousMixture<ThermoType>::mixture
64 (
65  const scalar b
66 ) const
67 {
68  if (b > 0.999)
69  {
70  return reactants_;
71  }
72  else if (b < 0.001)
73  {
74  return products_;
75  }
76  else
77  {
78  mixture_ = b/reactants_.W()*reactants_;
79  mixture_ += (1 - b)/products_.W()*products_;
80 
81  return mixture_;
82  }
83 }
