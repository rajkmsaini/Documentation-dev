100 void Foam::setToCellZone::applyToSet
101 (
102  const topoSetSource::setAction action,
103  topoSet& set
104 ) const
105 {
106  if (!isA<cellZoneSet>(set))
107  {
108  WarningInFunction
109  << "Operation only allowed on a cellZoneSet." << endl;
110  }
111  else
112  {
113  cellZoneSet& fzSet = refCast<cellZoneSet>(set);
114 
115  if ((action == topoSetSource::NEW) || (action == topoSetSource::ADD))
116  {
117  Info<< " Adding all cells from cellSet " << setName_
118  << " ..." << endl;
119 
120  // Load the sets
121  cellSet fSet(mesh_, setName_);
122 
123  // Start off from copy
124  DynamicList<label> newAddressing(fzSet.addressing());
125 
126  forAllConstIter(cellSet, fSet, iter)
127  {
128  label cellI = iter.key();
129 
130  if (!fzSet.found(cellI))
131  {
132  newAddressing.append(cellI);
133  }
134  }
135 
136  fzSet.addressing().transfer(newAddressing);
137  fzSet.updateSet();
138  }
139  else if (action == topoSetSource::DELETE)
140  {
141  Info<< " Removing all cells from cellSet " << setName_
142  << " ..." << endl;
143 
144  // Load the set
145  cellSet loadedSet(mesh_, setName_);
146 
147  // Start off empty
148  DynamicList<label> newAddressing(fzSet.addressing().size());
149 
150  forAll(fzSet.addressing(), i)
151  {
152  if (!loadedSet.found(fzSet.addressing()[i]))
153  {
154  newAddressing.append(fzSet.addressing()[i]);
155  }
156  }
157  fzSet.addressing().transfer(newAddressing);
158  fzSet.updateSet();
159  }
160  }
161 }
