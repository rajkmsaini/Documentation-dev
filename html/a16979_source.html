73 inline Foam::scalar Foam::JanevReactionRate::operator()
74 (
75  const scalar p,
76  const scalar T,
77  const scalarField&
78 ) const
79 {
80  scalar lta = A_;
81 
82  if (mag(beta_) > VSMALL)
83  {
84  lta *= pow(T, beta_);
85  }
86 
87  scalar expArg = 0.0;
88 
89  if (mag(Ta_) > VSMALL)
90  {
91  expArg -= Ta_/T;
92  }
93 
94  scalar lnT = log(T);
95 
96  for (int n=0; n<nb_; n++)
97  {
98  expArg += b_[n]*pow(lnT, n);
99  }
100 
101  lta *= exp(expArg);
102 
103  return lta;
104 }
