163 Foam::autoPtr<Foam::graph::writer> Foam::graph::writer::New
164 (
165  const word& graphFormat
166 )
167 {
168  if (!wordConstructorTablePtr_)
169  {
170  FatalErrorIn
171  (
172  "graph::writer::New(const word&)"
173  ) << "Graph writer table is empty"
174  << exit(FatalError);
175  }
176 
177  wordConstructorTable::iterator cstrIter =
178  wordConstructorTablePtr_->find(graphFormat);
179 
180  if (cstrIter == wordConstructorTablePtr_->end())
181  {
182  FatalErrorIn
183  (
184  "graph::writer::New(const word&)"
185  ) << "Unknown graph format " << graphFormat
186  << endl << endl
187  << "Valid graph formats are : " << endl
188  << wordConstructorTablePtr_->sortedToc()
189  << exit(FatalError);
190  }
191 
192  return autoPtr<graph::writer>(cstrIter()());
193 }
