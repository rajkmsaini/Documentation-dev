150 bool Foam::functionObjectList::readFunctionObject
151 (
152  const string& funcNameArgs,
153  dictionary& functionsDict,
154  HashSet<word>& requiredFields
155 )
156 {
157  // Parse the optional functionObject arguments:
158  // 'Q(U)' -> funcName = Q; args = (U); field = U
159  //
160  // Supports named arguments:
161  // 'patchAverage(patch=inlet, p)' -> funcName = patchAverage;
162  // args = (patch=inlet, p); field = p
163 
164  word funcName(funcNameArgs);
165 
166  int argLevel = 0;
167  wordList args;
168 
169  List<Tuple2<word, string>> namedArgs;
170  bool namedArg = false;
171  word argName;
172 
173  word::size_type start = 0;
174  word::size_type i = 0;
175 
176  for
177  (
178  word::const_iterator iter = funcNameArgs.begin();
179  iter != funcNameArgs.end();
180  ++iter
181  )
182  {
183  char c = *iter;
184 
185  if (c == '(')
186  {
187  if (argLevel == 0)
188  {
189  funcName = funcNameArgs(start, i - start);
190  start = i+1;
191  }
192  ++argLevel;
193  }
194  else if (c == ',' || c == ')')
195  {
196  if (argLevel == 1)
197  {
198  if (namedArg)
199  {
200  namedArgs.append
201  (
202  Tuple2<word, string>
203  (
204  argName,
205  funcNameArgs(start, i - start)
206  )
207  );
208  namedArg = false;
209  }
210  else
211  {
212  args.append
213  (
214  string::validate<word>(funcNameArgs(start, i - start))
215  );
216  }
217  start = i+1;
218  }
219 
220  if (c == ')')
221  {
222  if (argLevel == 1)
223  {
224  break;
225  }
226  --argLevel;
227  }
228  }
229  else if (c == '=')
230  {
231  argName = string::validate<word>(funcNameArgs(start, i - start));
232  start = i+1;
233  namedArg = true;
234  }
235 
236  ++i;
237  }
238 
239  // Search for the functionObject dictionary
240  fileName path = findDict(funcName);
241 
242  if (path == fileName::null)
243  {
244  WarningInFunction
245  << "Cannot find functionObject file " << funcName << endl;
246  return false;
247  }
248 
249  // Read the functionObject dictionary
250  IFstream fileStream(path);
251  dictionary funcsDict(fileStream);
252  dictionary* funcDictPtr = &funcsDict;
253 
254  if (funcsDict.found(funcName) && funcsDict.isDict(funcName))
255  {
256  funcDictPtr = &funcsDict.subDict(funcName);
257  }
258 
259  dictionary& funcDict = *funcDictPtr;
260 
261  // Insert the 'field' and/or 'fields' entry corresponding to the optional
262  // arguments or read the 'field' or 'fields' entry and add the required
263  // fields to requiredFields
264  if (args.size() == 1)
265  {
266  funcDict.set("field", args[0]);
267  funcDict.set("fields", args);
268  requiredFields.insert(args[0]);
269  }
270  else if (args.size() > 1)
271  {
272  funcDict.set("fields", args);
273  requiredFields.insert(args);
274  }
275  else if (funcDict.found("field"))
276  {
277  requiredFields.insert(word(funcDict.lookup("field")));
278  }
279  else if (funcDict.found("fields"))
280  {
281  requiredFields.insert(wordList(funcDict.lookup("fields")));
282  }
283 
284  // Insert named arguments
285  forAll(namedArgs, i)
286  {
287  IStringStream entryStream
288  (
289  namedArgs[i].first() + ' ' + namedArgs[i].second() + ';'
290  );
291  funcDict.set(entry::New(entryStream).ptr());
292  }
293 
294  // Merge this functionObject dictionary into functionsDict
295  dictionary funcArgsDict;
296  funcArgsDict.add(string::validate<word>(funcNameArgs), funcDict);
297  functionsDict.merge(funcArgsDict);
298 
299  return true;
300 }

339 (
340  const argList& args,
341  const Time& runTime,
342  dictionary& controlDict,
343  HashSet<word>& requiredFields
344 )

346  autoPtr<functionObjectList> functionsPtr;

348  controlDict.add

352 
353  dictionary& functionsDict = controlDict.subDict("functions");
354 
355  if
356  (
357  args.optionFound("dict")
358  || args.optionFound("func")
359  || args.optionFound("funcs")
360  )
361  {
362  if (args.optionFound("dict"))
363  {
364  controlDict.merge
365  (
366  IOdictionary
367  (
368  IOobject

376  }
377 
378  if (args.optionFound("func"))
379  {
380  readFunctionObject(args["func"], functionsDict, requiredFields);

526 bool Foam::functionObjectList::read()
527 {
528  bool ok = true;
529  updated_ = execution_;
530 
531  // Avoid reading/initializing if execution is off
532  if (!execution_)
533  {
534  return true;
535  }
536 
537  // Update existing and add new functionObjects
538  const entry* entryPtr = parentDict_.lookupEntryPtr
539  (
540  "functions",
541  false,
542  false
543  );
544 
545  if (entryPtr)
546  {
547  PtrList<functionObject> newPtrs;
548  List<SHA1Digest> newDigs;
549  HashTable<label> newIndices;
550 
551  label nFunc = 0;
552 
553  if (!entryPtr->isDict())
554  {
555  FatalIOErrorInFunction(parentDict_)
556  << "'functions' entry is not a dictionary"
557  << exit(FatalIOError);
558  }
559 
560  const dictionary& functionsDict = entryPtr->dict();
561 
562  const_cast<Time&>(time_).libs().open
563  (
564  functionsDict,
565  "libs",
566  functionObject::dictionaryConstructorTablePtr_
567  );
568 
569  newPtrs.setSize(functionsDict.size());
570  newDigs.setSize(functionsDict.size());
571 
572  forAllConstIter(dictionary, functionsDict, iter)
573  {
574  const word& key = iter().keyword();
575 
576  if (!iter().isDict())
577  {
578  if (key != "libs")
579  {
580  IOWarningInFunction(parentDict_)
581  << "Entry " << key << " is not a dictionary" << endl;
582  }
583 
584  continue;
585  }
586 
587  const dictionary& dict = iter().dict();
588  bool enabled = dict.lookupOrDefault("enabled", true);
589 
590  newDigs[nFunc] = dict.digest();
591 
592  label oldIndex;
593  functionObject* objPtr = remove(key, oldIndex);
594 
595  if (objPtr)
596  {
597  if (enabled)
598  {
599  // Dictionary changed for an existing functionObject
600  if (newDigs[nFunc] != digests_[oldIndex])
601  {
602  ok = objPtr->read(dict) && ok;
603  }
604  }
605  else
606  {
607  // Delete the disabled functionObject
608  delete objPtr;
609  objPtr = NULL;
610  continue;
611  }
612  }
613  else if (enabled)
614  {
615  autoPtr<functionObject> foPtr;
616 
617  FatalError.throwExceptions();
618  FatalIOError.throwExceptions();
619  try
620  {
621  if
622  (
623  dict.found("writeControl")
624  || dict.found("outputControl")
625  )
626  {
627  foPtr.set
628  (
629  new functionObjects::timeControl(key, time_, dict)
630  );
631  }
632  else
633  {
634  foPtr = functionObject::New(key, time_, dict);
635  }
636  }
637  catch (Foam::IOerror& ioErr)
638  {
639  Info<< ioErr << nl << endl;
640  ::exit(1);
641  }
642  catch (Foam::error& err)
643  {
644  WarningInFunction
645  << "Caught FatalError " << err << nl << endl;
646  }
647  FatalError.dontThrowExceptions();
648  FatalIOError.dontThrowExceptions();
649 
650  if (foPtr.valid())
651  {
652  objPtr = foPtr.ptr();
653  }
654  else
655  {
656  ok = false;
657  }
658  }
659 
660  // Insert active functionObjects into the list
661  if (objPtr)
662  {
663  newPtrs.set(nFunc, objPtr);
664  newIndices.insert(key, nFunc);
665  nFunc++;
666  }
667  }
668 
669  newPtrs.setSize(nFunc);
670  newDigs.setSize(nFunc);
671 
672  // Updating the PtrList of functionObjects deletes any
673  // existing unused functionObjects
674  PtrList<functionObject>::transfer(newPtrs);
675  digests_.transfer(newDigs);
676  indices_.transfer(newIndices);
677  }
678  else
679  {
680  PtrList<functionObject>::clear();
681  digests_.clear();
682  indices_.clear();
683  }
684 
685  return ok;
686 }
