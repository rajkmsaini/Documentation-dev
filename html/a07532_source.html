30 void Foam::blockDescriptor::check(const Istream& is)
31 {
32  forAll(blockShape_, pi)
33  {
34  if (blockShape_[pi] < 0)
35  {
36  FatalIOErrorInFunction(is)
37  << "Negative point label " << blockShape_[pi]
38  << " in block " << *this
39  << exit(FatalIOError);
40  }
41  else if (blockShape_[pi] >= blockPointField_.size())
42  {
43  FatalIOErrorInFunction(is)
44  << "Point label " << blockShape_[pi]
45  << " out of range 0.." << blockPointField_.size() - 1
46  << " in block " << *this
47  << exit(FatalIOError);
48  }
49  }
50 
51  const point blockCentre(blockShape_.centre(blockPointField_));
52  const faceList faces(blockShape_.faces());
53 
54  // Check each face is outward-pointing with respect to the block centre
55  label outwardFaceCount = 0;
56  boolList correctFaces(faces.size(), true);
57 
58  forAll(faces, i)
59  {
60  point faceCentre(faces[i].centre(blockPointField_));
61  vector faceNormal(faces[i].normal(blockPointField_));
62  if (mag(faceNormal) > SMALL)
63  {
64  if (((faceCentre - blockCentre) & faceNormal) > 0)
65  {
66  outwardFaceCount++;
67  }
68  else
69  {
70  correctFaces[i] = false;
71  }
72  }
73  else
74  {
75  outwardFaceCount++;
76  }
77  }
78 
79  // If all faces are inward-pointing the block is inside-out
80  if (outwardFaceCount == 0)
81  {
82  FatalIOErrorInFunction(is)
83  << "Block " << *this << " is inside-out"
84  << exit(FatalIOError);
85  }
86  else if (outwardFaceCount != faces.size())
87  {
88  FatalIOErrorInFunction(is)
89  << "Block " << *this << " has inward-pointing faces"
90  << nl << " ";
91 
92  forAll(correctFaces, i)
93  {
94  if (!correctFaces[i])
95  {
96  FatalIOError<< faces[i] << token::SPACE;
97  }
98  }
99 
100  FatalIOError << exit(FatalIOError);
101  }
102 }

138 Foam::blockDescriptor::blockDescriptor
139 (
140  const pointField& blockPointField,
141  const curvedEdgeList& edges,
142  Istream& is
143 )
144 :
145  blockPointField_(blockPointField),
146  curvedEdges_(edges),
147  blockShape_(is),
148  meshDensity_(),
149  edgePoints_(12),
150  edgeWeights_(12),
151  expand_
152  (
153  12,
154  gradingDescriptors()
155  ),
156  zoneName_()
157 {
158  // Examine next token
159  token t(is);
160 
161  // Optional zone name
162  if (t.isWord())
163  {
164  zoneName_ = t.wordToken();
165 
166  // Examine next token
167  is >> t;
168  }
169  is.putBack(t);
170 
171  if (t.isPunctuation())
172  {
173  // New-style: read a list of 3 values
174  if (t.pToken() == token::BEGIN_LIST)
175  {
176  is >> meshDensity_;
177  }
178  else
179  {
180  FatalIOErrorInFunction
181  (
182  is
183  ) << "incorrect token while reading n, expected '(', found "
184  << t.info()
185  << exit(FatalIOError);
186  }
187  }
188  else
189  {
190  // Old-style: read three labels
191  is >> meshDensity_.x()
192  >> meshDensity_.y()
193  >> meshDensity_.z();
194  }
195 
196  is >> t;
197  if (!t.isWord())
198  {
199  is.putBack(t);
200  }
201 
202  List<gradingDescriptors> expRatios(is);
203 
204  if (expRatios.size() == 1)
205  {
206  // Identical in x/y/z-directions
207  expand_ = expRatios[0];
208  }
209  else if (expRatios.size() == 3)
210  {
211  // x-direction
212  expand_[0] = expRatios[0];
213  expand_[1] = expRatios[0];
214  expand_[2] = expRatios[0];
215  expand_[3] = expRatios[0];
216 
217  // y-direction
218  expand_[4] = expRatios[1];
219  expand_[5] = expRatios[1];
220  expand_[6] = expRatios[1];
221  expand_[7] = expRatios[1];
222 
223  // z-direction
224  expand_[8] = expRatios[2];
225  expand_[9] = expRatios[2];
226  expand_[10] = expRatios[2];
227  expand_[11] = expRatios[2];
228  }
229  else if (expRatios.size() == 12)
230  {
231  expand_ = expRatios;
232  }
233  else
234  {
235  FatalIOErrorInFunction(is)
236  << "Unknown definition of expansion ratios: " << expRatios
237  << exit(FatalIOError);
238  }
239 
240  check(is);
241 
242  // Create a list of edges
243  makeBlockEdges();
244 }

322 Foam::Ostream& Foam::operator<<(Ostream& os, const blockDescriptor& bd)
323 {
324  const cellShape& bshape = bd.blockShape();
325  const labelList& blockLabels = bshape;
326 
327  os << bshape.model().name() << " (";
328 
329  forAll(blockLabels, labelI)
330  {
331  if (labelI)
332  {
333  os << ' ';
334  }
335  os << blockLabels[labelI];
336  }
337  os << ')';
338 
339  if (bd.zoneName().size())
340  {
341  os << ' ' << bd.zoneName();
342  }
343 
344  os << ' ' << bd.meshDensity()
345  << " simpleGrading (";
346 
347 
348  const List<gradingDescriptors>& expand = bd.expand_;
349 
350  // Can we use a compact notation?
351  if
352  (
353  // x-direction
354  (
355  expand[0] == expand[1]
356  && expand[0] == expand[2]
357  && expand[0] == expand[3]
358  )
359  && // y-direction
360  (
361  expand[4] == expand[5]
362  && expand[4] == expand[6]
363  && expand[4] == expand[7]
364  )
365  && // z-direction
366  (
367  expand[8] == expand[9]
368  && expand[8] == expand[10]
369  && expand[8] == expand[11]
370  )
371  )
372  {
373  os << expand[0] << ' ' << expand[4] << ' ' << expand[8];
374  }
375  else
376  {
377  forAll(expand, edgeI)
378  {
379  if (edgeI)
380  {
381  os << ' ';
382  }
383  os << expand[edgeI];
384  }
385  }
386 
387 
388  os << ")";
389 
390  return os;
391 }
