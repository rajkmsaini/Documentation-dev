77 template<class Type>
78 Type Foam::Function1Types::Sine<Type>::value(const scalar t) const
79 {
80  return
81  amplitude_->value(t)
82  *sin(constant::mathematical::twoPi*frequency_->value(t)*(t - t0_))
83  *scale_->value(t)
84  + level_->value(t);
