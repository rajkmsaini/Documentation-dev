81 template<class CompType, class ThermoType>
82 void
83 Foam::binaryNode<CompType, ThermoType>::calcV
84 (
85  chemPointISAT<CompType, ThermoType>*& elementLeft,
86  chemPointISAT<CompType, ThermoType>*& elementRight,
87  scalarField& v
88 )
89 {
90  // LT is the transpose of the L matrix
91  scalarSquareMatrix& LT = elementLeft->LT();
92  bool mechReductionActive = elementLeft->chemistry().mechRed()->active();
93  // difference of composition in the full species domain
94  scalarField phiDif(elementRight->phi() - elementLeft->phi());
95  const scalarField& scaleFactor(elementLeft->scaleFactor());
96  scalar epsTol = elementLeft->tolerance();
97 
98  // v = LT.T()*LT*phiDif
99  for (label i=0; i<elementLeft->completeSpaceSize(); i++)
100  {
101  label si = i;
102  bool outOfIndexI = true;
103  if (mechReductionActive)
104  {
105  if (i<elementLeft->completeSpaceSize()-2)
106  {
107  si = elementLeft->completeToSimplifiedIndex()[i];
108  outOfIndexI = (si==-1);
109  }
110  else// temperature and pressure
111  {
112  outOfIndexI = false;
113  label dif = i-(elementLeft->completeSpaceSize()-2);
114  si = elementLeft->nActiveSpecies()+dif;
115  }
116  }
117  if (!mechReductionActive || (mechReductionActive && !(outOfIndexI)))
118  {
119  v[i]=0.0;
120  for (label j=0; j<elementLeft->completeSpaceSize(); j++)
121  {
122  label sj = j;
123  bool outOfIndexJ = true;
124  if (mechReductionActive)
125  {
126  if (j<elementLeft->completeSpaceSize()-2)
127  {
128  sj = elementLeft->completeToSimplifiedIndex()[j];
129  outOfIndexJ = (sj==-1);
130  }
131  else
132  {
133  outOfIndexJ = false;
134  label dif = j-(elementLeft->completeSpaceSize()-2);
135  sj = elementLeft->nActiveSpecies()+dif;
136  }
137  }
138  if
139  (
140  !mechReductionActive
141  ||(mechReductionActive && !(outOfIndexJ))
142  )
143  {
144  // since L is a lower triangular matrix k=0->min(i, j)
145  for (label k=0; k<=min(si, sj); k++)
146  {
147  v[i] += LT(k, si)*LT(k, sj)*phiDif[j];
148  }
149  }
150  }
151  }
152  else
153  {
154  // when it is an inactive species the diagonal element of LT is
155  // 1/(scaleFactor*epsTol)
156  v[i] = phiDif[i]/sqr(scaleFactor[i]*epsTol);
157  }
158  }
159 }

162 template<class CompType, class ThermoType>
163 Foam::scalar Foam::binaryNode<CompType, ThermoType>::calcA
164 (
165  chemPointISAT<CompType, ThermoType>* elementLeft,
166  chemPointISAT<CompType, ThermoType>* elementRight
167 )
168 {
169  scalar a = 0.0;
170  scalarField phih((elementLeft->phi()+elementRight->phi())/2);
171  label completeSpaceSize = elementLeft->completeSpaceSize();
172  for (label i=0; i<completeSpaceSize; i++)
173  {
174  a += v_[i]*phih[i];
175  }
176  return a;
177 }
