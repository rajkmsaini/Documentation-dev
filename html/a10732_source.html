58 template<class BasicTurbulenceModel>
59 void Foam::ReynoldsStress<BasicTurbulenceModel>::correctWallShearStress
60 (
61  volSymmTensorField& R
62 ) const
63 {
64  const fvPatchList& patches = this->mesh_.boundary();
65 
66  forAll(patches, patchi)
67  {
68  const fvPatch& curPatch = patches[patchi];
69 
70  if (isA<wallFvPatch>(curPatch))
71  {
72  symmTensorField& Rw = R.boundaryField()[patchi];
73 
74  const scalarField& nutw = this->nut_.boundaryField()[patchi];
75 
76  const vectorField snGradU
77  (
78  this->U_.boundaryField()[patchi].snGrad()
79  );
80 
81  const vectorField& faceAreas
82  = this->mesh_.Sf().boundaryField()[patchi];
83 
84  const scalarField& magFaceAreas
85  = this->mesh_.magSf().boundaryField()[patchi];
86 
87  forAll(curPatch, facei)
88  {
89  // Calculate near-wall velocity gradient
90  tensor gradUw
91  = (faceAreas[facei]/magFaceAreas[facei])*snGradU[facei];
92 
93  // Calculate near-wall shear-stress tensor
94  tensor tauw = -nutw[facei]*2*dev(symm(gradUw));
95 
96  // Reset the shear components of the stress tensor
97  Rw[facei].xy() = tauw.xy();
98  Rw[facei].xz() = tauw.xz();
99  Rw[facei].yz() = tauw.yz();
100  }
101  }
102  }
103 }

231 Foam::tmp<Foam::fvVectorMatrix>
232 Foam::ReynoldsStress<BasicTurbulenceModel>::divDevRhoReff
233 (
234  volVectorField& U
235 ) const
236 {
237  if (couplingFactor_.value() > 0.0)
238  {
239  return
240  (
241  fvc::laplacian
242  (
243  (1.0 - couplingFactor_)*this->alpha_*this->rho_*this->nut(),
244  U,
245  "laplacian(nuEff,U)"
246  )
247  + fvc::div
248  (
249  this->alpha_*this->rho_*R_
250  + couplingFactor_
251  *this->alpha_*this->rho_*this->nut()*fvc::grad(U),
252  "div(devRhoReff)"
253  )
254  - fvc::div(this->alpha_*this->rho_*this->nu()*dev2(T(fvc::grad(U))))
255  - fvm::laplacian(this->alpha_*this->rho_*this->nuEff(), U)
256  );
257  }
258  else
259  {
260  return
261  (
262  fvc::laplacian
263  (
264  this->alpha_*this->rho_*this->nut(),
265  U,
266  "laplacian(nuEff,U)"
267  )
268  + fvc::div(this->alpha_*this->rho_*R_)
269  - fvc::div(this->alpha_*this->rho_*this->nu()*dev2(T(fvc::grad(U))))
270  - fvm::laplacian(this->alpha_*this->rho_*this->nuEff(), U)
271  );
272  }
273 
274  return
275  (
276  - fvc::div((this->alpha_*this->rho_*this->nuEff())*dev2(T(fvc::grad(U))))
277  - fvm::laplacian(this->alpha_*this->rho_*this->nuEff(), U)
