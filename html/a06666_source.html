39 Foam::patchInteractionDataList::patchInteractionDataList
40 (
41  const polyMesh& mesh,
42  const dictionary& dict
43 )
44 :
45  List<patchInteractionData>(dict.lookup("patches")),
46  patchGroupIDs_(this->size())
47 {
48  const polyBoundaryMesh& bMesh = mesh.boundaryMesh();
49  const wordList allPatchNames = bMesh.names();
50 
51  const List<patchInteractionData>& items = *this;
52  forAllReverse(items, i)
53  {
54  const word& patchName = items[i].patchName();
55  labelList patchIDs = findStrings(patchName, allPatchNames);
56 
57  if (patchIDs.empty())
58  {
59  WarningIn
60  (
61  "Foam::patchInteractionDataList::patchInteractionDataList"
62  "("
63  "const polyMesh&, "
64  "const dictionary&"
65  ")"
66  ) << "Cannot find any patch names matching " << patchName
67  << endl;
68  }
69 
70  patchGroupIDs_[i].transfer(patchIDs);
71  }
72 
73  // Check that all patches are specified
74  DynamicList<word> badPatches;
75  forAll(bMesh, patchI)
76  {
77  const polyPatch& pp = bMesh[patchI];
78  if
79  (
80  !pp.coupled()
81  && !isA<emptyPolyPatch>(pp)
82  && applyToPatch(pp.index()) < 0
83  )
84  {
85  badPatches.append(pp.name());
86  }
87  }
88 
89  if (badPatches.size() > 0)
90  {
91  FatalErrorIn

93  "Foam::patchInteractionDataList::patchInteractionDataList"
94  "("

97  ")"
98  ) << "All patches must be specified when employing local patch "
99  << "interaction. Please specify data for patches:" << nl

101  }
102 }

117 Foam::label Foam::patchInteractionDataList::applyToPatch(const label id) const
118 {
119  forAll(patchGroupIDs_, groupI)
120  {
121  const labelList& patchIDs = patchGroupIDs_[groupI];
122  forAll(patchIDs, patchI)
123  {
124  if (patchIDs[patchI] == id)
125  {
126  return groupI;
127  }
128  }
129  }
130 
131  return -1;
132 }
