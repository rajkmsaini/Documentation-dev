149 //- Get type (inside,outside,mixed,unknown) of point w.r.t. surface.
150 // Only makes sense for closed surfaces.
151 template<class PatchType>
152 Foam::volumeType Foam::treeDataPrimitivePatch<PatchType>::getVolumeType
153 (
154  const indexedOctree<treeDataPrimitivePatch<PatchType>>& oc,
155  const point& sample
156 ) const
157 {
158  // Need to determine whether sample is 'inside' or 'outside'
159  // Done by finding nearest face. This gives back a face which is
160  // guaranteed to contain nearest point. This point can be
161  // - in interior of face: compare to face normal
162  // - on edge of face: compare to edge normal
163  // - on point of face: compare to point normal
164  // Unfortunately the octree does not give us back the intersection point
165  // or where on the face it has hit so we have to recreate all that
166  // information.
167 
168 
169  // Find nearest face to sample
170  pointIndexHit info = oc.findNearest(sample, sqr(GREAT));
171 
172  if (info.index() == -1)
173  {
174  FatalErrorInFunction
175  << "Could not find " << sample << " in octree."
176  << abort(FatalError);
177  }
178 
179  // Get actual intersection point on face
180  label faceI = info.index();
181 
182  if (debug & 2)
183  {
184  Pout<< "getSampleType : sample:" << sample
185  << " nearest face:" << faceI;
186  }
187 
188  const pointField& points = patch_.localPoints();
189  const typename PatchType::FaceType& f = patch_.localFaces()[faceI];
190 
191  // Retest to classify where on face info is. Note: could be improved. We
192  // already have point.
193 
194  pointHit curHit = f.nearestPoint(sample, points);
195  const vector area = f.normal(points);
196  const point& curPt = curHit.rawPoint();
197 
198  //
199  // 1] Check whether sample is above face
200  //
201 
202  if (curHit.hit())
203  {
204  // Nearest point inside face. Compare to face normal.
205 
206  if (debug & 2)
207  {
208  Pout<< " -> face hit:" << curPt
209  << " comparing to face normal " << area << endl;
210  }
211  return indexedOctree<treeDataPrimitivePatch>::getSide
212  (
213  area,
214  sample - curPt
215  );
216  }
217 
218  if (debug & 2)
219  {
220  Pout<< " -> face miss:" << curPt;
221  }
222 
223  //
224  // 2] Check whether intersection is on one of the face vertices or
225  // face centre
226  //
227 
228  const scalar typDimSqr = mag(area) + VSMALL;
229 
230 
231  forAll(f, fp)
232  {
233  if ((magSqr(points[f[fp]] - curPt)/typDimSqr) < planarTol_)
234  {
235  // Face intersection point equals face vertex fp
236 
237  // Calculate point normal (wrong: uses face normals instead of
238  // triangle normals)
239 
240  return indexedOctree<treeDataPrimitivePatch>::getSide
241  (
242  patch_.pointNormals()[f[fp]],
243  sample - curPt
244  );
245  }
246  }
247 
248  const point fc(f.centre(points));
249 
250  if ((magSqr(fc - curPt)/typDimSqr) < planarTol_)
251  {
252  // Face intersection point equals face centre. Normal at face centre
253  // is already average of face normals
254 
255  if (debug & 2)
256  {
257  Pout<< " -> centre hit:" << fc
258  << " distance:" << magSqr(fc - curPt)/typDimSqr << endl;
259  }
260 
261  return indexedOctree<treeDataPrimitivePatch>::getSide
262  (
263  area,
264  sample - curPt
265  );
266  }
267 
268 
269 
270  //
271  // 3] Get the 'real' edge the face intersection is on
272  //
273 
274  const labelList& fEdges = patch_.faceEdges()[faceI];
275 
276  forAll(fEdges, fEdgeI)
277  {
278  label edgeI = fEdges[fEdgeI];
279  const edge& e = patch_.edges()[edgeI];
280 
281  pointHit edgeHit = e.line(points).nearestDist(sample);
282 
283  if ((magSqr(edgeHit.rawPoint() - curPt)/typDimSqr) < planarTol_)
284  {
285  // Face intersection point lies on edge e
286 
287  // Calculate edge normal (wrong: uses face normals instead of
288  // triangle normals)
289  const labelList& eFaces = patch_.edgeFaces()[edgeI];
290 
291  vector edgeNormal(vector::zero);
292 
293  forAll(eFaces, i)
294  {
295  edgeNormal += patch_.faceNormals()[eFaces[i]];
296  }
297 
298  if (debug & 2)
299  {
300  Pout<< " -> real edge hit point:" << edgeHit.rawPoint()
301  << " comparing to edge normal:" << edgeNormal
302  << endl;
303  }
304 
305  // Found face intersection point on this edge. Compare to edge
306  // normal
307  return indexedOctree<treeDataPrimitivePatch>::getSide
308  (
309  edgeNormal,
310  sample - curPt
311  );
312  }
313  }
314 
315 
316  //
317  // 4] Get the internal edge the face intersection is on
318  //
319 
320  forAll(f, fp)
321  {
322  pointHit edgeHit = linePointRef
323  (
324  points[f[fp]],
325  fc
326  ).nearestDist(sample);
327 
328  if ((magSqr(edgeHit.rawPoint() - curPt)/typDimSqr) < planarTol_)
329  {
330  // Face intersection point lies on edge between two face triangles
331 
332  // Calculate edge normal as average of the two triangle normals
333  vector e = points[f[fp]] - fc;
334  vector ePrev = points[f[f.rcIndex(fp)]] - fc;
335  vector eNext = points[f[f.fcIndex(fp)]] - fc;
336 
337  vector nLeft = ePrev ^ e;
338  nLeft /= mag(nLeft) + VSMALL;
339 
340  vector nRight = e ^ eNext;
341  nRight /= mag(nRight) + VSMALL;
342 
343  if (debug & 2)
344  {
345  Pout<< " -> internal edge hit point:" << edgeHit.rawPoint()
346  << " comparing to edge normal "
347  << 0.5*(nLeft + nRight)
348  << endl;
349  }
350 
351  // Found face intersection point on this edge. Compare to edge
352  // normal
353  return indexedOctree<treeDataPrimitivePatch>::getSide
354  (
355  0.5*(nLeft + nRight),
356  sample - curPt
357  );
358  }
359  }
360 
361  if (debug & 2)
362  {
363  Pout<< "Did not find sample " << sample
364  << " anywhere related to nearest face " << faceI << endl
365  << "Face:";
366 
367  forAll(f, fp)
368  {
369  Pout<< " vertex:" << f[fp] << " coord:" << points[f[fp]]
370  << endl;
371  }
372  }
373 
374  // Can't determine status of sample with respect to nearest face.
375  // Either
376  // - tolerances are wrong. (if e.g. face has zero area)
377  // - or (more likely) surface is not closed.
378 
379  return volumeType::UNKNOWN;
380 }

383 // Check if any point on shape is inside cubeBb.
384 template<class PatchType>
385 bool Foam::treeDataPrimitivePatch<PatchType>::overlaps
386 (
387  const label index,
388  const treeBoundBox& cubeBb
389 ) const
390 {
391  // 1. Quick rejection: bb does not intersect face bb at all
392  if (cacheBb_)
393  {
394  if (!cubeBb.overlaps(bbs_[index]))
395  {
396  return false;
397  }
398  }
399  else
400  {
401  if (!cubeBb.overlaps(calcBb(patch_.points(), patch_[index])))
402  {
403  return false;
404  }
405  }
406 
407 
408  // 2. Check if one or more face points inside
409 
410  const pointField& points = patch_.points();
411  const typename PatchType::FaceType& f = patch_[index];
412 
413  if (cubeBb.containsAny(points, f))
414  {
415  return true;
416  }
417 
418  // 3. Difficult case: all points are outside but connecting edges might
419  // go through cube. Use triangle-bounding box intersection.
420  const point fc = f.centre(points);
421 
422  if (f.size() == 3)
423  {
424  return triangleFuncs::intersectBb
425  (
426  points[f[0]],
427  points[f[1]],
428  points[f[2]],
429  cubeBb
430  );
431  }
432  else
433  {
434  forAll(f, fp)
435  {
436  bool triIntersects = triangleFuncs::intersectBb
437  (
438  points[f[fp]],
439  points[f[f.fcIndex(fp)]],
440  fc,
441  cubeBb
442  );
443 
444  if (triIntersects)
445  {
446  return true;
447  }
448  }
449  }
450 
451  return false;
452 }

455 // Check if any point on shape is inside sphere.
456 template<class PatchType>
457 bool Foam::treeDataPrimitivePatch<PatchType>::overlaps
458 (
459  const label index,
460  const point& centre,
461  const scalar radiusSqr
462 ) const
463 {
464  // 1. Quick rejection: sphere does not intersect face bb at all
465  if (cacheBb_)
466  {
467  if (!bbs_[index].overlaps(centre, radiusSqr))
468  {
469  return false;
470  }
471  }
472  else
473  {
474  if (!calcBb(patch_.points(), patch_[index]).overlaps(centre, radiusSqr))
475  {
476  return false;
477  }
478  }
479 
480  const pointField& points = patch_.points();
481  const face& f = patch_[index];
482 
483  pointHit nearHit = f.nearestPoint(centre, points);
484 
485  // If the distance to the nearest point on the face from the sphere centres
486  // is within the radius, then the sphere touches the face.
487  if (sqr(nearHit.distance()) < radiusSqr)
488  {
489  return true;
490  }
491 
492  return false;
493 }

611 template<class PatchType>
612 bool Foam::treeDataPrimitivePatch<PatchType>::findIntersection
613 (
614  const indexedOctree<treeDataPrimitivePatch<PatchType>>& tree,
615  const label index,
616  const point& start,
617  const point& end,
618  point& intersectionPoint
619 )
620 {
621  const treeDataPrimitivePatch<PatchType>& shape = tree.shapes();
622  const PatchType& patch = shape.patch();
623 
624  const pointField& points = patch.points();
625  const typename PatchType::FaceType& f = patch[index];
626 
627  // Do quick rejection test
628  if (shape.cacheBb_)
629  {
630  const treeBoundBox& faceBb = shape.bbs_[index];
631 
632  if ((faceBb.posBits(start) & faceBb.posBits(end)) != 0)
633  {
634  // start and end in same block outside of faceBb.
635  return false;
636  }
637  }
638 
639  const vector dir(end - start);
640  pointHit inter;
641 
642  if (f.size() == 3)
643  {
644  inter = triPointRef
645  (
646  points[f[0]],
647  points[f[1]],
648  points[f[2]]
649  ).intersection(start, dir, intersection::HALF_RAY, shape.planarTol_);
650  }
651  else
652  {
653  const pointField& faceCentres = patch.faceCentres();
654 
655  inter = f.intersection
656  (
657  start,
658  dir,
659  faceCentres[index],
660  points,
661  intersection::HALF_RAY,
662  shape.planarTol_
663  );
664  }
665 
666  if (inter.hit() && inter.distance() <= 1)
667  {
668  // Note: no extra test on whether intersection is in front of us
669  // since using half_ray
670  intersectionPoint = inter.hitPoint();
671 
672  return true;
673  }
674  else
675  {
676  return false;
677  }
678 }
