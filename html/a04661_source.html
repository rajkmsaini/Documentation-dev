166 bool Foam::functionObjects::fieldValues::volFieldValue::write()
167 {
168  fieldValue::write();
169 
170  if (Pstream::master())
171  {
172  writeTime(file());
173  }
174 
175  forAll(fields_, i)
176  {
177  const word& fieldName = fields_[i];
178  bool processed = false;
179 
180  processed = processed || writeValues<scalar>(fieldName);
181  processed = processed || writeValues<vector>(fieldName);
182  processed = processed || writeValues<sphericalTensor>(fieldName);
183  processed = processed || writeValues<symmTensor>(fieldName);
184  processed = processed || writeValues<tensor>(fieldName);
185 
186  if (!processed)
187  {
188  WarningInFunction
189  << "Requested field " << fieldName
190  << " not found in database and not processed"
191  << endl;
192  }
193  }
194 
195  if (Pstream::master())
196  {
197  file()<< endl;
198  }
199 
200  Log << endl;
201 
202  return true;
203 }
