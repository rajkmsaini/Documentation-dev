113 void Foam::functionObjects::Peclet::execute()
114 {
115  typedef compressible::turbulenceModel cmpTurbModel;
116  typedef incompressible::turbulenceModel icoTurbModel;
117 
118  const fvMesh& mesh = refCast<const fvMesh>(obr_);
119 
120  tmp<volScalarField> nuEff;
121  if (mesh.foundObject<cmpTurbModel>(turbulenceModel::propertiesName))
122  {
123  const cmpTurbModel& model =
124  mesh.lookupObject<cmpTurbModel>
125  (
126  turbulenceModel::propertiesName
127  );
128 
129  const volScalarField& rho =
130  mesh.lookupObject<volScalarField>(rhoName_);
131 
132  nuEff = model.muEff()/rho;
133  }
134  else if
135  (
136  mesh.foundObject<icoTurbModel>(turbulenceModel::propertiesName)
137  )
138  {
139  const icoTurbModel& model =
140  mesh.lookupObject<icoTurbModel>
141  (
142  turbulenceModel::propertiesName
143  );
144 
145  nuEff = model.nuEff();
146  }
147  else if (mesh.foundObject<dictionary>("transportProperties"))
148  {
149  const dictionary& model =
150  mesh.lookupObject<dictionary>("transportProperties");
151 
152  nuEff =
153  tmp<volScalarField>
154  (
155  new volScalarField
156  (
157  IOobject
158  (
159  "nuEff",
160  mesh.time().timeName(),
161  mesh,
162  IOobject::NO_READ,
163  IOobject::NO_WRITE
164  ),
165  mesh,
166  dimensionedScalar(model.lookup("nu"))
167  )
168  );
169  }
170  else
171  {
172  FatalErrorInFunction
173  << "Unable to determine the viscosity"
174  << exit(FatalError);
175  }
176 
177  const surfaceScalarField& phi =
178  mesh.lookupObject<surfaceScalarField>(phiName_);
179 
180  surfaceScalarField& Peclet =
181  const_cast<surfaceScalarField&>
182  (
183  mesh.lookupObject<surfaceScalarField>(type())
184  );
185 
186  Peclet =
187  mag(phi)
188  /(
189  mesh.magSf()
190  *mesh.surfaceInterpolation::deltaCoeffs()
191  *fvc::interpolate(nuEff)
192  );
193 }
