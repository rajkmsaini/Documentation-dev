294 template<class T>
295 void Foam::List<T>::setSize(const label newSize)
296 {
297  if (newSize < 0)
298  {
299  FatalErrorInFunction
300  << "bad set size " << newSize
301  << abort(FatalError);
302  }
303 
304  if (newSize != this->size_)
305  {
306  if (newSize > 0)
307  {
308  T* nv = new T[label(newSize)];
309 
310  if (this->size_)
311  {
312  label i = min(this->size_, newSize);
313 
314  #ifdef USEMEMCPY
315  if (contiguous<T>())
316  {
317  memcpy(nv, this->v_, i*sizeof(T));
318  }
319  else
320  #endif
321  {
322  T* vv = &this->v_[i];
323  T* av = &nv[i];
324  while (i--) *--av = *--vv;
325  }
326  }
327  if (this->v_) delete[] this->v_;
328 
329  this->size_ = newSize;
330  this->v_ = nv;
331  }
332  else
333  {
334  clear();
335  }
336  }
337 }
