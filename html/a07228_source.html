70 void Foam::cyclicACMIPolyPatch::resetAMI
71 (
72  const AMIPatchToPatchInterpolation::interpolationMethod&
73 ) const
74 {
75  if (owner())
76  {
77  const polyPatch& nonOverlapPatch = this->nonOverlapPatch();
78 
79  initPatchFaceAreas();
80 
81  // Reset patch face areas based on original patch for AMI calculation
82  vectorField::subField Sf = faceAreas();
83  vectorField::subField noSf = nonOverlapPatch.faceAreas();
84 
85  forAll(Sf, faceI)
86  {
87  Sf[faceI] = faceAreas0_[faceI];
88  noSf[faceI] = faceAreas0_[faceI];
89  }
90 
91  // Calculate the AMI using partial face-area-weighted
92  cyclicAMIPolyPatch::resetAMI
93  (
94  AMIPatchToPatchInterpolation::imPartialFaceAreaWeight
95  );
96 
97  srcMask_ =
98  min(scalar(1) - tolerance_, max(tolerance_, AMI().srcWeightsSum()));
99 
100  tgtMask_ =
101  min(scalar(1) - tolerance_, max(tolerance_, AMI().tgtWeightsSum()));
102 
103  forAll(Sf, faceI)
104  {
105  Sf[faceI] *= srcMask_[faceI];
106  noSf[faceI] *= 1.0 - srcMask_[faceI];
107  }
108 
109  setNeighbourFaceAreas();
110 
111  // Set the updated flag
112  updated_ = true;
113  }
114 }

117 void Foam::cyclicACMIPolyPatch::setNeighbourFaceAreas() const
118 {
119  const cyclicACMIPolyPatch& cp =
120  refCast<const cyclicACMIPolyPatch>(this->neighbPatch());
121  const polyPatch& pp = cp.nonOverlapPatch();
122 
123  const vectorField& faceAreas0 = cp.faceAreas0();
124 
125  if (tgtMask_.size() == cp.size())
126  {
127  vectorField::subField Sf = cp.faceAreas();
128  vectorField::subField noSf = pp.faceAreas();
129 
130  forAll(Sf, faceI)
131  {
132  Sf[faceI] = tgtMask_[faceI]*faceAreas0[faceI];
133  noSf[faceI] = (1.0 - tgtMask_[faceI])*faceAreas0[faceI];
134  }
135  }
136  else
137  {
138  WarningInFunction
139  << "Target mask size differs to that of the neighbour patch\n"
140  << " May occur when decomposing." << endl;
141  }
142 }

366 Foam::label Foam::cyclicACMIPolyPatch::nonOverlapPatchID() const
367 {
368  if (nonOverlapPatchID_ == -1)
369  {
370  nonOverlapPatchID_ =
371  this->boundaryMesh().findPatchID(nonOverlapPatchName_);
372 
373  if (nonOverlapPatchID_ == -1)
374  {
375  FatalErrorInFunction
376  << "Illegal non-overlapping patch name " << nonOverlapPatchName_
377  << nl << "Valid patch names are "
378  << this->boundaryMesh().names()
379  << exit(FatalError);
380  }
381 
382  if (nonOverlapPatchID_ < index())
383  {
384  FatalErrorInFunction
385  << "Boundary ordering error: " << type()
386  << " patch must be defined prior to its non-overlapping patch"
387  << nl
388  << type() << " patch: " << name() << ", ID:" << index() << nl
389  << "Non-overlap patch: " << nonOverlapPatchName_
390  << ", ID:" << nonOverlapPatchID_ << nl
391  << exit(FatalError);
392  }
393 
394  const polyPatch& noPp = this->boundaryMesh()[nonOverlapPatchID_];
395 
396  bool ok = true;
397 
398  if (size() == noPp.size())
399  {
400  const scalarField magSf(mag(faceAreas()));
401  const scalarField noMagSf(mag(noPp.faceAreas()));
402 
403  forAll(magSf, faceI)
404  {
405  scalar ratio = mag(magSf[faceI]/(noMagSf[faceI] + ROOTVSMALL));
406 
407  if (ratio - 1 > tolerance_)
408  {
409  ok = false;
410  break;
411  }
412  }
413  }
414  else
415  {
416  ok = false;
417  }
418 
419  if (!ok)
420  {
421  FatalErrorInFunction
422  << "Inconsistent ACMI patches " << name() << " and "
423  << noPp.name() << ". Patches should have identical topology"
424  << exit(FatalError);
425  }
426  }
427 
428  return nonOverlapPatchID_;
429 }
