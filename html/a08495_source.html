144 void Foam::processorGAMGInterfaceField::updateInterfaceMatrix
145 (
146  scalarField& result,
147  const scalarField&,
148  const scalarField& coeffs,
149  const direction cmpt,
150  const Pstream::commsTypes commsType
151 ) const
152 {
153  if (updatedMatrix())
154  {
155  return;
156  }
157 
158  label oldWarn = UPstream::warnComm;
159  UPstream::warnComm = comm();
160 
161  const labelUList& faceCells = procInterface_.faceCells();
162 
163  if (commsType == Pstream::nonBlocking && !Pstream::floatTransfer)
164  {
165  // Fast path.
166  if
167  (
168  outstandingRecvRequest_ >= 0
169  && outstandingRecvRequest_ < Pstream::nRequests()
170  )
171  {
172  UPstream::waitRequest(outstandingRecvRequest_);
173  }
174  // Recv finished so assume sending finished as well.
175  outstandingSendRequest_ = -1;
176  outstandingRecvRequest_ = -1;
177 
178  // Consume straight from scalarReceiveBuf_
179 
180  // Transform according to the transformation tensor
181  transformCoupleField(scalarReceiveBuf_, cmpt);
182 
183  // Multiply the field by coefficients and add into the result
184  forAll(faceCells, elemI)
185  {
186  result[faceCells[elemI]] -= coeffs[elemI]*scalarReceiveBuf_[elemI];
187  }
188  }
189  else
190  {
191  scalarField pnf
192  (
193  procInterface_.compressedReceive<scalar>(commsType, coeffs.size())
194  );
195  transformCoupleField(pnf, cmpt);
196 
197  forAll(faceCells, elemI)
198  {
199  result[faceCells[elemI]] -= coeffs[elemI]*pnf[elemI];
200  }
201  }
202 
203  const_cast<processorGAMGInterfaceField&>(*this).updatedMatrix() = true;
204 
205  UPstream::warnComm = oldWarn;
206 }
