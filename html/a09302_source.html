66 void Foam::partialWrite::read(const dictionary& dict)
67 {
68  dict.lookup("objectNames") >> objectNames_;
69  dict.lookup("writeInterval") >> writeInterval_;
70  writeInstance_ = 0;
71 
72  Info<< type() << " " << name() << ":" << nl
73  << " dumping every " << writeInterval_
74  << " th outputTime : " << nl << endl ;
75  forAllConstIter(HashSet<word>, objectNames_, iter)
76  {
77  Info<< ' ' << iter.key();
78  }
79 
80  if (writeInterval_ < 1)
81  {
82  FatalIOErrorInFunction(dict)
83  << "Illegal value for writeInterval " << writeInterval_
84  << ". It should be >= 1."
85  << exit(FatalIOError);
86  }
87 
88  // Clear out any previously loaded fields
89  vsf_.clear();
90  vvf_.clear();
91  vSpheretf_.clear();
92  vSymmtf_.clear();
93  vtf_.clear();
94 
95  ssf_.clear();
96  svf_.clear();
97  sSpheretf_.clear();
98  sSymmtf_.clear();
99  stf_.clear();
100 
101  forAllConstIter(HashSet<word>, objectNames_, iter)
102  {
103  loadField<scalar>(iter.key(), vsf_, ssf_);
104  loadField<vector>(iter.key(), vvf_, svf_);
105  loadField<sphericalTensor>(iter.key(), vSpheretf_, sSpheretf_);
106  loadField<symmTensor>(iter.key(), vSymmtf_, sSymmtf_);
107  loadField<tensor>(iter.key(), vtf_, stf_);
108  }

124 void Foam::partialWrite::timeSet()
125 {
126  if (obr_.time().outputTime())
127  {
128  writeInstance_++;
129 
130  if (writeInstance_ == writeInterval_)
131  {
132  // Next overall dump corresponds to partial write. Change
133  // write options to AUTO_WRITE
134  writeInstance_ = 0;
135 
136  changeWriteOptions<scalar>(vsf_, ssf_, IOobject::AUTO_WRITE);
137  changeWriteOptions<vector>(vvf_, svf_, IOobject::AUTO_WRITE);
138  changeWriteOptions<sphericalTensor>
139  (
140  vSpheretf_,
141  sSpheretf_,
142  IOobject::AUTO_WRITE
143  );
144  changeWriteOptions<symmTensor>
145  (
146  vSymmtf_,
147  sSymmtf_,
148  IOobject::AUTO_WRITE
149  );
150  changeWriteOptions<tensor>(vtf_, stf_, IOobject::AUTO_WRITE);
151  }
152  else
153  {
154  changeWriteOptions<scalar>(vsf_, ssf_, IOobject::NO_WRITE);
155  changeWriteOptions<vector>(vvf_, svf_, IOobject::NO_WRITE);
156  changeWriteOptions<sphericalTensor>
157  (
158  vSpheretf_,
159  sSpheretf_,
160  IOobject::NO_WRITE
161  );
162  changeWriteOptions<symmTensor>
163  (
164  vSymmtf_,
165  sSymmtf_,
166  IOobject::NO_WRITE
167  );
168  changeWriteOptions<tensor>(vtf_, stf_, IOobject::NO_WRITE);
169  }
170  }
171 }
