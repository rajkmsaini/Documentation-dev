90 {
91  for (int Dimension=0; Dimension<nDimensions; ++Dimension)
92  {
93  // ie, mag(exponents_[Dimension]) > smallExponent
94  if
95  (
96  exponents_[Dimension] > smallExponent

98  )
99  {
100  return false;
101  }
102  }
103 
104  return true;
