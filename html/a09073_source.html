87 void removeInjection::correct
88 (
89  scalarField& availableMass,
90  scalarField& massToInject,
91  scalarField& diameterToInject
92 )
93 {
94  const scalarField& delta = owner().delta();
95  const scalarField& rho = owner().rho();
96  const scalarField& magSf = owner().magSf();
97 
98  forAll(delta, cellI)
99  {
100  if (mask_[cellI] > 0)
101  {
102  scalar ddelta = max(0.0, delta[cellI] - deltaStable_);
103  scalar dMass = ddelta*rho[cellI]*magSf[cellI];
104  massToInject[cellI] += dMass;
105  availableMass[cellI] -= dMass;
106 
107  addToInjectedMass(dMass);
108  }
109  }
110 
111  injectionModel::correct();
112 }
