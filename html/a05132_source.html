160 template<class Type>
161 void mixedFvPatchField<Type>::evaluate(const Pstream::commsTypes)
162 {
163  if (!this->updated())
164  {
165  this->updateCoeffs();
166  }
167 
168  Field<Type>::operator=
169  (
170  valueFraction_*refValue_
171  +
172  (1.0 - valueFraction_)*
173  (
174  this->patchInternalField()
175  + refGrad_/this->patch().deltaCoeffs()
176  )
177  );
178 
179  fvPatchField<Type>::evaluate();
180 }
