74 void Foam::fieldValues::cellSource::setCellZoneCells()
75 {
76  switch (source_)
77  {
78  case stCellZone:
79  {
80  label zoneId = mesh().cellZones().findZoneID(sourceName_);
81 
82  if (zoneId < 0)
83  {
84  FatalErrorIn("cellSource::cellSource::setCellZoneCells()")
85  << "Unknown cell zone name: " << sourceName_
86  << ". Valid cell zones are: " << mesh().cellZones().names()
87  << nl << exit(FatalError);
88  }
89 
90  cellId_ = mesh().cellZones()[zoneId];
91  nCells_ = returnReduce(cellId_.size(), sumOp<label>());
92  break;
93  }
94 
95  case stAll:
96  {
97  cellId_ = identity(mesh().nCells());
98  nCells_ = returnReduce(cellId_.size(), sumOp<label>());
99  break;
100  }
101 
102  default:
103  {
104  FatalErrorIn("cellSource::setCellZoneCells()")
105  << "Unknown source type. Valid source types are:"
106  << sourceTypeNames_ << nl << exit(FatalError);
107  }
108  }
109 
110  if (debug)
111  {
112  Pout<< "Selected source size = " << cellId_.size() << endl;
113  }
114 }

119 void Foam::fieldValues::cellSource::initialise(const dictionary& dict)
120 {
121  setCellZoneCells();
122 
123  if (nCells_ == 0)
124  {
125  WarningIn
126  (
127  "Foam::fieldValues::cellSource::initialise(const dictionary&)"
128  )
129  << type() << " " << name_ << ": "
130  << sourceTypeNames_[source_] << "(" << sourceName_ << "):" << nl
131  << " Source has no cells - deactivating" << endl;
132 
133  active_ = false;
134  return;
135  }
136 
137  Info<< type() << " " << name_ << ":"
138  << sourceTypeNames_[source_] << "(" << sourceName_ << "):" << nl
139  << " total cells = " << nCells_ << nl
140  << " total volume = " << gSum(filterField(mesh().V()))
141  << nl << endl;
142 
143  if (dict.readIfPresent("weightField", weightFieldName_))
144  {
145  Info<< " weight field = " << weightFieldName_;
146  }
147 
148  Info<< nl << endl;
149 }

211 void Foam::fieldValues::cellSource::write()
212 {
213  fieldValue::write();
214 
215  if (active_)
216  {
217  scalar totalVolume = gSum(filterField(mesh().V()));
218  if (Pstream::master())
219  {
220  file() << obr_.time().value() << tab << totalVolume;
221  }
222 
223  forAll(fields_, i)
224  {
225  const word& fieldName = fields_[i];
226  bool processed = false;
227 
228  processed = processed || writeValues<scalar>(fieldName);
229  processed = processed || writeValues<vector>(fieldName);
230  processed = processed || writeValues<sphericalTensor>(fieldName);
231  processed = processed || writeValues<symmTensor>(fieldName);
232  processed = processed || writeValues<tensor>(fieldName);
233 
234  if (!processed)
235  {
236  WarningIn("void Foam::fieldValues::cellSource::write()")
237  << "Requested field " << fieldName
238  << " not found in database and not processed"
239  << endl;
240  }
241  }
242 
243  if (Pstream::master())
244  {
245  file()<< endl;
246  }
247 
248  Info(log_)<< endl;
249  }
250 }
