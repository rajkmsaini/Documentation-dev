31 Foam::autoPtr<Foam::engineMesh> Foam::engineMesh::New
32 (
33  const Foam::IOobject& io
34 )
35 {
36  // get model name, but do not register the dictionary
37  // otherwise it is registered in the database twice
38  const word modelType
39  (
40  IOdictionary
41  (
42  IOobject
43  (
44  "engineGeometry",
45  io.time().constant(),
46  io.db(),
47  IOobject::MUST_READ_IF_MODIFIED,
48  IOobject::NO_WRITE,
49  false
50  )
51  ).lookup("engineMesh")
52  );
53 
54  Info<< "Selecting engineMesh " << modelType << endl;
55 
56  IOobjectConstructorTable::iterator cstrIter =
57  IOobjectConstructorTablePtr_->find(modelType);
58 
59  if (cstrIter == IOobjectConstructorTablePtr_->end())
60  {
61  FatalErrorInFunction
62  << "Unknown engineMesh type "
63  << modelType << nl << nl
64  << "Valid engineMesh types are :" << endl
65  << IOobjectConstructorTablePtr_->sortedToc()
66  << exit(FatalError);
67  }
68 
69  return autoPtr<engineMesh>(cstrIter()(io));
70 }
