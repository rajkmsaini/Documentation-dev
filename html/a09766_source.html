128 void Foam::RBD::rigidBodyMotion::forwardDynamics
129 (
130  rigidBodyModelState& state,
131  const scalarField& tau,
132  const Field<spatialVector>& fx
133 ) const
134 {
135  scalarField qDdotPrev = state.qDdot();
136  rigidBodyModel::forwardDynamics(state, tau, fx);
137  state.qDdot() = aDamp_*(aRelax_*state.qDdot() + (1 - aRelax_)*qDdotPrev);
138 }

181 Foam::tmp<Foam::pointField> Foam::RBD::rigidBodyMotion::transformPoints
182 (
183  const label bodyID,
184  const scalarField& weight,
185  const pointField& initialPoints
186 ) const
187 {
188  // Calculate the transform from the initial state in the global frame
189  // to the current state in the global frame
190  spatialTransform X(X0(bodyID).inv() & X00(bodyID));
191 
192  // Calculate the septernion equivalent of the transformation for 'slerp'
193  // interpolation
194  septernion s(X);
195 
196  tmp<pointField> tpoints(new pointField(initialPoints));
197  pointField& points = tpoints.ref();
198 
199  forAll(points, i)
200  {
201  // Move non-stationary points
202  if (weight[i] > SMALL)
203  {
204  // Use solid-body motion where weight = 1
205  if (weight[i] > 1 - SMALL)
206  {
207  points[i] = X.transformPoint(initialPoints[i]);
208  }
209  // Slerp septernion interpolation
210  else
211  {
212  points[i] =
213  slerp(septernion::I, s, weight[i])
214  .transformPoint(initialPoints[i]);
215  }
216  }
217  }
218 
219  return tpoints;
220 }

223 Foam::tmp<Foam::pointField> Foam::RBD::rigidBodyMotion::transformPoints
224 (
225  const labelList& bodyIDs,
226  const List<const scalarField*>& weights,
227  const pointField& initialPoints
228 ) const
229 {
230  List<septernion> ss(bodyIDs.size() + 1);
231  ss[bodyIDs.size()] = septernion::I;
232 
233  forAll(bodyIDs, bi)
234  {
235  const label bodyID = bodyIDs[bi];
236 
237  // Calculate the transform from the initial state in the global frame
238  // to the current state in the global frame
239  spatialTransform X(X0(bodyID).inv() & X00(bodyID));
240 
241  // Calculate the septernion equivalent of the transformation
242  ss[bi] = septernion(X);
243  }
244 
245  tmp<pointField> tpoints(new pointField(initialPoints));
246  pointField& points = tpoints.ref();
247 
248  List<scalar> w(ss.size());
249 
250  forAll(points, i)
251  {
252  // Sum (1 - wi) and find the maximum wi
253  scalar sum1mw = 0;
254  scalar maxw = 0;
255 
256  forAll(bodyIDs, bi)
257  {
258  w[bi] = (*(weights[bi]))[i];
259  sum1mw += 1 - w[bi];
260  maxw = max(maxw, w[bi]);
261  }
262 
263  // Calculate the limiter for (1 - wi) to ensure the sum(wi) = maxw
264  scalar lambda = (w.size() - 1 - maxw)/sum1mw;
265 
266  // Limit (1 - wi) and sum the resulting wi
267  scalar sumw = 0;
268  forAll(bodyIDs, bi)
269  {
270  w[bi] = 1 - lambda*(1 - w[bi]);
271  sumw += w[bi];
272  }
273 
274  // Calculate the weight for the stationary far-field
275  w[bodyIDs.size()] = 1 - sumw;
276 
277  points[i] = average(ss, w).transformPoint(initialPoints[i]);
278  }
279 
280  return tpoints;
281 }
