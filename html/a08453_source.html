56 Foam::cachedRandom::cachedRandom(const label seed, const label count)
57 :
58  seed_(1),
59  samples_(0),
60  sampleI_(-1)
61 {
62  if (seed > 1)
63  {
64  seed_ = seed;
65  }
66 
67  // Samples will be cached if count > 0

73
