105 bool Foam::functionObjects::setTimeStepFunctionObject::adjustTimeStep()
106 {
107  if (enabled())
108  {
109  // Wanted timestep
110  scalar newDeltaT = timeStepPtr_().value(time_.timeOutputValue());
111 
112  const_cast<Time&>(time()).setDeltaT(newDeltaT, false);
113 
114  return true;
115  }
116  else
117  {
118  return false;
119  }
120 }

123 bool Foam::functionObjects::setTimeStepFunctionObject::read
124 (
125  const dictionary& dict
126 )
127 {
128  enabled_ = dict.lookupOrDefault("enabled", true);
129 
130  if (enabled_)
131  {
132  timeStepPtr_ = Function1<scalar>::New("deltaT", dict);
133 
134  // Check that time has adjustTimeStep
135  const dictionary& controlDict = time_.controlDict();
136 
137  Switch adjust;
138  if
139  (
140  !controlDict.readIfPresent<Switch>("adjustTimeStep", adjust)
141  || !adjust
142  )
143  {
144  FatalIOErrorInFunction(dict)
145  << "Need to have 'adjustTimeStep' true to enable external"
146  << " timestep control" << exit(FatalIOError);
147  }
148  }
149  return true;
150 }
