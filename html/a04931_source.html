164 bool Foam::functionObjects::scalarTransport::read(const dictionary& dict)
165 {
166  fvMeshFunctionObject::read(dict);
167 
168  phiName_ = dict.lookupOrDefault<word>("phi", "phi");
169  rhoName_ = dict.lookupOrDefault<word>("rho", "rho");
170  schemesField_ = dict.lookupOrDefault<word>("schemesField", fieldName_);
171 
172  constantD_ = false;
173  if (dict.readIfPresent("D", D_))
174  {
175  constantD_ = true;
176  }
177 
178  dict.readIfPresent("nCorr", nCorr_);
179 
180  if (dict.found("fvOptions"))
181  {
182  fvOptions_.reset(dict.subDict("fvOptions"));
183  }
184 
185  return true;
186 }

189 bool Foam::functionObjects::scalarTransport::execute()
190 {
191  Info<< type() << " write:" << endl;
192 
193  const surfaceScalarField& phi =
194  mesh_.lookupObject<surfaceScalarField>(phiName_);
195 
196  // Calculate the diffusivity
197  volScalarField D(this->D(phi));
198 
199  word divScheme("div(phi," + schemesField_ + ")");
200  word laplacianScheme("laplacian(" + D.name() + "," + schemesField_ + ")");
201 
202  // Set under-relaxation coeff
203  scalar relaxCoeff = 0.0;
204  if (mesh_.relaxEquation(schemesField_))
205  {
206  relaxCoeff = mesh_.equationRelaxationFactor(schemesField_);
207  }
208 
209  if (phi.dimensions() == dimMass/dimTime)
210  {
211  const volScalarField& rho =
212  mesh_.lookupObject<volScalarField>(rhoName_);
213 
214  for (label i = 0; i <= nCorr_; i++)
215  {
216  fvScalarMatrix sEqn
217  (
218  fvm::ddt(rho, s_)
219  + fvm::div(phi, s_, divScheme)
220  - fvm::laplacian(D, s_, laplacianScheme)
221  ==
222  fvOptions_(rho, s_)
223  );
224 
225  sEqn.relax(relaxCoeff);
226 
227  fvOptions_.constrain(sEqn);
228 
229  sEqn.solve(mesh_.solverDict(schemesField_));
230  }
231  }
232  else if (phi.dimensions() == dimVolume/dimTime)
233  {
234  for (label i = 0; i <= nCorr_; i++)
235  {
236  fvScalarMatrix sEqn
237  (
238  fvm::ddt(s_)
239  + fvm::div(phi, s_, divScheme)
240  - fvm::laplacian(D, s_, laplacianScheme)
241  ==
242  fvOptions_(s_)
243  );
244 
245  sEqn.relax(relaxCoeff);
246 
247  fvOptions_.constrain(sEqn);
248 
249  sEqn.solve(mesh_.solverDict(schemesField_));
250  }
251  }
252  else
253  {
254  FatalErrorInFunction
255  << "Incompatible dimensions for phi: " << phi.dimensions() << nl
256  << "Dimensions should be " << dimMass/dimTime << " or "
257  << dimVolume/dimTime << endl;
258  }
259 
260  Info<< endl;
261 
262  return true;
263 }
