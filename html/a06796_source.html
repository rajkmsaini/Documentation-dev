193 template<class CloudType>
194 template<class TrackData>
195 void Foam::CollidingCloud<CloudType>::motion(TrackData& td)
196 {
197  // Sympletic leapfrog integration of particle forces:
198  // + apply half deltaV with stored force
199  // + move positions with new velocity
200  // + calculate forces in new position
201  // + apply half deltaV with new force
202 
203  label nSubCycles = collision().nSubCycles();
204 
205  if (nSubCycles > 1)
206  {
207  Info<< " " << nSubCycles << " move-collide subCycles" << endl;
208 
209  subCycleTime moveCollideSubCycle
210  (
211  const_cast<Time&>(this->db().time()),
212  nSubCycles
213  );
214 
215  while(!(++moveCollideSubCycle).end())
216  {
217  moveCollide(td, this->db().time().deltaTValue());
218  }
219 
220  moveCollideSubCycle.endSubCycle();
221  }
222  else
223  {
224  moveCollide(td, this->db().time().deltaTValue());
225  }
226 }
