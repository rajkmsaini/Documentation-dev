150 template<class Type>
151 Type Foam::isoSurface::generatePoint
152 (
153  const scalar s0,
154  const Type& p0,
155  const bool hasSnap0,
156  const Type& snapP0,

158  const scalar s1,

161  const Type& snapP1
162 ) const
163 {
