103 tmp<GeometricField<Type, fvsPatchField, surfaceMesh> >
104 snGradScheme<Type>::snGrad
105 (
106  const GeometricField<Type, fvPatchField, volMesh>& vf,
107  const tmp<surfaceScalarField>& tdeltaCoeffs,
108  const word& snGradName
109 )
110 {
111  const fvMesh& mesh = vf.mesh();
112 
113  // construct GeometricField<Type, fvsPatchField, surfaceMesh>
114  tmp<GeometricField<Type, fvsPatchField, surfaceMesh> > tsf
115  (
116  new GeometricField<Type, fvsPatchField, surfaceMesh>
117  (
118  IOobject
119  (
120  snGradName + "("+vf.name()+')',
121  vf.instance(),
122  vf.mesh(),
123  IOobject::NO_READ,
124  IOobject::NO_WRITE
125  ),
126  mesh,
127  vf.dimensions()*tdeltaCoeffs().dimensions()
128  )
129  );
130  GeometricField<Type, fvsPatchField, surfaceMesh>& ssf = tsf();
131 
132  // set reference to difference factors array
133  const scalarField& deltaCoeffs = tdeltaCoeffs().internalField();
134 
135  // owner/neighbour addressing
136  const labelUList& owner = mesh.owner();
137  const labelUList& neighbour = mesh.neighbour();
138 
139  forAll(owner, facei)
140  {
141  ssf[facei] =
142  deltaCoeffs[facei]*(vf[neighbour[facei]] - vf[owner[facei]]);
143  }
144 
145  forAll(vf.boundaryField(), patchi)
146  {
147  const fvPatchField<Type>& pvf = vf.boundaryField()[patchi];
148 
149  if (pvf.coupled())
150  {
151  ssf.boundaryField()[patchi] =
152  pvf.snGrad(tdeltaCoeffs().boundaryField()[patchi]);
153  }
154  else
155  {
156  ssf.boundaryField()[patchi] = pvf.snGrad();
157  }
158  }
159 
160  return tsf;
161 }
