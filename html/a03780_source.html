61 // Calculate some edge statistics on mesh.
62 void printEdgeStats(const primitiveMesh& mesh)
63 {
64  label nX = 0;
65  label nY = 0;
66  label nZ = 0;
67 
68  scalar minX = GREAT;
69  scalar maxX = -GREAT;
70  vector x(1, 0, 0);
71 
72  scalar minY = GREAT;
73  scalar maxY = -GREAT;
74  vector y(0, 1, 0);
75 
76  scalar minZ = GREAT;
77  scalar maxZ = -GREAT;
78  vector z(0, 0, 1);
79 
80  scalar minOther = GREAT;
81  scalar maxOther = -GREAT;
82 
83  const edgeList& edges = mesh.edges();
84 
85  forAll(edges, edgeI)
86  {
87  const edge& e = edges[edgeI];
88 
89  vector eVec(e.vec(mesh.points()));
90 
91  scalar eMag = mag(eVec);
92 
93  eVec /= eMag;
94 
95  if (mag(eVec & x) > 1-edgeTol)
96  {
97  minX = min(minX, eMag);
98  maxX = max(maxX, eMag);
99  nX++;
100  }
101  else if (mag(eVec & y) > 1-edgeTol)
102  {
103  minY = min(minY, eMag);
104  maxY = max(maxY, eMag);
105  nY++;
106  }
107  else if (mag(eVec & z) > 1-edgeTol)
108  {
109  minZ = min(minZ, eMag);
110  maxZ = max(maxZ, eMag);
111  nZ++;
112  }
113  else
114  {
115  minOther = min(minOther, eMag);
116  maxOther = max(maxOther, eMag);
117  }
118  }
119 
120  Pout<< "Mesh edge statistics:" << endl
121  << " x aligned : number:" << nX << "\tminLen:" << minX
122  << "\tmaxLen:" << maxX << endl
123  << " y aligned : number:" << nY << "\tminLen:" << minY
124  << "\tmaxLen:" << maxY << endl
125  << " z aligned : number:" << nZ << "\tminLen:" << minZ
126  << "\tmaxLen:" << maxZ << endl
127  << " other : number:" << mesh.nEdges() - nX - nY - nZ
128  << "\tminLen:" << minOther
129  << "\tmaxLen:" << maxOther << endl << endl;
130 }

133 // Return index of coordinate axis.
134 label axis(const vector& normal)
135 {
136  label axisIndex = -1;
137 
138  if (mag(normal & point(1, 0, 0)) > (1-edgeTol))
139  {
140  axisIndex = 0;
141  }
142  else if (mag(normal & point(0, 1, 0)) > (1-edgeTol))
143  {
144  axisIndex = 1;
145  }
146  else if (mag(normal & point(0, 0, 1)) > (1-edgeTol))
147  {
148  axisIndex = 2;
149  }
150 
151  return axisIndex;
152 }

155 //- Returns -1 or cartesian coordinate component (0=x, 1=y, 2=z) of normal
156 // in case of 2D mesh
157 label twoDNess(const polyMesh& mesh)
158 {
159  const pointField& ctrs = mesh.cellCentres();
160 
161  if (ctrs.size() < 2)
162  {
163  return -1;
164  }
165 
166 
167  //
168  // 1. All cell centres on single plane aligned with x, y or z
169  //
170 
171  // Determine 3 points to base plane on.
172  vector vec10 = ctrs[1] - ctrs[0];
173  vec10 /= mag(vec10);
174 
175  label otherCellI = -1;
176 
177  for (label cellI = 2; cellI < ctrs.size(); cellI++)
178  {
179  vector vec(ctrs[cellI] - ctrs[0]);
180  vec /= mag(vec);
181 
182  if (mag(vec & vec10) < 0.9)
183  {
184  // ctrs[cellI] not in line with n
185  otherCellI = cellI;
186 
187  break;
188  }
189  }
190 
191  if (otherCellI == -1)
192  {
193  // Cannot find cell to make decent angle with cell0-cell1 vector.
194  // Note: what to do here? All cells (almost) in one line. Maybe 1D case?
195  return -1;
196  }
197 
198  plane cellPlane(ctrs[0], ctrs[1], ctrs[otherCellI]);
199 
200 
201  forAll(ctrs, cellI)
202  {
203  const labelList& cEdges = mesh.cellEdges()[cellI];
204 
205  scalar minLen = GREAT;
206 
207  forAll(cEdges, i)
208  {
209  minLen = min(minLen, mesh.edges()[cEdges[i]].mag(mesh.points()));
210  }
211 
212  if (cellPlane.distance(ctrs[cellI]) > 1e-6*minLen)
213  {
214  // Centres not in plane
215  return -1;
216  }
217  }
218 
219  label axisIndex = axis(cellPlane.normal());
220 
221  if (axisIndex == -1)
222  {
223  return axisIndex;
224  }
225 
226 
227  const polyBoundaryMesh& patches = mesh.boundaryMesh();
228 
229 
230  //
231  // 2. No edges without points on boundary
232  //
233 
234  // Mark boundary points
235  boolList boundaryPoint(mesh.points().size(), false);
236 
237  forAll(patches, patchI)
238  {
239  const polyPatch& patch = patches[patchI];
240 
241  forAll(patch, patchFaceI)
242  {
243  const face& f = patch[patchFaceI];
244 
245  forAll(f, fp)
246  {
247  boundaryPoint[f[fp]] = true;
248  }
249  }
250  }
251 
252 
253  const edgeList& edges = mesh.edges();
254 
255  forAll(edges, edgeI)
256  {
257  const edge& e = edges[edgeI];
258 
259  if (!boundaryPoint[e.start()] && !boundaryPoint[e.end()])
260  {
261  // Edge has no point on boundary.
262  return -1;
263  }
264  }
265 
266 
267  // 3. For all non-wedge patches: all faces either perp or aligned with
268  // cell-plane normal. (wedge patches already checked upon construction)
269 
270  forAll(patches, patchI)
271  {
272  const polyPatch& patch = patches[patchI];
273 
274  if (!isA<wedgePolyPatch>(patch))
275  {
276  const vectorField& n = patch.faceAreas();
277 
278  const scalarField cosAngle(mag(n/mag(n) & cellPlane.normal()));
279 
280  if (mag(min(cosAngle) - max(cosAngle)) > 1e-6)
281  {
282  // cosAngle should be either ~1 over all faces (2D front and
283  // back) or ~0 (all other patches perp to 2D)
284  return -1;
285  }
286  }
287  }
288 
289  return axisIndex;
290 }
