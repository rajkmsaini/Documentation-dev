87 void Foam::RBD::joints::Ryxz::jcalc
88 (
89  joint::XSvc& J,
90  const scalarField& q,
91  const scalarField& qDot
92 ) const
93 {
94  vector qj(q.block<vector>(qIndex_));
95 
96  scalar s0 = sin(qj.x());
97  scalar c0 = cos(qj.x());
98  scalar s1 = sin(qj.y());
99  scalar c1 = cos(qj.y());
100  scalar s2 = sin(qj.z());
101  scalar c2 = cos(qj.z());
102 
103  J.X.E() = tensor
104  (
105  c2*c0 + s2*s1*s0, s2*c1, -c2*s0 + s2*s1*c0,
106  -s2*c0 + c2*s1*s0, c2*c1, s2*s0 + c2*s1*c0,
107  c1*s0, -s1, c1*c0
108  );
109  J.X.r() = Zero;
110 
111  J.S = Zero;
112  J.S.xx() = s2*c1;
113  J.S.xy() = c2;
114  J.S.yx() = c2*c1;
115  J.S.yy() = -s2;
116  J.S.zx() = -s1;
117  J.S.zz() = 1;
118 
119  vector qDotj(qDot.block<vector>(qIndex_));
120  J.v = J.S & qDotj;
121 
122  J.c = spatialVector
123  (
124  c2*c1*qDotj.z()*qDotj.x()
125  - s2*s1*qDotj.y()*qDotj.x()
126  - s2*qDotj.z()*qDotj.y(),
127 
128  -s2*c1*qDotj.z()*qDotj.x()
129  - c2*s1*qDotj.y()*qDotj.x()
130  - c2*qDotj.z()*qDotj.y(),
131 
132  -c1*qDotj.y()*qDotj.x(),
133 
134  0,
135  0,
136  0
137  );
138 }
