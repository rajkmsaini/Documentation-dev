134 void Foam::fv::pressureGradientExplicitSource::correct(volVectorField& U)
135 {
136  const scalarField& rAU = invAPtr_().internalField();
137 
138  // Integrate flow variables over cell set
139  scalar magUbarAve = 0.0;
140  scalar rAUave = 0.0;
141  const scalarField& cv = mesh_.V();
142  forAll(cells_, i)
143  {
144  label cellI = cells_[i];
145  scalar volCell = cv[cellI];
146  magUbarAve += (flowDir_ & U[cellI])*volCell;
147  rAUave += rAU[cellI]*volCell;
148  }
149 
150  // Collect across all processors
151  reduce(magUbarAve, sumOp<scalar>());
152  reduce(rAUave, sumOp<scalar>());
153 
154  // Volume averages
155  magUbarAve /= V_;
156  rAUave /= V_;
157 
158  // Calculate the pressure gradient increment needed to adjust the average
159  // flow-rate to the desired value
160  dGradP_ = (mag(Ubar_) - magUbarAve)/rAUave;
161 
162  // Apply correction to velocity field
163  forAll(cells_, i)
164  {
165  label cellI = cells_[i];
166  U[cellI] += flowDir_*rAU[cellI]*dGradP_;
167  }
168 
169  scalar gradP = gradP0_ + dGradP_;
170 
171  Info<< "Pressure gradient source: uncorrected Ubar = " << magUbarAve
172  << ", pressure gradient = " << gradP << endl;
173 
174  writeProps(gradP);
175 }
