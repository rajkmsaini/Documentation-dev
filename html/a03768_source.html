97  );
98 
99  PtrList<volScalarField>& Yi = Y();
100 
101  forAll(Yi, i)
102  {
103  if (i != inertIndex_)
104  {
105  Yt += Yi[i];
106  }
107  }
108 
109  if (inertIndex_ != -1)
