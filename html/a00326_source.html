72 Foam::septernion
73 Foam::solidBodyMotionFunctions::rotatingMotion::transformation() const
74 {
75  scalar t = time_.value();
76 
77  // Rotation around axis
78  scalar angle = omega_->integrate(0, t);
79 
80  quaternion R(axis_, angle);
81  septernion TR(septernion(-origin_)*R*septernion(origin_));
82 
83  DebugInFunction << "Time = " << t << " transformation: " << TR << endl;
84 
85  return TR;
