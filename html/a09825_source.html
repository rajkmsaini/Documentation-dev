32 void Foam::RBD::rigidBodyModel::applyRestraints(Field<spatialVector>& fx) const
33 {
34  if (restraints_.empty())

36  return;
37  }
38 
39  forAll(restraints_, ri)
40  {
41  DebugInfo << "Restraint " << restraints_[ri].name();

49 void Foam::RBD::rigidBodyModel::forwardDynamics
50 (
51  rigidBodyModelState& state,
52  const scalarField& tau,
53  const Field<spatialVector>& fx
54 ) const
55 {
56  const scalarField& q = state.q();
57  const scalarField& qDot = state.qDot();
58  scalarField& qDdot = state.qDdot();
59 
60  DebugInFunction
61  << "q = " << q << nl
62  << "qDot = " << qDot << nl
63  << "tau = " << tau << endl;
64 
65  // Joint state returned by jcalc
66  joint::XSvc J;
67 
68  v_[0] = Zero;
69 
70  for (label i=1; i<nBodies(); i++)
71  {
72  const joint& jnt = joints()[i];
73  jnt.jcalc(J, q, qDot);
74 
75  S_[i] = J.S;
76  S1_[i] = J.S1;
77 
78  Xlambda_[i] = J.X & XT_[i];
79 
80  const label lambdai = lambda_[i];
81 
82  if (lambdai != 0)
83  {
84  X0_[i] = Xlambda_[i] & X0_[lambdai];
85  }
86  else
87  {
88  X0_[i] = Xlambda_[i];
89  }
90 
91  v_[i] = (Xlambda_[i] & v_[lambdai]) + J.v;
92  c_[i] = J.c + (v_[i] ^ J.v);
93  IA_[i] = I(i);
94  pA_[i] = v_[i] ^* (I(i) & v_[i]);
95 
96  if (fx.size())
97  {
98  pA_[i] -= *X0_[i] & fx[i];
99  }
100  }
101 
102  for (label i=nBodies()-1; i>0; i--)
103  {
104  const joint& jnt = joints()[i];
105  const label qi = jnt.qIndex();
106 
107  if (jnt.nDoF() == 1)
108  {
109  U1_[i] = IA_[i] & S1_[i];
110  Dinv_[i].xx() = 1/(S1_[i] && U1_[i]);
111  u_[i].x() = tau[qi] - (S1_[i] && pA_[i]);
112 
113  const label lambdai = lambda_[i];
114 
115  if (lambdai != 0)
116  {
117  const spatialTensor Ia
118  (
119  IA_[i] - (U1_[i]*(Dinv_[i].xx()*U1_[i]))
120  );
121 
122  const spatialVector pa
123  (
124  pA_[i] + (Ia & c_[i]) + U1_[i]*(Dinv_[i].xx()*u_[i].x())
125  );
126 
127  IA_[lambdai] +=
128  spatialTensor(Xlambda_[i].T())
129  & Ia
130  & spatialTensor(Xlambda_[i]);
131 
132  pA_[lambdai] += Xlambda_[i].T() & pa;
133  }
134  }
135  else
136  {
137  U_[i] = IA_[i] & S_[i];
138  Dinv_[i] = (S_[i].T() & U_[i]).inv();
139 
140  u_[i] = tau.block<vector>(qi) - (S_[i].T() & pA_[i]);
141 
142  const label lambdai = lambda_[i];
143 
144  if (lambdai != 0)
145  {
146  spatialTensor Ia
147  (
148  IA_[i]
149  - (U_[i] & Dinv_[i] & U_[i].T())
150  );
151 
152  spatialVector pa
153  (
154  pA_[i]
155  + (Ia & c_[i])
156  + (U_[i] & Dinv_[i] & u_[i])
157  );
158 
159  IA_[lambdai] +=
160  spatialTensor(Xlambda_[i].T())
161  & Ia
162  & spatialTensor(Xlambda_[i]);
163 
164  pA_[lambdai] += Xlambda_[i].T() & pa;
165  }
166  }
167  }
168 
169  a_[0] = spatialVector(Zero, -g_);
170 
171  for (label i=1; i<nBodies(); i++)
172  {
173  const joint& jnt = joints()[i];
174  const label qi = jnt.qIndex();
175 
176  a_[i] = (Xlambda_[i] & a_[lambda_[i]]) + c_[i];
177 
178  if (jnt.nDoF() == 1)
179  {
180  qDdot[qi] = Dinv_[i].xx()*(u_[i].x() - (U1_[i] && a_[i]));
181  a_[i] += S1_[i]*qDdot[qi];
182  }
183  else
184  {
185  vector qDdoti(Dinv_[i] & (u_[i] - (U_[i].T() & a_[i])));
186 
187  // Need to add mutable "block<vector>" to Field
188  qDdot[qi] = qDdoti.x();
189  qDdot[qi+1] = qDdoti.y();
190  qDdot[qi+2] = qDdoti.z();
191 
192  a_[i] += (S_[i] & qDdoti);
193  }
194  }
195 
196  DebugInfo
197  << "qDdot = " << qDdot << nl
198  << "a = " << a_ << endl;
199 }
