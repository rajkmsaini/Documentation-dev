113 void Foam::cloudInfo::write()
114 {
115  if (active_)
116  {
117  functionObjectFile::write();
118 
119  forAll(names(), i)
120  {
121  const word& cloudName = names()[i];
122 
123  const kinematicCloud& cloud =
124  obr_.lookupObject<kinematicCloud>(cloudName);
125 
126  label nParcels = returnReduce(cloud.nParcels(), sumOp<label>());
127  scalar massInSystem =
128  returnReduce(cloud.massInSystem(), sumOp<scalar>());
129 
130  if (Pstream::master())
131  {
132  writeTime(file(i));
133  file(i)
134  << token::TAB
135  << nParcels << token::TAB
136  << massInSystem << endl;
137  }
138  }
139  }
140 }
