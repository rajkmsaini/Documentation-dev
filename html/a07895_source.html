97 const HashTable<dimensionedScalar>& unitSet()
98 {
99  if (!unitSetPtr_)
100  {
101  const dictionary& dict = dimensionSystems();
102 
103  if (!dict.found("unitSet"))
104  {
105  FatalIOErrorInFunction(dict)
106  << "Cannot find unitSet in dictionary " << dict.name()
107  << exit(FatalIOError);
108  }
109 
110  const word unitSetCoeffs(word(dict.lookup("unitSet")) + "Coeffs");
111 
112  if (!dict.found(unitSetCoeffs))
113  {
114  FatalIOErrorInFunction(dict)
115  << "Cannot find " << unitSetCoeffs << " in dictionary "
116  << dict.name() << exit(FatalIOError);
117  }
118 
119  const dictionary& unitDict = dict.subDict(unitSetCoeffs);
120 
121  unitSetPtr_ = new HashTable<dimensionedScalar>(unitDict.size());
122 
123  forAllConstIter(dictionary, unitDict, iter)
124  {
125  if (iter().keyword() != "writeUnits")
126  {
127  dimensionedScalar dt;
128  dt.read(iter().stream(), unitDict);
129  bool ok = unitSetPtr_->insert(iter().keyword(), dt);
130  if (!ok)
131  {
132  FatalIOErrorInFunction(dict)
133  << "Duplicate unit " << iter().keyword()
134  << " in DimensionSets dictionary"
135  << exit(FatalIOError);
136  }
137  }
138  }
139 
140  wordList writeUnitNames
141  (
142  unitDict.lookupOrDefault<wordList>
143  (
144  "writeUnits",
145  wordList(0)
146  )
147  );
148 
149  writeUnitSetPtr_ = new dimensionSets(*unitSetPtr_, writeUnitNames);
150 
151  if (writeUnitNames.size() != 0 && writeUnitNames.size() != 7)
152  {
153  FatalIOErrorInFunction(dict)
154  << "Cannot find entry \"writeUnits\" in " << unitDict.name()
155  << " or it is not a wordList of size 7"
156  << exit(FatalIOError);
157  }
158  }
159  return *unitSetPtr_;
160 }

209 Foam::dimensionSets::dimensionSets
210 (
211  const HashTable<dimensionedScalar>& units,
212  const wordList& unitNames
213 )
214 :
215  units_(unitNames.size()),
216  conversion_(unitNames.size()),
217  conversionPivots_(unitNames.size()),
218  valid_(false)
219 {
220  forAll(unitNames, i)
221  {
222  units_.set
223  (
224  i,
225  new dimensionedScalar
226  (
227  units[unitNames[i]]
228  )
229  );
230  }
231 
232  if (unitNames.size() == 7)
233  {
234  valid_ = true;
235 
236  // Determine conversion from basic units to write units
237  for (label rowI = 0; rowI < conversion_.n(); rowI++)
238  {
239  scalar* row = conversion_[rowI];
240 
241  for (label columnI = 0; columnI < conversion_.m(); columnI++)
242  {
243  const dimensionedScalar& dSet = units_[columnI];
244  row[columnI] = dSet.dimensions()[rowI];
245  }
246  }
247  conversionPivots_.setSize(conversion_.n());
248  LUDecompose(conversion_, conversionPivots_);
249  }
250 }
