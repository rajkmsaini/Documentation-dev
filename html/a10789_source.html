153 template<class BasicTurbulenceModel>
154 void dynamicLagrangian<BasicTurbulenceModel>::correct()
155 {
156  if (!this->turbulence_)
157  {
158  return;
159  }
160 
161  // Local references
162  const surfaceScalarField& phi = this->phi_;
163  const volVectorField& U = this->U_;
164  fv::options& fvOptions(fv::options::New(this->mesh_));
165 
166  LESeddyViscosity<BasicTurbulenceModel>::correct();
167 
168  tmp<volTensorField> tgradU(fvc::grad(U));
169  const volTensorField& gradU = tgradU();
170 
171  volSymmTensorField S(dev(symm(gradU)));
172  volScalarField magS(mag(S));
173 
174  volVectorField Uf(filter_(U));
175  volSymmTensorField Sf(dev(symm(fvc::grad(Uf))));
176  volScalarField magSf(mag(Sf));
177 
178  volSymmTensorField L(dev(filter_(sqr(U)) - (sqr(filter_(U)))));
179  volSymmTensorField M
180  (
181  2.0*sqr(this->delta())*(filter_(magS*S) - 4.0*magSf*Sf)
182  );
183 
184  volScalarField invT
185  (
186  (1.0/(theta_.value()*this->delta()))*pow(flm_*fmm_, 1.0/8.0)
187  );
188 
189  volScalarField LM(L && M);
190 
191  fvScalarMatrix flmEqn
192  (
193  fvm::ddt(flm_)
194  + fvm::div(phi, flm_)
195  ==
196  invT*LM
197  - fvm::Sp(invT, flm_)
198  + fvOptions(flm_)
199  );
200 
201  flmEqn.relax();
202  fvOptions.constrain(flmEqn);
203  flmEqn.solve();
204  fvOptions.correct(flm_);
205  bound(flm_, flm0_);
206 
207  volScalarField MM(M && M);
208 
209  fvScalarMatrix fmmEqn
210  (
211  fvm::ddt(fmm_)
212  + fvm::div(phi, fmm_)
213  ==
214  invT*MM
215  - fvm::Sp(invT, fmm_)
216  + fvOptions(fmm_)
217  );
218 
219  fmmEqn.relax();
220  fvOptions.constrain(fmmEqn);
221  fmmEqn.solve();
222  fvOptions.correct(fmm_);
223  bound(fmm_, fmm0_);
224 
225  correctNut(gradU);
226 }
