404 template<class BasicTurbulenceModel>
405 void SpalartAllmarasDES<BasicTurbulenceModel>::correct()
406 {
407  if (!this->turbulence_)
408  {
409  return;
410  }
411 
412  // Local references
413  const alphaField& alpha = this->alpha_;
414  const rhoField& rho = this->rho_;
415  const surfaceScalarField& alphaRhoPhi = this->alphaRhoPhi_;
416  const volVectorField& U = this->U_;
417  fv::options& fvOptions(fv::options::New(this->mesh_));
418 
419  LESeddyViscosity<BasicTurbulenceModel>::correct();
420 
421  const volScalarField chi(this->chi());
422  const volScalarField fv1(this->fv1(chi));
423 
424  tmp<volTensorField> tgradU = fvc::grad(U);
425  const volScalarField Omega(this->Omega(tgradU()));
426  const volScalarField dTilda(this->dTilda(chi, fv1, tgradU()));
427  const volScalarField Stilda(this->Stilda(chi, fv1, Omega, dTilda));
428 
429  tmp<fvScalarMatrix> nuTildaEqn
430  (
431  fvm::ddt(alpha, rho, nuTilda_)
432  + fvm::div(alphaRhoPhi, nuTilda_)
433  - fvm::laplacian(alpha*rho*DnuTildaEff(), nuTilda_)
434  - Cb2_/sigmaNut_*alpha*rho*magSqr(fvc::grad(nuTilda_))
435  ==
436  Cb1_*alpha*rho*Stilda*nuTilda_
437  - fvm::Sp
438  (
439  Cw1_*alpha*rho*fw(Stilda, dTilda)*nuTilda_/sqr(dTilda),
440  nuTilda_
441  )
442  + fvOptions(alpha, rho, nuTilda_)
443  );
444 
445  nuTildaEqn().relax();
446  fvOptions.constrain(nuTildaEqn());
447  solve(nuTildaEqn);
448  fvOptions.correct(nuTilda_);
449  bound(nuTilda_, dimensionedScalar("0", nuTilda_.dimensions(), 0.0));
450  nuTilda_.correctBoundaryConditions();
451 
452  correctNut(fv1);
453 }
