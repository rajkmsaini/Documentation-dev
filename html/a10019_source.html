99 )
100 {
101  sixDoFRigidBodyMotionConstraint::read(sDoFRBMCDict);
102 
103  centreOfRotation_ = sDoFRBMCCoeffs_.lookupOrDefault
104  (
105  "centreOfRotation",
106  motion_.initialCentreOfMass()
107  );
108 
109  sDoFRBMCCoeffs_.lookup("direction") >> direction_;
110 
111  scalar magDir(mag(direction_));
112 
113  if (magDir > VSMALL)
114  {
115  direction_ /= magDir;
116  }
117  else

119  FatalErrorInFunction

121  << abort(FatalError);
122  }
123
