65 inline Foam::scalar Foam::TroeFallOffFunction::operator()
66 (
67  const scalar T,
68  const scalar Pr
69 ) const
70 {
71  scalar logFcent = log10
72  (
73  max
74  (
75  (1 - alpha_)*exp(-T/Tsss_) + alpha_*exp(-T/Ts_) + exp(-Tss_/T),
76  SMALL
77  )
78  );
79 
80  scalar c = -0.4 - 0.67*logFcent;
81  static const scalar d = 0.14;
82  scalar n = 0.75 - 1.27*logFcent;
83 
84  scalar logPr = log10(max(Pr, SMALL));
85  return pow(10.0, logFcent/(1.0 + sqr((logPr + c)/(n - d*(logPr + c)))));
86 }
