135 template<class OutputFilter>
136 bool Foam::OutputFilterFunctionObject<OutputFilter>::adjustTimeStep()
137 {
138  if
139  (
140  active()
141  && outputControl_.outputControl()
142  == outputFilterOutputControl::ocAdjustableRunTime
143  )
144  {
145  const label outputTimeIndex = outputControl_.outputTimeLastDump();
146  const scalar writeInterval = outputControl_.writeInterval();
147 
148  scalar timeToNextWrite = max
149  (
150  0.0,
151  (outputTimeIndex + 1)*writeInterval
152  - (time_.value() - time_.startTime().value())
153  );
154 
155  scalar deltaT = time_.deltaTValue();
156 
157  scalar nSteps = timeToNextWrite/deltaT - SMALL;
158 
159  // functionObjects modify deltaT within nStepsToStartTimeChange
160  // NOTE: Potential problems arise if two function objects dump within
161  // the same interval
162  if (nSteps < nStepsToStartTimeChange_)
163  {
164  label nStepsToNextWrite = label(nSteps) + 1;
165 
166  scalar newDeltaT = timeToNextWrite/nStepsToNextWrite;
167 
168  // Adjust time step
169  if (newDeltaT < deltaT)
170  {
171  deltaT = max(newDeltaT, 0.2*deltaT);
172  const_cast<Time&>(time_).setDeltaT(deltaT, false);
173  }
174  }
175  }
176 
177  return true;
178 }
