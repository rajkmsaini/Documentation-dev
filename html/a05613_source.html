99 void Foam::translatingWallVelocityFvPatchVectorField::updateCoeffs()
100 {
101  if (updated())
102  {
103  return;
104  }
105 
106  const scalar t = this->db().time().timeOutputValue();
107  const vector U = U_->value(t);
108 
109  // Remove the component of U normal to the wall in case the wall is not flat
110  const vectorField n(patch().nf());
111  vectorField::operator=(U - n*(n & U));
112 
113  fixedValueFvPatchVectorField::updateCoeffs();
114 }
