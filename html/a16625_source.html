69 template<class ThermoType>
70 const ThermoType& Foam::egrMixture<ThermoType>::mixture
71 (
72  const scalar ft,
73  const scalar b,
74  const scalar egr
75 ) const
76 {
77  if (ft < 0.0001)
78  {
79  return oxidant_;
80  }
81  else
82  {
83 
84  scalar fu = b*ft + (1.0 - b)*fres(ft, stoicRatio().value());
85  scalar ox = 1 - ft - (ft - fu)*stoicRatio().value();
86 
87  fu *= (1.0 - egr);
88  ox *= (1.0 - egr);
89 
90  scalar pr = 1 - fu - ox;
91 
92  mixture_ = fu/fuel_.W()*fuel_;
93  mixture_ += ox/oxidant_.W()*oxidant_;
94  mixture_ += pr/products_.W()*products_;
95 
96  return mixture_;
97  }
98 }
