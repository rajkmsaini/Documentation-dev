157 void Foam::sampledSurfaces::write()
158 {
159  if (size())
160  {
161  // Finalize surfaces, merge points etc.
162  update();
163 
164  const label nFields = classifyFields();
165 
166  if (Pstream::master())
167  {
168  if (debug)
169  {
170  Pout<< "Creating directory "
171  << outputPath_/mesh_.time().timeName() << nl << endl;
172 
173  }
174 
175  mkDir(outputPath_/mesh_.time().timeName());
176  }
177 
178  // Write geometry first if required,
179  // or when no fields would otherwise be written
180  if (nFields == 0 || formatter_->separateGeometry())
181  {
182  writeGeometry();
183  }
184 
185  const IOobjectList objects(mesh_, mesh_.time().timeName());
186 
187  sampleAndWrite<volScalarField>(objects);
188  sampleAndWrite<volVectorField>(objects);
189  sampleAndWrite<volSphericalTensorField>(objects);
190  sampleAndWrite<volSymmTensorField>(objects);
191  sampleAndWrite<volTensorField>(objects);
192 
193  sampleAndWrite<surfaceScalarField>(objects);
194  sampleAndWrite<surfaceVectorField>(objects);
195  sampleAndWrite<surfaceSphericalTensorField>(objects);
196  sampleAndWrite<surfaceSymmTensorField>(objects);
197  sampleAndWrite<surfaceTensorField>(objects);
198  }

202 void Foam::sampledSurfaces::read(const dictionary& dict)
203 {
204  bool surfacesFound = dict.found("surfaces");
205 
206  if (surfacesFound)
207  {
208  dict.lookup("fields") >> fieldSelection_;
209 
210  dict.lookup("interpolationScheme") >> interpolationScheme_;
211  const word writeType(dict.lookup("surfaceFormat"));
212 
213  // Define the surface formatter
214  // Optionally defined extra controls for the output formats
215  formatter_ = surfaceWriter::New
216  (
217  writeType,
218  dict.subOrEmptyDict("formatOptions").subOrEmptyDict(writeType)
219  );
220 
221  PtrList<sampledSurface> newList
222  (
223  dict.lookup("surfaces"),
224  sampledSurface::iNew(mesh_)
225  );
226  transfer(newList);
227 
228  if (Pstream::parRun())
229  {
230  mergeList_.setSize(size());
231  }
232 
233  // Ensure all surfaces and merge information are expired
234  expire();
235 
236  if (this->size())
237  {
238  Info<< "Reading surface description:" << nl;
239  forAll(*this, surfI)
240  {
241  Info<< " " << operator[](surfI).name() << nl;
242  }
243  Info<< endl;
244  }
245  }
246 
247  if (Pstream::master() && debug)
248  {
249  Pout<< "sample fields:" << fieldSelection_ << nl
250  << "sample surfaces:" << nl << "(" << nl;
251 
252  forAll(*this, surfI)
253  {
254  Pout<< " " << operator[](surfI) << endl;
255  }
256  Pout<< ")" << endl;
257  }
258 }

298 bool Foam::sampledSurfaces::expire()

301

307  }
308

321 bool Foam::sampledSurfaces::update()
322 {
323  bool updated = false;
324 
325  if (!needsUpdate())
326  {
327  return updated;
328  }
329 
330  // Serial: quick and easy, no merging required
331  if (!Pstream::parRun())
332  {
333  forAll(*this, surfI)
334  {
335  if (operator[](surfI).update())
336  {
337  updated = true;
338  }
339  }
340 
341  return updated;
342  }
343 
344  // Dimension as fraction of mesh bounding box
345  scalar mergeDim = mergeTol_ * mesh_.bounds().mag();
346 
347  if (Pstream::master() && debug)
348  {
349  Pout<< nl << "Merging all points within "
350  << mergeDim << " metre" << endl;
351  }
352 
353  forAll(*this, surfI)
354  {
355  sampledSurface& s = operator[](surfI);
356 
357  if (s.update())
358  {
359  updated = true;
360  }
361  else
362  {
363  continue;
364  }
365 
366  PatchTools::gatherAndMerge
367  (
368  mergeDim,
369  primitivePatch
370  (
371  SubList<face>(s.faces(), s.faces().size()),
372  s.points()
373  ),
374  mergeList_[surfI].points,
375  mergeList_[surfI].faces,
376  mergeList_[surfI].pointsMap
377  );
378  }
379 
380  return updated;
381 }
