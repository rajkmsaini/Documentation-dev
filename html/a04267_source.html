105 template<class CombThermoType, class ThermoType>
106 tmp<fvScalarMatrix> singleStepCombustion<CombThermoType, ThermoType>::R
107 (
108  volScalarField& Y
109 ) const
110 {
111  const label specieI = this->thermoPtr_->composition().species()[Y.name()];
112 
113  volScalarField wSpecie
114  (
115  wFuel_*singleMixturePtr_->specieStoichCoeffs()[specieI]
116  );
117 
118  if (semiImplicit_)
119  {
120  const label fNorm = singleMixturePtr_->specieProd()[specieI];
121  const volScalarField fres(singleMixturePtr_->fres(specieI));
122  wSpecie /= max(fNorm*(Y - fres), scalar(1e-2));
123 
124  return -fNorm*wSpecie*fres + fNorm*fvm::Sp(wSpecie, Y);
125  }
126  else
127  {
128  return wSpecie + fvm::Sp(0.0*wSpecie, Y);
129  }
130 }

135 singleStepCombustion<CombThermoType, ThermoType>::Sh() const
136 {
137  const label fuelI = singleMixturePtr_->fuelIndex();
138  volScalarField& YFuel =
139  const_cast<volScalarField&>(this->thermoPtr_->composition().Y(fuelI));
