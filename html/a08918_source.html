50 Foam::wordList Foam::scalarTransport::boundaryTypes() const
51 {
52  const volVectorField& U = mesh_.lookupObject<volVectorField>(UName_);
53 
54  wordList bTypes(U.boundaryField().size());
55 
56  forAll(bTypes, patchI)
57  {
58  const fvPatchField<vector>& pf = U.boundaryField()[patchI];
59  if (isA<fixedValueFvPatchVectorField>(pf))
60  {
61  bTypes[patchI] = fixedValueFvPatchScalarField::typeName;
62  }
63  else
64  {
65  bTypes[patchI] = zeroGradientFvPatchScalarField::typeName;
66  }
67  }
68 
69  return bTypes;
70 }

194 void Foam::scalarTransport::read(const dictionary& dict)
195 {
196  if (active_)
197  {
198  Info<< type() << ":" << nl;
199 
200  phiName_ = dict.lookupOrDefault<word>("phiName", "phi");
201  UName_ = dict.lookupOrDefault<word>("UName", "U");
202  rhoName_ = dict.lookupOrDefault<word>("rhoName", "rho");
203 
204  userDT_ = false;
205  if (dict.readIfPresent("DT", DT_))
206  {
207  userDT_ = true;
208  }
209 
210  dict.lookup("resetOnStartUp") >> resetOnStartUp_;
211 
212  dict.readIfPresent("nCorr", nCorr_);
213 
214  dict.lookup("autoSchemes") >> autoSchemes_;
215 
216  fvOptions_.reset(dict.subDict("fvOptions"));
217  }

221 void Foam::scalarTransport::execute()
222 {
223  if (active_)
224  {
225  Info<< type() << " output:" << endl;
226 
227  const surfaceScalarField& phi =
228  mesh_.lookupObject<surfaceScalarField>(phiName_);
229 
230  // calculate the diffusivity
231  volScalarField DT(this->DT(phi));
232 
233  // set schemes
234  word schemeVar = T_.name();
235  if (autoSchemes_)
236  {
237  schemeVar = UName_;
238  }
239 
240  word divScheme("div(phi," + schemeVar + ")");
241  word laplacianScheme("laplacian(" + DT.name() + "," + schemeVar + ")");
242 
243  // set under-relaxation coeff
244  scalar relaxCoeff = 0.0;
245  if (mesh_.relaxEquation(schemeVar))
246  {
247  relaxCoeff = mesh_.equationRelaxationFactor(schemeVar);
248  }
249 
250  if (phi.dimensions() == dimMass/dimTime)
251  {
252  const volScalarField& rho =
253  mesh_.lookupObject<volScalarField>(rhoName_);
254 
255  // solve
256  for (label i = 0; i <= nCorr_; i++)
257  {
258  fvScalarMatrix TEqn
259  (
260  fvm::ddt(rho, T_)
261  + fvm::div(phi, T_, divScheme)
262  - fvm::laplacian(DT, T_, laplacianScheme)
263  ==
264  fvOptions_(rho, T_)
265  );
266 
267  TEqn.relax(relaxCoeff);
268 
269  fvOptions_.constrain(TEqn);
270 
271  TEqn.solve(mesh_.solverDict(schemeVar));
272  }
273  }
274  else if (phi.dimensions() == dimVolume/dimTime)
275  {
276  // solve
277  for (label i = 0; i <= nCorr_; i++)
278  {
279  fvScalarMatrix TEqn
280  (
281  fvm::ddt(T_)
282  + fvm::div(phi, T_, divScheme)
283  - fvm::laplacian(DT, T_, laplacianScheme)
284  ==
285  fvOptions_(T_)
286  );
287 
288  TEqn.relax(relaxCoeff);
289 
290  fvOptions_.constrain(TEqn);
291 
292  TEqn.solve(mesh_.solverDict(schemeVar));
293  }
294  }
295  else
296  {
297  FatalErrorIn("void Foam::scalarTransport::execute()")
298  << "Incompatible dimensions for phi: " << phi.dimensions() << nl
299  << "Dimensions should be " << dimMass/dimTime << " or "
300  << dimVolume/dimTime << endl;
301  }
302 
303  Info<< endl;
304  }
305 }
