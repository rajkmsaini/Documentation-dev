65 cellZoneSet::cellZoneSet
66 (
67  const polyMesh& mesh,
68  const word& name,
69  readOption r,
70  writeOption w
71 )
72 :
73  cellSet(mesh, name, 1000), // do not read cellSet
74  mesh_(mesh),
75  addressing_(0)
76 {
77  const cellZoneMesh& cellZones = mesh.cellZones();
78  label zoneID = cellZones.findZoneID(name);
79 
80  if
81  (
82  (r == IOobject::MUST_READ)
83  || (r == IOobject::MUST_READ_IF_MODIFIED)
84  || (r == IOobject::READ_IF_PRESENT && zoneID != -1)
85  )
86  {
87  const cellZone& fz = cellZones[zoneID];
88  addressing_ = fz;
89  }

137 void cellZoneSet::invert(const label maxLen)
138 {
139  // Count
140  label n = 0;
141 
142  for (label celli = 0; celli < maxLen; celli++)
143  {
144  if (!found(celli))
145  {
146  n++;
147  }
148  }
149 
150  // Fill
151  addressing_.setSize(n);
152  n = 0;
153 
154  for (label celli = 0; celli < maxLen; celli++)
155  {
156  if (!found(celli))
157  {
158  addressing_[n] = celli;
159  n++;
160  }
161  }
162 
163  updateSet();
164 }

241 bool cellZoneSet::writeObject
242 (
243  IOstream::streamFormat s,
244  IOstream::versionNumber v,
245  IOstream::compressionType c
246 ) const
247 {
248  // Write shadow cellSet
249  word oldTypeName = typeName;
250  const_cast<word&>(type()) = cellSet::typeName;
251  bool ok = cellSet::writeObject(s, v, c);
252  const_cast<word&>(type()) = oldTypeName;
253 
254  // Modify cellZone
255  cellZoneMesh& cellZones = const_cast<polyMesh&>(mesh_).cellZones();
256  label zoneID = cellZones.findZoneID(name());
257 
258  if (zoneID == -1)
259  {
260  zoneID = cellZones.size();
261 
262  cellZones.setSize(zoneID+1);
263  cellZones.set
264  (
265  zoneID,
266  new cellZone
267  (
268  name(),
269  addressing_,
270  zoneID,
271  cellZones
272  )
273  );
274  }
275  else
276  {
277  cellZones[zoneID] = addressing_;
278  }
279  cellZones.clearAddressing();
280 
281  return ok && cellZones.write();
282 }

285 void cellZoneSet::updateMesh(const mapPolyMesh& morphMap)
286 {
287  // cellZone
288  labelList newAddressing(addressing_.size());
289 
290  label n = 0;
291  forAll(addressing_, i)
292  {
293  label celli = addressing_[i];
294  label newCelli = morphMap.reverseCellMap()[celli];
295  if (newCelli >= 0)
296  {
297  newAddressing[n] = newCelli;
298  n++;
299  }
300  }
301  newAddressing.setSize(n);
302 
303  addressing_.transfer(newAddressing);
304
