208 Foam::tmp<Foam::volSymmTensorField>
209 Foam::functionObjects::forces::devRhoReff() const
210 {
211  typedef compressible::turbulenceModel cmpTurbModel;
212  typedef incompressible::turbulenceModel icoTurbModel;
213 
214  if (obr_.foundObject<cmpTurbModel>(cmpTurbModel::propertiesName))
215  {
216  const cmpTurbModel& turb =
217  obr_.lookupObject<cmpTurbModel>(cmpTurbModel::propertiesName);
218 
219  return turb.devRhoReff();
220  }
221  else if (obr_.foundObject<icoTurbModel>(icoTurbModel::propertiesName))
222  {
223  const incompressible::turbulenceModel& turb =
224  obr_.lookupObject<icoTurbModel>(icoTurbModel::propertiesName);
225 
226  return rho()*turb.devReff();
227  }
228  else if (obr_.foundObject<fluidThermo>(fluidThermo::typeName))
229  {
230  const fluidThermo& thermo =
231  obr_.lookupObject<fluidThermo>(fluidThermo::typeName);
232 
233  const volVectorField& U = obr_.lookupObject<volVectorField>(UName_);
234 
235  return -thermo.mu()*dev(twoSymm(fvc::grad(U)));
236  }
237  else if
238  (
239  obr_.foundObject<transportModel>("transportProperties")
240  )
241  {
242  const transportModel& laminarT =
243  obr_.lookupObject<transportModel>("transportProperties");
244 
245  const volVectorField& U = obr_.lookupObject<volVectorField>(UName_);
246 
247  return -rho()*laminarT.nu()*dev(twoSymm(fvc::grad(U)));
248  }
249  else if (obr_.foundObject<dictionary>("transportProperties"))
250  {
251  const dictionary& transportProperties =
252  obr_.lookupObject<dictionary>("transportProperties");
253 
254  dimensionedScalar nu(transportProperties.lookup("nu"));
255 
256  const volVectorField& U = obr_.lookupObject<volVectorField>(UName_);
257 
258  return -rho()*nu*dev(twoSymm(fvc::grad(U)));
259  }
260  else
261  {
262  FatalErrorInFunction
263  << "No valid model for viscous stress calculation"
264  << exit(FatalError);
265 
266  return volSymmTensorField::null();
267  }
268 }

358 void Foam::functionObjects::forces::applyBins
359 (
360  const vectorField& Md,
361  const vectorField& fN,
362  const vectorField& fT,
363  const vectorField& fP,
364  const vectorField& d
365 )
366 {
367  if (nBin_ == 1)
368  {
369  force_[0][0] += sum(fN);
370  force_[1][0] += sum(fT);
371  force_[2][0] += sum(fP);
372  moment_[0][0] += sum(Md^fN);
373  moment_[1][0] += sum(Md^fT);
374  moment_[2][0] += sum(Md^fP);
375  }
376  else
377  {
378  scalarField dd((d & binDir_) - binMin_);
379 
380  forAll(dd, i)
381  {
382  label bini = min(max(floor(dd[i]/binDx_), 0), force_[0].size() - 1);
383 
384  force_[0][bini] += fN[i];
385  force_[1][bini] += fT[i];
386  force_[2][bini] += fP[i];
387  moment_[0][bini] += Md[i]^fN[i];
388  moment_[1][bini] += Md[i]^fT[i];
389  moment_[2][bini] += Md[i]^fP[i];
390  }
391  }

446  }

631 void Foam::functionObjects::forces::read(const dictionary& dict)
632 {
633  initialised_ = false;
634 
635  log_ = dict.lookupOrDefault<Switch>("log", false);
636 
637  if (log_) Info<< type() << " " << name_ << ":" << nl;
638 
639  directForceDensity_ = dict.lookupOrDefault("directForceDensity", false);
640 
641  const fvMesh& mesh = refCast<const fvMesh>(obr_);
642  const polyBoundaryMesh& pbm = mesh.boundaryMesh();
643 
644  patchSet_ = pbm.patchSet(wordReList(dict.lookup("patches")));
645 
646  if (directForceDensity_)
647  {
648  // Optional entry for fDName
649  fDName_ = dict.lookupOrDefault<word>("fDName", "fD");
650  }
651  else
652  {
653  // Optional entries U and p
654  pName_ = dict.lookupOrDefault<word>("pName", "p");
655  UName_ = dict.lookupOrDefault<word>("UName", "U");
656  rhoName_ = dict.lookupOrDefault<word>("rhoName", "rho");
657 
658  // Reference density needed for incompressible calculations
659  rhoRef_ = readScalar(dict.lookup("rhoInf"));
660 
661  // Reference pressure, 0 by default
662  pRef_ = dict.lookupOrDefault<scalar>("pRef", 0.0);
663  }
664 
665  coordSys_.clear();
666 
667  // Centre of rotation for moment calculations
668  // specified directly, from coordinate system, or implicitly (0 0 0)
669  if (!dict.readIfPresent<point>("CofR", coordSys_.origin()))
670  {
671  coordSys_ = coordinateSystem(obr_, dict);
672  localSystem_ = true;
673  }
674 
675  dict.readIfPresent("porosity", porosity_);
676  if (porosity_)
677  {
678  if (log_) Info<< " Including porosity effects" << endl;
679  }
680  else
681  {
682  if (log_) Info<< " Not including porosity effects" << endl;
683  }
684 
685  if (dict.found("binData"))
686  {
687  const dictionary& binDict(dict.subDict("binData"));
688  binDict.lookup("nBin") >> nBin_;
689 
690  if (nBin_ < 0)
691  {
692  FatalIOErrorInFunction(dict)
693  << "Number of bins (nBin) must be zero or greater"
694  << exit(FatalIOError);
695  }
696  else if ((nBin_ == 0) || (nBin_ == 1))
697  {
698  nBin_ = 1;
699  forAll(force_, i)
700  {
701  force_[i].setSize(1);
702  moment_[i].setSize(1);
703  }
704  }
705 
706  if (nBin_ > 1)
707  {
708  binDict.lookup("direction") >> binDir_;
709  binDir_ /= mag(binDir_);
710 
711  binMin_ = GREAT;
712  scalar binMax = -GREAT;
713  forAllConstIter(labelHashSet, patchSet_, iter)
714  {
715  label patchi = iter.key();
716  const polyPatch& pp = pbm[patchi];
717  scalarField d(pp.faceCentres() & binDir_);
718  binMin_ = min(min(d), binMin_);
719  binMax = max(max(d), binMax);
720  }
721  reduce(binMin_, minOp<scalar>());
722  reduce(binMax, maxOp<scalar>());
723 
724  // slightly boost binMax so that region of interest is fully
725  // within bounds
726  binMax = 1.0001*(binMax - binMin_) + binMin_;
727 
728  binDx_ = (binMax - binMin_)/scalar(nBin_);
729 
730  // create the bin points used for writing
731  binPoints_.setSize(nBin_);
732  forAll(binPoints_, i)
733  {
734  binPoints_[i] = (i + 0.5)*binDir_*binDx_;
735  }
736 
737  binDict.lookup("cumulative") >> binCumulative_;
738 
739  // allocate storage for forces and moments
740  forAll(force_, i)
741  {
742  force_[i].setSize(nBin_);
743  moment_[i].setSize(nBin_);
744  }
745  }
746  }
747 
748  if (nBin_ == 1)
749  {
750  // allocate storage for forces and moments
751  force_[0].setSize(1);

790 void Foam::functionObjects::forces::calcForcesMoment()
791 {
792  initialise();
793 
794  force_[0] = Zero;
795  force_[1] = Zero;
796  force_[2] = Zero;
797 
798  moment_[0] = Zero;
799  moment_[1] = Zero;
800  moment_[2] = Zero;
801 
802  if (directForceDensity_)
803  {
804  const volVectorField& fD = obr_.lookupObject<volVectorField>(fDName_);
805 
806  const fvMesh& mesh = fD.mesh();
807 
808  const surfaceVectorField::Boundary& Sfb =
809  mesh.Sf().boundaryField();
810 
811  forAllConstIter(labelHashSet, patchSet_, iter)
812  {
813  label patchi = iter.key();
814 
815  vectorField Md
816  (
817  mesh.C().boundaryField()[patchi] - coordSys_.origin()
818  );
819 
820  scalarField sA(mag(Sfb[patchi]));
821 
822  // Normal force = surfaceUnitNormal*(surfaceNormal & forceDensity)
823  vectorField fN
824  (
825  Sfb[patchi]/sA
826  *(
827  Sfb[patchi] & fD.boundaryField()[patchi]
828  )
829  );
830 
831  // Tangential force (total force minus normal fN)
832  vectorField fT(sA*fD.boundaryField()[patchi] - fN);
833 
834  //- Porous force
835  vectorField fP(Md.size(), Zero);
836 
837  applyBins(Md, fN, fT, fP, mesh.C().boundaryField()[patchi]);
838  }
839  }
840  else
841  {
842  const volVectorField& U = obr_.lookupObject<volVectorField>(UName_);
843  const volScalarField& p = obr_.lookupObject<volScalarField>(pName_);
844 
845  const fvMesh& mesh = U.mesh();
846 
847  const surfaceVectorField::Boundary& Sfb =
848  mesh.Sf().boundaryField();
849 
850  tmp<volSymmTensorField> tdevRhoReff = devRhoReff();
851  const volSymmTensorField::Boundary& devRhoReffb
852  = tdevRhoReff().boundaryField();
853 
854  // Scale pRef by density for incompressible simulations
855  scalar pRef = pRef_/rho(p);
856 
857  forAllConstIter(labelHashSet, patchSet_, iter)
858  {
859  label patchi = iter.key();
860 
861  vectorField Md
862  (
863  mesh.C().boundaryField()[patchi] - coordSys_.origin()
864  );
865 
866  vectorField fN
867  (
868  rho(p)*Sfb[patchi]*(p.boundaryField()[patchi] - pRef)
869  );
870 
871  vectorField fT(Sfb[patchi] & devRhoReffb[patchi]);
872 
873  vectorField fP(Md.size(), Zero);
874 
875  applyBins(Md, fN, fT, fP, mesh.C().boundaryField()[patchi]);
876  }
877  }
878 
879  if (porosity_)
880  {
881  const volVectorField& U = obr_.lookupObject<volVectorField>(UName_);
882  const volScalarField rho(this->rho());
883  const volScalarField mu(this->mu());
884 
885  const fvMesh& mesh = U.mesh();
886 
887  const HashTable<const porosityModel*> models =
888  obr_.lookupClass<porosityModel>();
889 
890  if (models.empty())
891  {
892  WarningInFunction
893  << "Porosity effects requested, but no porosity models found "
894  << "in the database"
895  << endl;
896  }
897 
898  forAllConstIter(HashTable<const porosityModel*>, models, iter)
899  {
900  // non-const access required if mesh is changing
901  porosityModel& pm = const_cast<porosityModel&>(*iter());
902 
903  vectorField fPTot(pm.force(U, rho, mu));
904 
905  const labelList& cellZoneIDs = pm.cellZoneIDs();
906 
907  forAll(cellZoneIDs, i)
908  {
909  label zoneI = cellZoneIDs[i];
910  const cellZone& cZone = mesh.cellZones()[zoneI];
911 
912  const vectorField d(mesh.C(), cZone);
913  const vectorField fP(fPTot, cZone);
914  const vectorField Md(d - coordSys_.origin());
915 
916  const vectorField fDummy(Md.size(), Zero);
917 
918  applyBins(Md, fDummy, fDummy, fP, d);
919  }
920  }
921  }
922 
923  Pstream::listCombineGather(force_, plusEqOp<vectorField>());
924  Pstream::listCombineGather(moment_, plusEqOp<vectorField>());
925  Pstream::listCombineScatter(force_);
926  Pstream::listCombineScatter(moment_);
927 }
