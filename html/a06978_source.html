80 template<class CloudType>
81 Foam::forceSuSp Foam::DistortedSphereDragForce<CloudType>::calcCoupled
82 (
83  const typename CloudType::parcelType& p,
84  const scalar dt,
85  const scalar mass,
86  const scalar Re,
87  const scalar muc
88 ) const
89 {
90  forceSuSp value(Zero, 0.0);
91 
92  // Limit the drop distortion to y=0 (sphere) and y=1 (disk)
93  scalar y = min(max(p.y(), 0), 1);
94 
95  value.Sp() = mass*0.75*muc*CdRe(Re)*(1 + 2.632*y)/(p.rho()*sqr(p.d()));
