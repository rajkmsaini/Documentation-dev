93 bool Foam::functionObjects::turbulenceFields::compressible()
94 {
95  if (obr_.foundObject<compressible::turbulenceModel>(modelName))
96  {
97  return true;
98  }
99  else if (obr_.foundObject<incompressible::turbulenceModel>(modelName))
100  {
101  return false;
102  }
103  else
104  {
105  FatalErrorInFunction
106  << "Turbulence model not found in database, deactivating"
107  << exit(FatalError);
108  }
109 
110  return false;
111 }

161  Info<< type() << " " << name_ << ": ";

178 void Foam::functionObjects::turbulenceFields::execute()
179 {
180  bool comp = compressible();
181 
182  if (comp)
183  {
184  const compressible::turbulenceModel& model =
185  obr_.lookupObject<compressible::turbulenceModel>(modelName);
186 
187  forAllConstIter(wordHashSet, fieldSet_, iter)
188  {
189  const word& f = iter.key();
190  switch (compressibleFieldNames_[f])
191  {
192  case cfK:
193  {
194  processField<scalar>(f, model.k());
195  break;
196  }
197  case cfEpsilon:
198  {
199  processField<scalar>(f, model.epsilon());
200  break;
201  }
202  case cfMut:
203  {
204  processField<scalar>(f, model.mut());
205  break;
206  }
207  case cfMuEff:
208  {
209  processField<scalar>(f, model.muEff());
210  break;
211  }
212  case cfAlphat:
213  {
214  processField<scalar>(f, model.alphat());
215  break;
216  }
217  case cfAlphaEff:
218  {
219  processField<scalar>(f, model.alphaEff());
220  break;
221  }
222  case cfR:
223  {
224  processField<symmTensor>(f, model.R());
225  break;
226  }
227  case cfDevRhoReff:
228  {
229  processField<symmTensor>(f, model.devRhoReff());
230  break;
231  }
232  default:
233  {
234  FatalErrorInFunction
235  << "Invalid field selection" << abort(FatalError);
236  }
237  }
238  }
239  }
240  else
241  {
242  const incompressible::turbulenceModel& model =
243  obr_.lookupObject<incompressible::turbulenceModel>(modelName);
244 
245  forAllConstIter(wordHashSet, fieldSet_, iter)
246  {
247  const word& f = iter.key();
248  switch (incompressibleFieldNames_[f])
249  {
250  case ifK:
251  {
252  processField<scalar>(f, model.k());
253  break;
254  }
255  case ifEpsilon:
256  {
257  processField<scalar>(f, model.epsilon());
258  break;
259  }
260  case ifNut:
261  {
262  processField<scalar>(f, model.nut());
263  break;
264  }
265  case ifNuEff:
266  {
267  processField<scalar>(f, model.nuEff());
268  break;
269  }
270  case ifR:
271  {
272  processField<symmTensor>(f, model.R());
273  break;
274  }
275  case ifDevReff:

277  processField<symmTensor>(f, model.devReff());
278  break;
279  }
280  default:
281  {
282  FatalErrorInFunction
283  << "Invalid field selection" << abort(FatalError);
284  }
285  }
286  }
287  }
