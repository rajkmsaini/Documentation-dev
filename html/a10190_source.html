128 template<class CompType, class ThermoType>
129 void Foam::chemistryTabulationMethods::ISAT<CompType, ThermoType>::addToMRU
130 (
131  chemPointISAT<CompType, ThermoType>* phi0
132 )
133 {
134  if (maxMRUSize_ > 0 && MRURetrieve_)
135  {
136  // First search if the chemPoint is already in the list
137  bool isInList = false;
138  typename SLList <chemPointISAT<CompType, ThermoType>*>::iterator iter =
139  MRUList_.begin();
140  for ( ; iter != MRUList_.end(); ++iter)
141  {
142  if (iter() == phi0)
143  {
144  isInList = true;
145  break;
146  }
147  }
148  // If it is in the list, then move it to front
149  if (isInList)
150  {
151  if (iter() != MRUList_.first())
152  {
153  // iter hold the position of the element to move
154  MRUList_.remove(iter);
155 
156  // Insert the element in front of the list
157  MRUList_.insert(phi0);
158  }
159  }
160  else // chemPoint not yet in the list, iter is last
161  {
162  if (MRUList_.size() == maxMRUSize_)
163  {
164  if (iter() == MRUList_.last())
165  {
166  MRUList_.remove(iter);
167  MRUList_.insert(phi0);
168  }
169  else
170  {
171  FatalErrorInFunction
172  << "wrong MRUList construction"
173  << exit(FatalError);
174  }
175  }
176  else
177  {
178  MRUList_.insert(phi0);
179  }
180  }
181  }

185 template<class CompType, class ThermoType>
186 void Foam::chemistryTabulationMethods::ISAT<CompType, ThermoType>::calcNewC
187 (
188  chemPointISAT<CompType, ThermoType>* phi0,
189  const scalarField& phiq,
190  scalarField& Rphiq
191 )
192 {
193  label nEqns = this->chemistry_.nEqns(); // Full set of species
194  bool mechRedActive = this->chemistry_.mechRed()->active();
195  Rphiq = phi0->Rphi();
196  scalarField dphi(phiq-phi0->phi());
197  const scalarSquareMatrix& gradientsMatrix = phi0->A();
198  List<label>& completeToSimplified(phi0->completeToSimplifiedIndex());
199 
200  // Rphiq[i] = Rphi0[i]+A(i, j)dphi[j]
201  // where Aij is dRi/dphi_j
202  for (label i=0; i<nEqns-2; i++)
203  {
204  if (mechRedActive)
205  {
206  label si = completeToSimplified[i];
207  // The species is active
208  if (si != -1)
209  {
210  for (label j=0; j<nEqns-2; j++)
211  {
212  label sj = completeToSimplified[j];
213  if (sj != -1)
214  {
215  Rphiq[i] += gradientsMatrix(si, sj)*dphi[j];
216  }
217  }
218  Rphiq[i] +=
219  gradientsMatrix(si, phi0->nActiveSpecies())*dphi[nEqns-2];
220  Rphiq[i] +=
221  gradientsMatrix(si, phi0->nActiveSpecies()+1)*dphi[nEqns-1];
222  // As we use an approximation of A, Rphiq should be checked for
223  // negative values
224  Rphiq[i] = max(0.0,Rphiq[i]);
225  }
226  // The species is not active A(i, j) = I(i, j)
227  else
228  {
229  Rphiq[i] += dphi[i];
230  Rphiq[i] = max(0.0,Rphiq[i]);
231  }
232  }
233  else // Mechanism reduction is not active
234  {
235  for (label j=0; j<nEqns; j++)
236  {
237  Rphiq[i] += gradientsMatrix(i, j)*dphi[j];
238  }
239  // As we use a first order gradient matrix, Rphiq should be checked
240  // for negative values
241  Rphiq[i] = max(0.0,Rphiq[i]);
242  }
243  }
244 }

249 (
250  chemPointISAT<CompType, ThermoType>* phi0,
251  const scalarField& phiq,
252  const scalarField& Rphiq
253 )
254 {
255  // If the pointer to the chemPoint is nullptr, the function stops
256  if (!phi0)
257  {
258  return false;
259  }
260 
261  // Raise a flag when the chemPoint used has been grown more than the
262  // allowed number of time
263  if (!phi0->toRemove() && phi0->nGrowth() > maxGrowth_)
264  {
265  cleaningRequired_ = true;
266  phi0->toRemove() = true;
267  }
268

270  // in some cases this might result in a failure and the grow function of
271  // the chemPoint returns false
272  if (phi0->checkSolution(phiq, Rphiq))
273  {
274  return phi0->grow(phiq);
275  }
276  // The actual solution and the approximation given by ISAT are too different
277  else
278  {
279  return false;

284 template<class CompType, class ThermoType>
285 bool
286 Foam::chemistryTabulationMethods::ISAT<CompType, ThermoType>::cleanAndBalance()
287 {
288  bool treeModified(false);
289 
290  // Check all chemPoints to see if we need to delete some of the chemPoints
291  // according to the ellapsed time and number of growths
292  chemPointISAT<CompType, ThermoType>* x = chemisTree_.treeMin();
293  while(x != nullptr)
294  {
295  chemPointISAT<CompType, ThermoType>* xtmp =
296  chemisTree_.treeSuccessor(x);
297  // timeOutputValue returns timeToUserTime(value()), therefore, it should
298  // be compare with timeToUserTime(deltaT)
299  scalar elapsedTime = runTime_.timeOutputValue() - x->timeTag();
300  scalar maxElapsedTime =
301  chPMaxLifeTime_
302  * runTime_.timeToUserTime(runTime_.deltaTValue());
303 
304  if ((elapsedTime > maxElapsedTime) || (x->nGrowth() > maxGrowth_))
305  {
306  chemisTree_.deleteLeaf(x);
307  treeModified = true;
308  }
309  x = xtmp;
310  }
311  // Check if the tree should be balanced according to criterion:
312  // -the depth of the tree bigger than a*log2(size), log2(size) being the
313  // ideal depth (e.g. 4 leafs can be stored in a tree of depth 2)
314  if
315  (
316  chemisTree_.size() > minBalanceThreshold_
317  && chemisTree_.depth() >
318  maxDepthFactor_*log(scalar(chemisTree_.size()))/log(2.0)
319  )
320  {
321  chemisTree_.balance();
322  MRUList_.clear();
323  treeModified = true;
324  }
325 
326  // Return a bool to specify if the tree structure has been modified and is
327  // now below the user specified limit (true if not full)
328  return (treeModified && !chemisTree_.isFull());
329 }

332 template<class CompType, class ThermoType>
333 void Foam::chemistryTabulationMethods::ISAT<CompType, ThermoType>::computeA
334 (
335  scalarSquareMatrix& A,
336  const scalarField& Rphiq,
337  const scalar rhoi
338 )
339 {
340  scalar dt = runTime_.deltaTValue();
341  bool mechRedActive = this->chemistry_.mechRed()->active();
342  label speciesNumber = this->chemistry_.nSpecie();
343  scalarField Rcq(this->chemistry_.nEqns());
344  for (label i=0; i<speciesNumber; i++)
345  {
346  label s2c = i;
347  if (mechRedActive)
348  {
349  s2c = this->chemistry_.simplifiedToCompleteIndex()[i];
350  }
351  Rcq[i] = rhoi*Rphiq[s2c]/this->chemistry_.specieThermo()[s2c].W();
352  }
353  Rcq[speciesNumber] = Rphiq[Rphiq.size()-2];
354  Rcq[speciesNumber+1] = Rphiq[Rphiq.size()-1];
355 
356  // Aaa is computed implicitely,
357  // A is given by A = C(psi0, t0+dt), where C is obtained through solving
358  // d/dt C(psi0,t) = J(psi(t))C(psi0,t)
359  // If we solve it implicitely:
360  // (C(psi0, t0+dt) - C(psi0,t0))/dt = J(psi(t0+dt))C(psi0,t0+dt)
361  // The Jacobian is thus computed according to the mapping
362  // C(psi0,t0+dt)*(I-dt*J(psi(t0+dt))) = C(psi0, t0)
363  // A = C(psi0,t0)/(I-dt*J(psi(t0+dt)))
364  // where C(psi0,t0) = I
365 
366  this->chemistry_.jacobian(runTime_.value(), Rcq, A);
367 
368  // The jacobian is computed according to the molar concentration
369  // the following conversion allows the code to use A with mass fraction
370  for (label i=0; i<speciesNumber; i++)
371  {
372  label si = i;
373 
374  if (mechRedActive)
375  {
376  si = this->chemistry_.simplifiedToCompleteIndex()[i];
377  }
378 
379  for (label j=0; j<speciesNumber; j++)
380  {
381  label sj = j;
382  if (mechRedActive)
383  {
384  sj = this->chemistry_.simplifiedToCompleteIndex()[j];
385  }
386  A(i, j) *=
387  -dt*this->chemistry_.specieThermo()[si].W()
388  /this->chemistry_.specieThermo()[sj].W();
389  }
390 
391  A(i, i) += 1;
392  // Columns for pressure and temperature
393  A(i, speciesNumber) *=
394  -dt*this->chemistry_.specieThermo()[si].W()/rhoi;
395  A(i, speciesNumber+1) *=
396  -dt*this->chemistry_.specieThermo()[si].W()/rhoi;
397  }
398 
399  // For temperature and pressure, only unity on the diagonal
400  A(speciesNumber, speciesNumber) = 1;
401  A(speciesNumber+1, speciesNumber+1) = 1;
402 
403  // Inverse of (I-dt*J(psi(t0+dt)))
404  LUscalarMatrix LUA(A);
405  LUA.inv(A);
406 }

411 template<class CompType, class ThermoType>
412 bool Foam::chemistryTabulationMethods::ISAT<CompType, ThermoType>::retrieve
413 (
414  const Foam::scalarField& phiq,
415  scalarField& Rphiq
416 )
417 {
418  bool retrieved(false);
419  chemPointISAT<CompType, ThermoType>* phi0;
420 
421  // If the tree is not empty
422  if (chemisTree_.size())
423  {
424  chemisTree_.binaryTreeSearch(phiq, chemisTree_.root(), phi0);
425 
426  // lastSearch keeps track of the chemPoint we obtain by the regular
427  // binary tree search
428  lastSearch_ = phi0;
429  if (phi0->inEOA(phiq))
430  {
431  retrieved = true;
432  }
433  // After a successful secondarySearch, phi0 store a pointer to the
434  // found chemPoint
435  else if (chemisTree_.secondaryBTSearch(phiq, phi0))
436  {
437  retrieved = true;
438  }
439  else if (MRURetrieve_)
440  {
441  typename SLList
442  <
443  chemPointISAT<CompType, ThermoType>*
444  >::iterator iter = MRUList_.begin();
445 
446  for ( ; iter != MRUList_.end(); ++iter)
447  {
448  phi0 = iter();
449  if (phi0->inEOA(phiq))
450  {
451  retrieved = true;
452  break;
453  }
454  }
455  }
456  }
457  // The tree is empty, retrieved is still false
458  else
459  {
460  // There is no chempoints that we can try to grow
461  lastSearch_ = nullptr;
462  }
463 
464  if (retrieved)
465  {
466  scalar elapsedTime =
467  runTime_.timeOutputValue() - phi0->timeTag();
468  scalar maxElapsedTime =
469  chPMaxLifeTime_
470  * runTime_.timeToUserTime(runTime_.deltaTValue());
471 
472  // Raise a flag when the chemPoint has been used more than the allowed
473  // number of time steps
474  if (elapsedTime > maxElapsedTime && !phi0->toRemove())
475  {
476  cleaningRequired_ = true;
477  phi0->toRemove() = true;
478  }
479  lastSearch_->lastTimeUsed() = runTime_.timeOutputValue();
480  addToMRU(phi0);
481  calcNewC(phi0,phiq, Rphiq);
482  nRetrieved_++;
483  return true;
484  }
485  else
486  {
487  // This point is reached when every retrieve trials have failed
488  // or if the tree is empty
489  return false;
490  }
491 }

494 template<class CompType, class ThermoType>
495 bool Foam::chemistryTabulationMethods::ISAT<CompType, ThermoType>::add
496 (
497  const scalarField& phiq,
498  const scalarField& Rphiq,
499  const scalar rho
500 )
501 {
502  // If lastSearch_ holds a valid pointer to a chemPoint AND the growPoints_
503  // option is on, the code first tries to grow the point hold by lastSearch_
504  if (lastSearch_ && growPoints_)
505  {
506  if (grow(lastSearch_,phiq, Rphiq))
507  {
508  nGrowth_++;
509  // the structure of the tree is not modified, return false
510  return false;
511  }
512  }
513 
514  bool treeCleanedOrCleared(false);
515 
516  // If the code reach this point, it is either because lastSearch_ is not
517  // valid, OR because growPoints_ is not on, OR because the grow operation
518  // has failed. In the three cases, a new point is added to the tree.
519  if (chemisTree().isFull())
520  {
521  // If cleanAndBalance operation do not result in a reduction of the tree
522  // size, the last possibility is to delete completely the tree.
523  // It can be partially rebuild with the MRU list if this is used.
524  if (!cleanAndBalance())
525  {
526  DynamicList<chemPointISAT<CompType, ThermoType>*> tempList;
527  if (maxMRUSize_>0)
528  {
529  // Create a copy of each chemPointISAT of the MRUList_ before
530  // they are deleted
531  typename SLList
532  <
533  chemPointISAT<CompType, ThermoType>*
534  >::iterator iter = MRUList_.begin();
535  for ( ; iter != MRUList_.end(); ++iter)
536  {
537  tempList.append
538  (
539  new chemPointISAT<CompType, ThermoType>(*iter())
540  );
541  }
542  }
543  chemisTree().clear();
544 
545  // Pointers to chemPoint are not valid anymore, clear the list
546  MRUList_.clear();
547 
548  // Construct the tree without giving a reference to attach to it
549  // since the structure has been completely discarded
550  chemPointISAT<CompType, ThermoType>* nulPhi = 0;
551  forAll(tempList, i)
552  {
553  chemisTree().insertNewLeaf
554  (
555  tempList[i]->phi(),
556  tempList[i]->Rphi(),
557  tempList[i]->A(),
558  scaleFactor(),
559  this->tolerance(),
560  scaleFactor_.size(),
561  nulPhi
562  );
563  deleteDemandDrivenData(tempList[i]);
564  }
565  }
566 
567  // The structure has been changed, it will force the binary tree to
568  // perform a new search and find the most appropriate point still stored
569  lastSearch_ = nullptr;
570 
571  // Either cleanAndBalance has changed the tree or it has been cleared
572  // in any case treeCleanedOrCleared should be set to true
573  treeCleanedOrCleared = true;
574  }
575 
576  // Compute the A matrix needed to store the chemPoint.
577  label ASize = this->chemistry_.nEqns(); // Reduced when mechRed is active
578  scalarSquareMatrix A(ASize, Zero);
579  computeA(A, Rphiq, rho);
580 
581  chemisTree().insertNewLeaf
582  (
583  phiq,
584  Rphiq,
585  A,
586  scaleFactor(),
587  this->tolerance(),
588  scaleFactor_.size(),
589  lastSearch_ // lastSearch_ may be nullptr (handled by binaryTree)
590  );
591 
592  nAdd_++;
593 
594  return treeCleanedOrCleared;
595 }
