170 inline Scalar sqrtSumSqr(const Scalar a, const Scalar b)
171 {
172  Scalar maga = mag(a);
173  Scalar magb = mag(b);
174 
175  if (maga > magb)
176  {
177  return maga*sqrt(1.0 + sqr(magb/maga));
178  }
179  else
180  {
181  return magb < ScalarVSMALL ? 0.0 : magb*sqrt(1.0 + sqr(maga/magb));
182  }
183 }
