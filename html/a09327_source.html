167 void Foam::sixDoFRigidBodyMotionSolver::solve()
168 {
169  const Time& t = mesh().time();
170 
171  if (mesh().nPoints() != points0().size())
172  {
173  FatalErrorIn
174  (
175  "sixDoFRigidBodyMotionSolver::curPoints() const"
176  ) << "The number of points in the mesh seems to have changed." << endl
177  << "In constant/polyMesh there are " << points0().size()
178  << " points; in the current mesh there are " << mesh().nPoints()
179  << " points." << exit(FatalError);
180  }
181 
182  // Store the motion state at the beginning of the time-step
183  if (curTimeIndex_ != this->db().time().timeIndex())
184  {
185  motion_.newTime();
186  curTimeIndex_ = this->db().time().timeIndex();
187  }
188 
189  // Patch force data is valid for the current positions, so
190  // calculate the forces on the motion object from this data, then
191  // update the positions
192 
193  motion_.updatePosition(t.deltaTValue(), t.deltaT0Value());
194 
195  dictionary forcesDict;
196 
197  forcesDict.add("type", forces::typeName);
198  forcesDict.add("patches", patches_);
199  forcesDict.add("rhoInf", rhoInf_);
200  forcesDict.add("rhoName", rhoName_);
201  forcesDict.add("CofR", motion_.centreOfRotation());
202 
203  forces f("forces", db(), forcesDict);
204 
205  f.calcForcesMoment();
206 
207  dimensionedVector g("g", dimAcceleration, vector::zero);
208 
209  if (db().foundObject<uniformDimensionedVectorField>("g"))
210  {
211  g = db().lookupObject<uniformDimensionedVectorField>("g");
212  }
213  else if (coeffDict().found("g"))
214  {
215  coeffDict().lookup("g") >> g;
216  }
217 
218  // scalar ramp = min(max((this->db().time().value() - 5)/10, 0), 1);
219  scalar ramp = 1.0;
220 
221  motion_.updateAcceleration
222  (
223  ramp*(f.forceEff() + motion_.mass()*g.value()),
224  ramp*(f.momentEff() + motion_.mass()*(motion_.momentArm() ^ g.value())),
225  t.deltaTValue()
226  );
227 
228  // Update the displacements
229  pointDisplacement_.internalField() =
230  motion_.transform(points0(), scale_) - points0();
231 
232  // Displacement has changed. Update boundary conditions
233  pointConstraints::New
234  (
235  pointDisplacement_.mesh()
236  ).constrainDisplacement(pointDisplacement_);
237 }
