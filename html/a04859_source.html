59 void Foam::functionObjects::wallHeatFlux::calcHeatFlux
60 (
61  const compressible::turbulenceModel& model,
62  volScalarField& wallHeatFlux
63 )
64 {
65  surfaceScalarField heatFlux
66  (
67  fvc::interpolate(model.alphaEff())*fvc::snGrad(model.transport().he())
68  );
69 
70  volScalarField::Boundary& wallHeatFluxBf =
71  wallHeatFlux.boundaryFieldRef();
72 
73  const surfaceScalarField::Boundary& heatFluxBf =
74  heatFlux.boundaryField();
75 
76  forAll(wallHeatFluxBf, patchi)
77  {
78  wallHeatFluxBf[patchi] = heatFluxBf[patchi];
79  }
80 
81  if (foundObject<volScalarField>("Qr"))
82  {
83  const volScalarField& Qr = lookupObject<volScalarField>("Qr");
84 
85  const volScalarField::Boundary& radHeatFluxBf =
86  Qr.boundaryField();
87 
88  forAll(wallHeatFluxBf, patchi)
89  {
90  wallHeatFluxBf[patchi] += radHeatFluxBf[patchi];
91  }
92  }
93 }

141 bool Foam::functionObjects::wallHeatFlux::read(const dictionary& dict)
142 {
143  fvMeshFunctionObject::read(dict);
144 
145  const polyBoundaryMesh& pbm = mesh_.boundaryMesh();
146 
147  patchSet_ =
148  mesh_.boundaryMesh().patchSet
149  (
150  wordReList(dict.lookupOrDefault("patches", wordReList()))
151  );
152 
153  Info<< type() << " " << name() << ":" << nl;
154 
155  if (patchSet_.empty())
156  {
157  forAll(pbm, patchi)
158  {
159  if (isA<wallPolyPatch>(pbm[patchi]))
160  {
161  patchSet_.insert(patchi);
162  }
163  }
164 
165  Info<< " processing all wall patches" << nl << endl;
166  }
167  else
168  {
169  Info<< " processing wall patches: " << nl;
170  labelHashSet filteredPatchSet;
171  forAllConstIter(labelHashSet, patchSet_, iter)
172  {
173  label patchi = iter.key();
174  if (isA<wallPolyPatch>(pbm[patchi]))
175  {
176  filteredPatchSet.insert(patchi);
177  Info<< " " << pbm[patchi].name() << endl;
178  }
179  else
180  {
181  WarningInFunction
182  << "Requested wall heat-flux on non-wall boundary "
183  << "type patch: " << pbm[patchi].name() << endl;
184  }
185  }
186 
187  Info<< endl;
188 
189  patchSet_ = filteredPatchSet;
190  }
191 
192  return true;
193 }

230 bool Foam::functionObjects::wallHeatFlux::write()
231 {
232  logFiles::write();
233 
234  const volScalarField& wallHeatFlux =
235  obr_.lookupObject<volScalarField>(type());
236 
237  Log << type() << " " << name() << " write:" << nl
238  << " writing field " << wallHeatFlux.name() << endl;
239 
240  wallHeatFlux.write();
241 
242  const fvPatchList& patches = mesh_.boundary();
243 
244  const surfaceScalarField::Boundary& magSf =
245  mesh_.magSf().boundaryField();
246 
247  forAllConstIter(labelHashSet, patchSet_, iter)
248  {
249  label patchi = iter.key();
250  const fvPatch& pp = patches[patchi];
251 
252  const scalarField& hfp =
253  wallHeatFlux.boundaryField()[patchi];
254 
255  const scalar minHfp = gMin(hfp);
256  const scalar maxHfp = gMax(hfp);
257  const scalar integralHfp = gSum(magSf[patchi]*hfp);
258 
259  if (Pstream::master())
260  {
261  file()
262  << mesh_.time().value()
263  << token::TAB << pp.name()
264  << token::TAB << minHfp
265  << token::TAB << maxHfp
266  << token::TAB << integralHfp
267  << endl;
268  }
269 
270  Log << " min/max(" << pp.name() << ") = "
271  << minHfp << ", " << maxHfp << ", " << integralHfp << endl;
272  }
273 
274  return true;
275 }
