99 bool Foam::setTimeStepFunctionObject::adjustTimeStep()
100 {

102  {
103  // Wanted timestep
104  scalar newDeltaT = timeStepPtr_().value(time_.timeOutputValue());
105 
106  const_cast<Time&>(time()).setDeltaT(newDeltaT, false);

112  return false;

117 bool Foam::setTimeStepFunctionObject::read(const dictionary& dict)
118 {
119  enabled_ = dict.lookupOrDefault("enabled", true);
120 
121  if (enabled_)
122  {
123  timeStepPtr_ = Function1<scalar>::New("deltaT", dict);
124 
125  // Check that time has adjustTimeStep
126  const dictionary& controlDict = time_.controlDict();
127 
128  Switch adjust;
129  if
130  (
131  !controlDict.readIfPresent<Switch>("adjustTimeStep", adjust)
132  || !adjust
133  )
134  {
135  FatalIOErrorInFunction(dict)
136  << "Need to have 'adjustTimeStep' true to enable external"
137  << " timestep control" << exit(FatalIOError);
138  }
139  }
140  return true;
141 }
