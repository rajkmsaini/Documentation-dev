35 bool Foam::IOobject::writeHeader(Ostream& os, const word& type) const
36 {
37  if (!os.good())
38  {
39  InfoInFunction
40  << "No stream open for write" << nl
41  << os.info() << endl;
42 
43  return false;
44  }
45 
46  writeBanner(os)
47  << "FoamFile\n{\n"
48  << " version " << os.version() << ";\n"
49  << " format " << os.format() << ";\n"
50  << " class " << type << ";\n";
51 
52  if (note().size())
53  {
54  os << " note " << note() << ";\n";
55  }
56 
57  os << " location " << instance()/db().dbDir()/local() << ";\n"
58  << " object " << name() << ";\n"
59  << "}" << nl;
60 
61  writeDivider(os) << endl;
62 
63  return true;
64 }
