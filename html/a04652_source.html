27 Description

56 void executeFunctionObjects
57 (
58  const argList& args,
59  const Time& runTime,
60  fvMesh& mesh,
61  const HashSet<word>& selectedFields,
62  functionObjectList& functions
63 )
64 {
65  Info<< nl << "Reading fields:" << endl;
66 
67  // Maintain a stack of the stored objects to clear after executing
68  // the functionObjects
69  LIFOStack<regIOobject*> storedObjects;
70 
71  // Read objects in time directory
72  IOobjectList objects(mesh, runTime.timeName());
73 
74  // Read volFields
75  ReadFields(volScalarField);
76  ReadFields(volVectorField);
77  ReadFields(volSphericalTensorField);
78  ReadFields(volSymmTensorField);
79  ReadFields(volTensorField);
80 
81  // Read internal fields
82  ReadFields(volScalarField::Internal);
83  ReadFields(volVectorField::Internal);
84  ReadFields(volSphericalTensorField::Internal);
85  ReadFields(volSymmTensorField::Internal);
86  ReadFields(volTensorField::Internal);
87 
88  // Read surface fields
89  ReadFields(surfaceScalarField);
90  ReadFields(surfaceVectorField);
91  ReadFields(surfaceSphericalTensorField);
92  ReadFields(surfaceSymmTensorField);
93  ReadFields(surfaceTensorField);
94 
95  // Read point fields.
96  const pointMesh& pMesh = pointMesh::New(mesh);
97 
98  ReadPointFields(pointScalarField)
99  ReadPointFields(pointVectorField);
100  ReadPointFields(pointSphericalTensorField);
101  ReadPointFields(pointSymmTensorField);
102  ReadPointFields(pointTensorField);
103 
104  // Read uniform dimensioned fields
105  IOobjectList constantObjects(mesh, runTime.constant());
106 
107  ReadUniformFields(uniformDimensionedScalarField);
108  ReadUniformFields(uniformDimensionedVectorField);
109  ReadUniformFields(uniformDimensionedSphericalTensorField);
110  ReadUniformFields(uniformDimensionedSymmTensorField);
111  ReadUniformFields(uniformDimensionedTensorField);
112 
113  Info<< nl << "Executing functionObjects" << endl;
114 
115  // Execute the functionObjects in post-processing mode
116  functions.execute();
117 
118  while (!storedObjects.empty())
119  {
120  storedObjects.pop()->checkOut();
121  }
122 }

169  runTime.setTime(timeDirs[timeI], timeI);
170

172 
173  if (mesh.readUpdate() != polyMesh::UNCHANGED)
174  {
175  // Update functionObjectList if mesh changes
176  functionsPtr = functionObjectList::New

179  runTime,
