192 Foam::tmp<Foam::volVectorField> Foam::SRF::SRFModel::Uabs() const
193 {
194  tmp<volVectorField> Usrf = U();
195 
196  tmp<volVectorField> tUabs
197  (
198  new volVectorField
199  (
200  IOobject
201  (
202  "Uabs",
203  mesh_.time().timeName(),
204  mesh_,
205  IOobject::NO_READ,
206  IOobject::NO_WRITE,
207  false
208  ),
209  Usrf
210  )
211  );
212 
213  // Add SRF contribution to internal field
214  tUabs().internalField() += Urel_.internalField();
215 
216  // Add Urel boundary contributions
217  const volVectorField::GeometricBoundaryField& bvf = Urel_.boundaryField();
218 
219  forAll(bvf, i)
220  {
221  if (isA<SRFVelocityFvPatchVectorField>(bvf[i]))
222  {
223  // Only include relative contributions from
224  // SRFVelocityFvPatchVectorField's
225  const SRFVelocityFvPatchVectorField& UrelPatch =
226  refCast<const SRFVelocityFvPatchVectorField>(bvf[i]);
227  if (UrelPatch.relative())
228  {
229  tUabs().boundaryField()[i] += Urel_.boundaryField()[i];
230  }
231  }
232  else
233  {
234  tUabs().boundaryField()[i] += Urel_.boundaryField()[i];
235  }
236  }
237 
238  return tUabs;
239 }
