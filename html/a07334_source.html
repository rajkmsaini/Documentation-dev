67 void Foam::pointToCell::combine(topoSet& set, const bool add) const
68 {
69  // Load the set
70  pointSet loadedSet(mesh_, setName_);
71 
72 
73  // Handle any selection
74  if (option_ == ANY)
75  {
76  forAllConstIter(pointSet, loadedSet, iter)
77  {
78  const label pointI = iter.key();
79  const labelList& pCells = mesh_.pointCells()[pointI];
80 
81  forAll(pCells, pCellI)
82  {
83  addOrDelete(set, pCells[pCellI], add);
84  }
85  }
86  }
87  else if (option_ == EDGE)
88  {
89  const faceList& faces = mesh_.faces();
90  forAll(faces, faceI)
91  {
92  const face& f = faces[faceI];
93 
94  forAll(f, fp)
95  {
96  if (loadedSet.found(f[fp]) && loadedSet.found(f.nextLabel(fp)))
97  {
98  addOrDelete(set, mesh_.faceOwner()[faceI], add);
99  if (mesh_.isInternalFace(faceI))
100  {
101  addOrDelete(set, mesh_.faceNeighbour()[faceI], add);
102  }
103  }
104  }
105  }
106  }
107 }
