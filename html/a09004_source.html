39 Foam::Istream& Foam::operator>>(Istream& is, Switch& s)
40 {
41  token t(is);
42 
43  if (!t.good())
44  {
45  is.setBad();
46  return is;
47  }
48 
49  if (t.isLabel())
50  {
51  s = bool(t.labelToken());
52  }
53  else if (t.isWord())
54  {
55  // allow invalid values, but catch after for correct error message
56  Switch sw(t.wordToken(), true);
57 
58  if (sw.valid())
59  {
60  s = sw;
61  }
