59 template<class CloudType>
60 bool Foam::Rebound<CloudType>::correct
61 (
62  typename CloudType::parcelType& p,
63  const polyPatch& pp,
64  bool& keepParticle,
65  const scalar trackFraction,
66  const tetIndices& tetIs
67 )
68 {
69  vector& U = p.U();
70 
71  keepParticle = true;
72  p.active() = true;
73 
74  vector nw;
75  vector Up;
76 
77  this->owner().patchData(p, pp, trackFraction, tetIs, nw, Up);
78 
79  // Calculate motion relative to patch velocity
80  U -= Up;
81 
82  scalar Un = U & nw;
83 
84  if (Un > 0.0)
85  {
86  U -= UFactor_*2.0*Un*nw;
87  }
88 
89  // Return velocity to global space
90  U += Up;
91 
92  return true;
93 }
