377 void extractSurface
378 (
379  const polyMesh& mesh,
380  const Time& runTime,
381  const labelHashSet& includePatches,
382  const fileName& outFileName
383 )
384 {
385  const polyBoundaryMesh& bMesh = mesh.boundaryMesh();
386 
387  // Collect sizes. Hash on names to handle local-only patches (e.g.
388  // processor patches)
389  HashTable<label> patchSize(1000);
390  label nFaces = 0;
391  forAllConstIter(labelHashSet, includePatches, iter)
392  {
393  const polyPatch& pp = bMesh[iter.key()];
394  patchSize.insert(pp.name(), pp.size());
395  nFaces += pp.size();
396  }
397  Pstream::mapCombineGather(patchSize, plusEqOp<label>());
398 
399 
400  // Allocate zone/patch for all patches
401  HashTable<label> compactZoneID(1000);
402  forAllConstIter(HashTable<label>, patchSize, iter)
403  {
404  label sz = compactZoneID.size();
405  compactZoneID.insert(iter.key(), sz);
406  }
407  Pstream::mapCombineScatter(compactZoneID);
408 
409 
410  // Rework HashTable into labelList just for speed of conversion
411  labelList patchToCompactZone(bMesh.size(), -1);
412  forAllConstIter(HashTable<label>, compactZoneID, iter)
413  {
414  label patchI = bMesh.findPatchID(iter.key());
415  if (patchI != -1)
416  {
417  patchToCompactZone[patchI] = iter();
418  }
419  }
420 
421  // Collect faces on zones
422  DynamicList<label> faceLabels(nFaces);
423  DynamicList<label> compactZones(nFaces);
424  forAllConstIter(labelHashSet, includePatches, iter)
425  {
426  const polyPatch& pp = bMesh[iter.key()];
427  forAll(pp, i)
428  {
429  faceLabels.append(pp.start()+i);
430  compactZones.append(patchToCompactZone[pp.index()]);
431  }
432  }
433 
434  // Addressing engine for all faces
435  uindirectPrimitivePatch allBoundary
436  (
437  UIndirectList<face>(mesh.faces(), faceLabels),
438  mesh.points()
439  );
440 
441 
442  // Find correspondence to master points
443  labelList pointToGlobal;
444  labelList uniqueMeshPoints;
445  autoPtr<globalIndex> globalNumbers = mesh.globalData().mergePoints
446  (
447  allBoundary.meshPoints(),
448  allBoundary.meshPointMap(),
449  pointToGlobal,
450  uniqueMeshPoints
451  );
452 
453  // Gather all unique points on master
454  List<pointField> gatheredPoints(Pstream::nProcs());
455  gatheredPoints[Pstream::myProcNo()] = pointField
456  (
457  mesh.points(),
458  uniqueMeshPoints
459  );
460  Pstream::gatherList(gatheredPoints);
461 
462  // Gather all faces
463  List<faceList> gatheredFaces(Pstream::nProcs());
464  gatheredFaces[Pstream::myProcNo()] = allBoundary.localFaces();
465  forAll(gatheredFaces[Pstream::myProcNo()], i)
466  {
467  inplaceRenumber(pointToGlobal, gatheredFaces[Pstream::myProcNo()][i]);
468  }
469  Pstream::gatherList(gatheredFaces);
470 
471  // Gather all ZoneIDs
472  List<labelList> gatheredZones(Pstream::nProcs());
473  gatheredZones[Pstream::myProcNo()] = compactZones.xfer();
474  Pstream::gatherList(gatheredZones);
475 
476  // On master combine all points, faces, zones
477  if (Pstream::master())
478  {
479  pointField allPoints = ListListOps::combine<pointField>
480  (
481  gatheredPoints,
482  accessOp<pointField>()
483  );
484  gatheredPoints.clear();
485 
486  faceList allFaces = ListListOps::combine<faceList>
487  (
488  gatheredFaces,
489  accessOp<faceList>()
490  );
491  gatheredFaces.clear();
492 
493  labelList allZones = ListListOps::combine<labelList>
494  (
495  gatheredZones,
496  accessOp<labelList>()
497  );
498  gatheredZones.clear();
499 
500 
501  // Zones
502  surfZoneIdentifierList surfZones(compactZoneID.size());
503  forAllConstIter(HashTable<label>, compactZoneID, iter)
504  {
505  surfZones[iter()] = surfZoneIdentifier(iter.key(), iter());
506  Info<< "surfZone " << iter() << " : " << surfZones[iter()].name()
507  << endl;
508  }
509 
510  UnsortedMeshedSurface<face> unsortedFace
511  (
512  xferMove(allPoints),
513  xferMove(allFaces),
514  xferMove(allZones),
515  xferMove(surfZones)
516  );
517 
518 
519  MeshedSurface<face> sortedFace(unsortedFace);
520 
521  fileName globalCasePath
522  (
523  runTime.processorCase()
524  ? runTime.path()/".."/outFileName
525  : runTime.path()/outFileName
526  );
527 
528  Info<< "Writing merged surface to " << globalCasePath << endl;
529 
530  sortedFace.write(globalCasePath);
531  }
532 }
