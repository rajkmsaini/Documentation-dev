135 void Foam::surfaceInterpolation::makeWeights() const
136 {
137  if (debug)
138  {
139  Pout<< "surfaceInterpolation::makeWeights() : "
140  << "Constructing weighting factors for face interpolation"
141  << endl;
142  }
143 
144  weights_ = new surfaceScalarField
145  (
146  IOobject
147  (
148  "weights",
149  mesh_.pointsInstance(),
150  mesh_,
151  IOobject::NO_READ,
152  IOobject::NO_WRITE,
153  false // Do not register
154  ),
155  mesh_,
156  dimless
157  );
158  surfaceScalarField& weights = *weights_;
159 
160  // Set local references to mesh data
161  // (note that we should not use fvMesh sliced fields at this point yet
162  // since this causes a loop when generating weighting factors in
163  // coupledFvPatchField evaluation phase)
164  const labelUList& owner = mesh_.owner();
165  const labelUList& neighbour = mesh_.neighbour();
166 
167  const vectorField& Cf = mesh_.faceCentres();
168  const vectorField& C = mesh_.cellCentres();
169  const vectorField& Sf = mesh_.faceAreas();
170 
171  // ... and reference to the internal field of the weighting factors
172  scalarField& w = weights.internalField();
173 
174  forAll(owner, facei)
175  {
176  // Note: mag in the dot-product.
177  // For all valid meshes, the non-orthogonality will be less that
178  // 90 deg and the dot-product will be positive. For invalid
179  // meshes (d & s <= 0), this will stabilise the calculation
180  // but the result will be poor.
181  scalar SfdOwn = mag(Sf[facei] & (Cf[facei] - C[owner[facei]]));
182  scalar SfdNei = mag(Sf[facei] & (C[neighbour[facei]] - Cf[facei]));
183  w[facei] = SfdNei/(SfdOwn + SfdNei);
184  }
185 
186  forAll(mesh_.boundary(), patchi)
187  {
188  mesh_.boundary()[patchi].makeWeights
189  (
190  weights.boundaryField()[patchi]
191  );
192  }
193 
194  if (debug)
195  {
196  Pout<< "surfaceInterpolation::makeWeights() : "
197  << "Finished constructing weighting factors for face interpolation"
198  << endl;
199  }
200 }

203 void Foam::surfaceInterpolation::makeDeltaCoeffs() const
204 {
205  if (debug)
206  {
207  Pout<< "surfaceInterpolation::makeDeltaCoeffs() : "
208  << "Constructing differencing factors array for face gradient"
209  << endl;
210  }
211 
212  // Force the construction of the weighting factors
213  // needed to make sure deltaCoeffs are calculated for parallel runs.
214  weights();
215 
216  deltaCoeffs_ = new surfaceScalarField
217  (
218  IOobject
219  (
220  "deltaCoeffs",
221  mesh_.pointsInstance(),
222  mesh_,
223  IOobject::NO_READ,
224  IOobject::NO_WRITE,
225  false // Do not register
226  ),
227  mesh_,
228  dimless/dimLength
229  );
230  surfaceScalarField& DeltaCoeffs = *deltaCoeffs_;
231 
232 
233  // Set local references to mesh data
234  const volVectorField& C = mesh_.C();
235  const labelUList& owner = mesh_.owner();
236  const labelUList& neighbour = mesh_.neighbour();
237 
238  forAll(owner, facei)
239  {
240  DeltaCoeffs[facei] = 1.0/mag(C[neighbour[facei]] - C[owner[facei]]);
241  }
242 
243  forAll(DeltaCoeffs.boundaryField(), patchi)
244  {
245  DeltaCoeffs.boundaryField()[patchi] =
246  1.0/mag(mesh_.boundary()[patchi].delta());
247  }
248 }

251 void Foam::surfaceInterpolation::makeNonOrthDeltaCoeffs() const
252 {
253  if (debug)
254  {
255  Pout<< "surfaceInterpolation::makeNonOrthDeltaCoeffs() : "
256  << "Constructing differencing factors array for face gradient"
257  << endl;
258  }
259 
260  // Force the construction of the weighting factors
261  // needed to make sure deltaCoeffs are calculated for parallel runs.
262  weights();
263 
264  nonOrthDeltaCoeffs_ = new surfaceScalarField
265  (
266  IOobject
267  (
268  "nonOrthDeltaCoeffs",
269  mesh_.pointsInstance(),
270  mesh_,
271  IOobject::NO_READ,
272  IOobject::NO_WRITE,
273  false // Do not register
274  ),
275  mesh_,
276  dimless/dimLength
277  );
278  surfaceScalarField& nonOrthDeltaCoeffs = *nonOrthDeltaCoeffs_;
279 
280 
281  // Set local references to mesh data
282  const volVectorField& C = mesh_.C();
283  const labelUList& owner = mesh_.owner();
284  const labelUList& neighbour = mesh_.neighbour();
285  const surfaceVectorField& Sf = mesh_.Sf();
286  const surfaceScalarField& magSf = mesh_.magSf();
287 
288  forAll(owner, facei)
289  {
290  vector delta = C[neighbour[facei]] - C[owner[facei]];
291  vector unitArea = Sf[facei]/magSf[facei];
292 
293  // Standard cell-centre distance form
294  //NonOrthDeltaCoeffs[facei] = (unitArea & delta)/magSqr(delta);
295 
296  // Slightly under-relaxed form
297  //NonOrthDeltaCoeffs[facei] = 1.0/mag(delta);
298 
299  // More under-relaxed form
300  //NonOrthDeltaCoeffs[facei] = 1.0/(mag(unitArea & delta) + VSMALL);
301 
302  // Stabilised form for bad meshes
303  nonOrthDeltaCoeffs[facei] = 1.0/max(unitArea & delta, 0.05*mag(delta));
304  }
305 
306  forAll(nonOrthDeltaCoeffs.boundaryField(), patchi)
307  {
308  vectorField delta(mesh_.boundary()[patchi].delta());
309 
310  nonOrthDeltaCoeffs.boundaryField()[patchi] =
311  1.0/max(mesh_.boundary()[patchi].nf() & delta, 0.05*mag(delta));
312  }
313 }

316 void Foam::surfaceInterpolation::makeNonOrthCorrectionVectors() const
317 {
318  if (debug)
319  {
320  Pout<< "surfaceInterpolation::makeNonOrthCorrectionVectors() : "
321  << "Constructing non-orthogonal correction vectors"
322  << endl;
323  }
324 
325  nonOrthCorrectionVectors_ = new surfaceVectorField
326  (
327  IOobject
328  (
329  "nonOrthCorrectionVectors",
330  mesh_.pointsInstance(),
331  mesh_,
332  IOobject::NO_READ,
333  IOobject::NO_WRITE,
334  false // Do not register
335  ),
336  mesh_,
337  dimless
338  );
339  surfaceVectorField& corrVecs = *nonOrthCorrectionVectors_;
340 
341  // Set local references to mesh data
342  const volVectorField& C = mesh_.C();
343  const labelUList& owner = mesh_.owner();
344  const labelUList& neighbour = mesh_.neighbour();
345  const surfaceVectorField& Sf = mesh_.Sf();
346  const surfaceScalarField& magSf = mesh_.magSf();
347  const surfaceScalarField& NonOrthDeltaCoeffs = nonOrthDeltaCoeffs();
348 
349  forAll(owner, facei)
350  {
351  vector unitArea = Sf[facei]/magSf[facei];
352  vector delta = C[neighbour[facei]] - C[owner[facei]];
353 
354  corrVecs[facei] = unitArea - delta*NonOrthDeltaCoeffs[facei];
355  }
356 
357  // Boundary correction vectors set to zero for boundary patches
358  // and calculated consistently with internal corrections for
359  // coupled patches
360 
361  forAll(corrVecs.boundaryField(), patchi)
362  {
363  fvsPatchVectorField& patchCorrVecs = corrVecs.boundaryField()[patchi];
364 
365  if (!patchCorrVecs.coupled())
366  {
367  patchCorrVecs = vector::zero;
368  }
369  else
370  {
371  const fvsPatchScalarField& patchNonOrthDeltaCoeffs
372  = NonOrthDeltaCoeffs.boundaryField()[patchi];
373 
374  const fvPatch& p = patchCorrVecs.patch();
375 
376  const vectorField patchDeltas(mesh_.boundary()[patchi].delta());
377 
378  forAll(p, patchFacei)
379  {
380  vector unitArea =
381  Sf.boundaryField()[patchi][patchFacei]
382  /magSf.boundaryField()[patchi][patchFacei];
383 
384  const vector& delta = patchDeltas[patchFacei];
385 
386  patchCorrVecs[patchFacei] =
387  unitArea - delta*patchNonOrthDeltaCoeffs[patchFacei];
388  }
389  }
390  }
391 
392  if (debug)
393  {
394  Pout<< "surfaceInterpolation::makeNonOrthCorrectionVectors() : "
395  << "Finished constructing non-orthogonal correction vectors"
396  << endl;
397  }
398 }
