184 template<class Type>
185 void Foam::fieldAverage::calculateMeanFieldType(const label fieldI) const
186 {
187  const word& fieldName = faItems_[fieldI].fieldName();
188 
189  if (obr_.foundObject<Type>(fieldName))
190  {
191  const Type& baseField = obr_.lookupObject<Type>(fieldName);
192 
193  Type& meanField = const_cast<Type&>
194  (
195  obr_.lookupObject<Type>(faItems_[fieldI].meanFieldName())
196  );
197 
198  scalar dt = obr_.time().deltaTValue();
199  scalar Dt = totalTime_[fieldI];
200 
201  if (faItems_[fieldI].iterBase())
202  {
203  dt = 1.0;
204  Dt = scalar(totalIter_[fieldI]);
205  }
206 
207  scalar alpha = (Dt - dt)/Dt;
208  scalar beta = dt/Dt;
209 
210  if (faItems_[fieldI].window() > 0)
211  {
212  const scalar w = faItems_[fieldI].window();
213 
214  if (Dt - dt >= w)
215  {
216  alpha = (w - dt)/w;
217  beta = dt/w;
218  }
219  }
220 
221  meanField = alpha*meanField + beta*baseField;
222  }
223 }
