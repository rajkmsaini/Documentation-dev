69 scalar maitlandSmith::unscaledEnergy(const scalar r) const
70 {
71  scalar nr = (m_ + gamma_*(r/rm_ - 1.0));
72 
73  return epsilon_
74  *(
75  (6.0 / (nr - 6.0))*Foam::pow(r/rm_, -nr)
76  - (nr / (nr - 6.0))*Foam::pow(r/rm_, -6)
77  );
78 }
