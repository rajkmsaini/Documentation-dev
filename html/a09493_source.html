82 void Foam::functionObjects::partialWrite::read(const dictionary& dict)
83 {
84  dict.lookup("objectNames") >> objectNames_;
85  dict.lookup("writeInterval") >> writeInterval_;
86  writeInstance_ = 0;
87 
88  Info<< type() << " " << name() << ":" << nl
89  << " dumping every " << writeInterval_
90  << " th outputTime : " << nl << endl ;
91 
92  forAllConstIter(HashSet<word>, objectNames_, iter)
93  {
94  Info<< ' ' << iter.key();
95  }
96 
97  if (writeInterval_ < 1)
98  {
99  FatalIOErrorInFunction(dict)
100  << "Illegal value for writeInterval " << writeInterval_
101  << ". It should be >= 1."
102  << exit(FatalIOError);
103  }
104 
105  // Clear out any previously loaded fields
106  vsf_.clear();
107  vvf_.clear();
108  vSpheretf_.clear();
109  vSymmtf_.clear();
110  vtf_.clear();
111 
112  ssf_.clear();
113  svf_.clear();
114  sSpheretf_.clear();
115  sSymmtf_.clear();
116  stf_.clear();
117 
118  forAllConstIter(HashSet<word>, objectNames_, iter)
119  {
120  loadField<scalar>(iter.key(), vsf_, ssf_);
121  loadField<vector>(iter.key(), vvf_, svf_);
122  loadField<sphericalTensor>(iter.key(), vSpheretf_, sSpheretf_);
123  loadField<symmTensor>(iter.key(), vSymmtf_, sSymmtf_);
124  loadField<tensor>(iter.key(), vtf_, stf_);
125  }
126 }

137 void Foam::functionObjects::partialWrite::timeSet()
138 {
139  if (obr_.time().outputTime())
140  {
141  writeInstance_++;
142 
143  if (writeInstance_ == writeInterval_)
144  {
145  // Next overall dump corresponds to partial write. Change
146  // write options to AUTO_WRITE
147  writeInstance_ = 0;
148 
149  changeWriteOptions<scalar>(vsf_, ssf_, IOobject::AUTO_WRITE);
150  changeWriteOptions<vector>(vvf_, svf_, IOobject::AUTO_WRITE);
151  changeWriteOptions<sphericalTensor>
152  (
153  vSpheretf_,
154  sSpheretf_,
155  IOobject::AUTO_WRITE
156  );
157  changeWriteOptions<symmTensor>
158  (
159  vSymmtf_,
160  sSymmtf_,
161  IOobject::AUTO_WRITE
162  );
163  changeWriteOptions<tensor>(vtf_, stf_, IOobject::AUTO_WRITE);
164  }
165  else
166  {
167  changeWriteOptions<scalar>(vsf_, ssf_, IOobject::NO_WRITE);
168  changeWriteOptions<vector>(vvf_, svf_, IOobject::NO_WRITE);
169  changeWriteOptions<sphericalTensor>
170  (
171  vSpheretf_,
172  sSpheretf_,
173  IOobject::NO_WRITE
174  );
175  changeWriteOptions<symmTensor>
176  (
177  vSymmtf_,
178  sSymmtf_,
179  IOobject::NO_WRITE
180  );
181  changeWriteOptions<tensor>(vtf_, stf_, IOobject::NO_WRITE);
182  }
183  }
184 }
