105 kOmegaSSTSato<BasicTurbulenceModel>::gasTurbulence() const
106 {
107  if (!gasTurbulencePtr_)
108  {
109  const volVectorField& U = this->U_;
110 
111  const transportModel& liquid = this->transport();
112  const twoPhaseSystem& fluid =

132 template<class BasicTurbulenceModel>
133 void kOmegaSSTSato<BasicTurbulenceModel>::correctNut()
134 {
135  const PhaseCompressibleTurbulenceModel<transportModel>& gasTurbulence =
136  this->gasTurbulence();
137 
138  volScalarField yPlus
139  (
140  pow(this->betaStar_, 0.25)*this->y_*sqrt(this->k_)/this->nu()
141  );
142 
143  this->nut_ =
144  this->a1_*this->k_
145  /max
146  (
147  this->a1_*this->omega_,
148  this->F23()*sqrt(2.0)*mag(symm(fvc::grad(this->U_)))
149  )
150  + sqr(1 - exp(-yPlus/16.0))
151  *Cmub_*gasTurbulence.transport().d()*gasTurbulence.alpha()
152  *(mag(this->U_ - gasTurbulence.U()));
153 
154  this->nut_.correctBoundaryConditions();
155  fv::options::New(this->mesh_).correct(this->nut_);
156 
157  BasicTurbulenceModel::correctNut();
158 }
