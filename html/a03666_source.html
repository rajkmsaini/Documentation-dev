64 Foam::tmp<Foam::volScalarField> Foam::liftModels::LegendreMagnaudet::Cl() const
65 {
66  volScalarField Re(max(pair_.Re(), residualRe_));
67 
68  volScalarField Sr
69  (
70  sqr(pair_.dispersed().d())
71  /(
72  Re
73  *pair_.continuous().nu()
74  )
75  *mag(fvc::grad(pair_.continuous().U()))
76  );
77 
78  volScalarField ClLowSqr
79  (
80  sqr(6.0*2.255)
81  *sqr(Sr)
82  /(
83  pow4(constant::mathematical::pi)
84  *Re
85  *pow3(Sr + 0.2*Re)
86  )
87  );
88 
89  volScalarField ClHighSqr
90  (
91  sqr(0.5*(Re + 16.0)/(Re + 29.0))
