61 void Foam::sixDoFSolvers::symplectic::solve
62 (
63  bool firstIter,
64  const vector& fGlobal,
65  const vector& tauGlobal,
66  scalar deltaT,
67  scalar deltaT0
68 )
69 {
70  if (!firstIter)
71  {
72  FatalErrorInFunction
73  << "The symplectic integrator is explicit "
74  "and can only be solved once per time-step"
75  << exit(FatalError);
76  }
77 
78  // First simplectic step:
79  // Half-step for linear and angular velocities
80  // Update position and orientation
81 
82  v() = tConstraints() & (v0() + aDamp()*0.5*deltaT0*a0());
83  pi() = rConstraints() & (pi0() + aDamp()*0.5*deltaT0*tau0());
84 
85  centreOfRotation() = centreOfRotation0() + deltaT*v();
86 
87  Tuple2<tensor, vector> Qpi = rotate(Q0(), pi(), deltaT);
88  Q() = Qpi.first();
89  pi() = rConstraints() & Qpi.second();
90 
91  // Update the linear acceleration and torque
92  updateAcceleration(fGlobal, tauGlobal);
93 
94  // Second simplectic step:
95  // Complete update of linear and angular velocities
96 
97  v() += tConstraints() & aDamp()*0.5*deltaT*a();
98  pi() += rConstraints() & aDamp()*0.5*deltaT*tau();
99 }
