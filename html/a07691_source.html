68 point sigmaRadial::operator()
69 (
70  const point& surfacePoint,
71  const vector& surfaceNormal,
72  const label layer
73 ) const
74 {
75  // radius of the surface
76  scalar rs = mag(surfacePoint);
77  vector rsHat = surfacePoint/rs;
78 
79  scalar p = pRef_ - layer*(pRef_ - pStrat_)/nLayers_;
80  scalar r = rs - RTbyg_*log(p/pRef_);
81 
82  return r*rsHat;
83 }
