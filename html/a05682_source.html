238 template<class Type>
239 void Foam::fvPatchField<Type>::autoMap
240 (
241  const fvPatchFieldMapper& mapper
242 )
243 {
244  Field<Type>& f = *this;
245 
246  if (!this->size())
247  {
248  f.setSize(mapper.size());
249  if (f.size())
250  {
251  f = this->patchInternalField();
252  }
253  }
254  else
255  {
256  // Map all faces provided with mapping data
257  Field<Type>::autoMap(mapper);
258 
259  // For unmapped faces set to internal field value (zero-gradient)
260  if
261  (
262  mapper.direct()
263  && notNull(mapper.directAddressing())
264  && mapper.directAddressing().size()
265  )
266  {
267  Field<Type> pif(this->patchInternalField());
268 
269  const labelList& mapAddressing = mapper.directAddressing();
270 
271  forAll(mapAddressing, i)
272  {
273  if (mapAddressing[i] < 0)
274  {
275  f[i] = pif[i];
276  }
277  }
278  }
279  else if (!mapper.direct() && mapper.addressing().size())
280  {
281  Field<Type> pif(this->patchInternalField());
282 
283  const labelListList& mapAddressing = mapper.addressing();
284 
285  forAll(mapAddressing, i)
286  {
287  const labelList& localAddrs = mapAddressing[i];
288 
289  if (!localAddrs.size())
290  {
291  f[i] = pif[i];
292  }
293  }
294  }
295  }
296 }
