51 Foam::label Foam::metisDecomp::decompose
52 (
53  const List<label>& adjncy,
54  const List<label>& xadj,
55  const scalarField& cWeights,
56 
57  List<label>& finalDecomp
58 )
59 {
60  // C style numbering
61  //int numFlag = 0;
62 
63  // Method of decomposition
64  // recursive: multi-level recursive bisection (default)
65  // k-way: multi-level k-way
66  word method("recursive");
67 
68  label numCells = xadj.size()-1;
69 
70  // decomposition options
71  List<label> options(METIS_NOPTIONS);
72  METIS_SetDefaultOptions(options.begin());
73 
74  // processor weights initialised with no size, only used if specified in
75  // a file
76  Field<floatScalar> processorWeights;
77 
78  // cell weights (so on the vertices of the dual)
79  List<label> cellWeights;
80 
81  // face weights (so on the edges of the dual)
82  List<label> faceWeights;
83 
84 
85  // Check for externally provided cellweights and if so initialise weights
86  scalar minWeights = gMin(cWeights);
87  if (cWeights.size() > 0)
88  {
89  if (minWeights <= 0)
90  {
91  WarningIn
92  (
93  "metisDecomp::decompose"
94  "(const pointField&, const scalarField&)"
95  ) << "Illegal minimum weight " << minWeights
96  << endl;
97  }
98 
99  if (cWeights.size() != numCells)
100  {
101  FatalErrorIn
102  (
103  "metisDecomp::decompose"
104  "(const pointField&, const scalarField&)"
105  ) << "Number of cell weights " << cWeights.size()
106  << " does not equal number of cells " << numCells
107  << exit(FatalError);
108  }
109  // Convert to integers.
110  cellWeights.setSize(cWeights.size());
111  forAll(cellWeights, i)
112  {
113  cellWeights[i] = int(cWeights[i]/minWeights);
114  }
115  }
116 
117 
118  // Check for user supplied weights and decomp options
119  if (decompositionDict_.found("metisCoeffs"))
120  {
121  const dictionary& metisCoeffs =
122  decompositionDict_.subDict("metisCoeffs");
123  word weightsFile;
124 
125  if (metisCoeffs.readIfPresent("method", method))
126  {
127  if (method != "recursive" && method != "k-way")
128  {
129  FatalErrorIn("metisDecomp::decompose()")
130  << "Method " << method << " in metisCoeffs in dictionary : "
131  << decompositionDict_.name()
132  << " should be 'recursive' or 'k-way'"
133  << exit(FatalError);
134  }
135 
136  Info<< "metisDecomp : Using Metis method " << method
137  << nl << endl;
138  }
139 
140  if (metisCoeffs.readIfPresent("options", options))
141  {
142  if (options.size() != METIS_NOPTIONS)
143  {
144  FatalErrorIn("metisDecomp::decompose()")
145  << "Number of options in metisCoeffs in dictionary : "
146  << decompositionDict_.name()
147  << " should be " << METIS_NOPTIONS
148  << exit(FatalError);
149  }
150 
151  Info<< "metisDecomp : Using Metis options " << options
152  << nl << endl;
153  }
154 
155  if (metisCoeffs.readIfPresent("processorWeights", processorWeights))
156  {
157  processorWeights /= sum(processorWeights);
158 
159  if (processorWeights.size() != nProcessors_)
160  {
161  FatalErrorIn("metisDecomp::decompose(const pointField&)")
162  << "Number of processor weights "
163  << processorWeights.size()
164  << " does not equal number of domains " << nProcessors_
165  << exit(FatalError);
166  }
167  }
168 
169  //if (metisCoeffs.readIfPresent("cellWeightsFile", weightsFile))
170  //{
171  // Info<< "metisDecomp : Using cell-based weights." << endl;
172  //
173  // IOList<label> cellIOWeights
174  // (
175  // IOobject
176  // (
177  // weightsFile,
178  // mesh_.time().timeName(),
179  // mesh_,
180  // IOobject::MUST_READ,
181  // IOobject::AUTO_WRITE
182  // )
183  // );
184  // cellWeights.transfer(cellIOWeights);
185  //
186  // if (cellWeights.size() != xadj.size()-1)
187  // {
188  // FatalErrorIn("metisDecomp::decompose(const pointField&)")
189  // << "Number of cell weights " << cellWeights.size()
190  // << " does not equal number of cells " << xadj.size()-1
191  // << exit(FatalError);
192  // }
193  //}
194  }
195 
196  int ncon = 1;
197 
198  int nProcs = nProcessors_;
199 
200  // output: cell -> processor addressing
201  finalDecomp.setSize(numCells);
202 
203  // output: number of cut edges
204  int edgeCut = 0;
205 
206  if (method == "recursive")
207  {
208  METIS_PartGraphRecursive
209  (
210  &numCells, // num vertices in graph
211  &ncon, // num balancing constraints
212  const_cast<List<label>&>(xadj).begin(), // indexing into adjncy
213  const_cast<List<label>&>(adjncy).begin(), // neighbour info
214  cellWeights.begin(),// vertexweights
215  NULL, // vsize: total communication vol
216  faceWeights.begin(),// edgeweights
217  &nProcs, // nParts
218  processorWeights.begin(), // tpwgts
219  NULL, // ubvec: processor imbalance (default)
220  options.begin(),
221  &edgeCut,
222  finalDecomp.begin()
223  );
224  }
225  else
226  {
227  METIS_PartGraphKway
228  (
229  &numCells, // num vertices in graph
230  &ncon, // num balancing constraints
231  const_cast<List<label>&>(xadj).begin(), // indexing into adjncy
232  const_cast<List<label>&>(adjncy).begin(), // neighbour info
233  cellWeights.begin(),// vertexweights
234  NULL, // vsize: total communication vol
235  faceWeights.begin(),// edgeweights
236  &nProcs, // nParts
237  processorWeights.begin(), // tpwgts
238  NULL, // ubvec: processor imbalance (default)
239  options.begin(),
240  &edgeCut,
241  finalDecomp.begin()
242  );
243  }
244 
245  return edgeCut;
246 }

297 Foam::labelList Foam::metisDecomp::decompose
298 (
299  const polyMesh& mesh,
300  const labelList& agglom,
301  const pointField& agglomPoints,
302  const scalarField& agglomWeights
303 )
304 {
305  if (agglom.size() != mesh.nCells())
306  {
307  FatalErrorIn
308  (
309  "metisDecomp::decompose"
310  "(const labelList&, const pointField&, const scalarField&)"
311  ) << "Size of cell-to-coarse map " << agglom.size()
312  << " differs from number of cells in mesh " << mesh.nCells()
313  << exit(FatalError);
314  }
315 
316  // Make Metis CSR (Compressed Storage Format) storage
317  // adjncy : contains neighbours (= edges in graph)
318  // xadj(celli) : start of information in adjncy for celli
319 
320  CompactListList<label> cellCells;
321  calcCellCells(mesh, agglom, agglomPoints.size(), false, cellCells);
322 
323  // Decompose using default weights
324  labelList finalDecomp;
325  decompose(cellCells.m(), cellCells.offsets(), agglomWeights, finalDecomp);
326 
327 
328  // Rework back into decomposition for original mesh
329  labelList fineDistribution(agglom.size());
330 
331  forAll(fineDistribution, i)
332  {
333  fineDistribution[i] = finalDecomp[agglom[i]];
334  }
335 
336  return finalDecomp;
337 }
