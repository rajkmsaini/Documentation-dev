63 Foam::tmp<Foam::volScalarField>
64 Foam::dragModels::IshiiZuber::CdRe() const
65 {
66  volScalarField Re(pair_.Re());
67  volScalarField Eo(pair_.Eo());
68 
69  volScalarField mud(pair_.dispersed().mu());
70  volScalarField muc(pair_.continuous().mu());
71 
72  volScalarField muStar((mud + 0.4*muc)/(mud + muc));
73 
74  volScalarField muMix
75  (
76  muc
77  *pow(max(1 - pair_.dispersed(), scalar(1e-3)), -2.5*muStar)
78  );
79 
80  volScalarField ReM(Re*muc/muMix);
81  volScalarField CdRe
82  (
83  pos(1000 - ReM)*24.0*(scalar(1) + 0.15*pow(ReM, 0.687))
84  + neg(1000 - ReM)*0.44*ReM
85  );
86 
87  volScalarField F((muc/muMix)*sqrt(1 - pair_.dispersed()));
88  F.max(1e-3);
89 
90  volScalarField Ealpha((1 + 17.67*pow(F, 0.8571428))/(18.67*F));
91 
92  volScalarField CdReEllipse(Ealpha*0.6666*sqrt(Eo)*Re);
93 
94  return
95  pos(CdReEllipse - CdRe)
96  *min(CdReEllipse, Re*sqr(1 - pair_.dispersed())*2.66667)
97  + neg(CdReEllipse - CdRe)*CdRe;
98 }
