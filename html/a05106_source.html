40 void Foam::porosityModel::adjustNegativeResistance(dimensionedVector& resist)
41 {
42  scalar maxCmpt = max(0, cmptMax(resist.value()));
43 
44  if (maxCmpt < 0)
45  {
46  FatalErrorInFunction
47  << "Negative resistances are invalid, resistance = " << resist
48  << exit(FatalError);
49  }
50  else
51  {
52  vector& val = resist.value();
53  for (label cmpt = 0; cmpt < vector::nComponents; cmpt++)
54  {
55  if (val[cmpt] < 0)
56  {
57  val[cmpt] *= -maxCmpt;
58  }
59  }
60  }
61 }
