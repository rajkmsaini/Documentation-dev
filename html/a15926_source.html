39 template<class CompType, class ThermoType>
40 void Foam::chemPointISAT<CompType, ThermoType>::qrDecompose
41 (
42  const label nCols,
43  scalarSquareMatrix& R
44 )
45 {
46  scalarField c(nCols);
47  scalarField d(nCols);
48  scalar scale, sigma, sum;
49 
50  for (label k=0; k<nCols-1; k++)
51  {
52  scale = 0;
53  for (label i=k; i<nCols; i++)
54  {
55  scale=max(scale, mag(R(i, k)));
56  }
57  if (scale == 0)
58  {
59  c[k] = d[k] = 0;
60  }
61  else
62  {
63  for (label i=k; i<nCols; i++)
64  {
65  R(i, k) /= scale;
66  }
67  sum = 0;
68  for (label i=k; i<nCols; i++)
69  {
70  sum += sqr(R(i, k));
71  }
72  sigma = sign(R(k, k))*sqrt(sum);
73  R(k, k) += sigma;
74  c[k] = sigma*R(k, k);
75  d[k] = -scale*sigma;
76  for (label j=k+1; j<nCols; j++)
77  {
78  sum=0;
79  for ( label i=k; i<nCols; i++)
80  {
81  sum += R(i, k)*R(i, j);
82  }
83  scalar tau = sum/c[k];
84  for ( label i=k; i<nCols; i++)
85  {
86  R(i, j) -= tau*R(i, k);
87  }
88  }
89  }
90  }
91 
92  d[nCols-1] = R(nCols-1, nCols-1);
93 
94  // form R
95  for (label i=0; i<nCols; i++)
96  {
97  R(i, i) = d[i];
98  for ( label j=0; j<i; j++)
99  {
100  R(i, j)=0;
101  }
102  }
103 }

106 template<class CompType, class ThermoType>
107 void Foam::chemPointISAT<CompType, ThermoType>::qrUpdate
108 (
109  scalarSquareMatrix& R,
110  const label n,
111  const Foam::scalarField &u,
112  const Foam::scalarField &v
113 )
114 {
115  label k;
116 
117  scalarField w(u);
118  for (k=n-1; k>=0; k--)
119  {
120  if (w[k] != 0)
121  {
122  break;
123  }
124  }
125 
126  if (k < 0)
127  {
128  k = 0;
129  }
130 
131  for (label i=k-1; i>=0; i--)
132  {
133  rotate(R, i, w[i],-w[i+1], n);
134  if (w[i] == 0)
135  {
136  w[i] = mag(w[i+1]);
137  }
138  else if (mag(w[i]) > mag(w[i+1]))
139  {
140  w[i] = mag(w[i])*sqrt(1 + sqr(w[i+1]/w[i]));
141  }
142  else
143  {
144  w[i] = mag(w[i+1])*sqrt(1 + sqr(w[i]/w[i+1]));
145  }
146  }
147 
148  for (label i=0; i<n; i++)
149  {
150  R(0, i) += w[0]*v[i];
151  }
152 
153  for (label i=0; i<k; i++)
154  {
155  rotate(R, i, R(i, i), -R(i+1, i), n);
156  }
157 }

160 template<class CompType, class ThermoType>
161 void Foam::chemPointISAT<CompType, ThermoType>::rotate
162 (
163  scalarSquareMatrix& R,
164  const label i,
165  const scalar a,
166  const scalar b,
167  label n
168 )
169 {
170  scalar c, fact, s, w, y;
171  if (a == 0)
172  {
173  c = 0;
174  s = (b >= 0 ? 1.0 : -1.0);
175  }
176  else if (mag(a) > mag(b))
177  {
178  fact = b/a;
179  c = sign(a)/sqrt(1 + sqr(fact));
180  s = fact*c;
181  }
182  else
183  {
184  fact = a/b;
185  s = sign(b)/sqrt(1 + sqr(fact));
186  c = fact*s;
187  }
188  for (label j=i;j<n;j++)
189  {
190  y = R(i, j);
191  w = R(i+1, j);
192  R(i, j) = c*y-s*w;
193  R(i+1, j) = s*y+c*w;
194  }
195 }

200 template<class CompType, class ThermoType>
201 Foam::chemPointISAT<CompType, ThermoType>::chemPointISAT
202 (
203  TDACChemistryModel<CompType, ThermoType>& chemistry,
204  const scalarField& phi,
205  const scalarField& Rphi,
206  const scalarSquareMatrix& A,
207  const scalarField& scaleFactor,
208  const scalar& tolerance,
209  const label& completeSpaceSize,
210  const dictionary& coeffsDict,
211  binaryNode<CompType, ThermoType>* node
212 )
213 :
214  chemistry_(chemistry),
215  phi_(phi),
216  Rphi_(Rphi),
217  A_(A),
218  scaleFactor_(scaleFactor),
219  node_(node),
220  completeSpaceSize_(completeSpaceSize),
221  nGrowth_(0),
222  nActiveSpecies_(chemistry.mechRed()->NsSimp()),
223  completeToSimplifiedIndex_(completeSpaceSize-2),
224  simplifiedToCompleteIndex_(nActiveSpecies_),
225  timeTag_(chemistry_.time().timeOutputValue()),
226  lastTimeUsed_(chemistry_.time().timeOutputValue()),
227  toRemove_(false),
228  maxNumNewDim_(coeffsDict.lookupOrDefault("maxNumNewDim",0)),
229  printProportion_(coeffsDict.lookupOrDefault("printProportion",false))
230 {
231  tolerance_=tolerance;
232 
233  bool isMechRedActive = chemistry_.mechRed()->active();
234  if (isMechRedActive)
235  {
236  for (label i=0; i<completeSpaceSize-2; i++)
237  {
238  completeToSimplifiedIndex_[i] =
239  chemistry.completeToSimplifiedIndex()[i];
240  }
241  for (label i=0; i<nActiveSpecies_; i++)
242  {
243  simplifiedToCompleteIndex_[i] =
244  chemistry.simplifiedToCompleteIndex()[i];
245  }
246  }
247 
248  label reduOrCompDim = completeSpaceSize;
249  if (isMechRedActive)
250  {
251  reduOrCompDim = nActiveSpecies_ + 2;
252  }
253 
254  // SVD decomposition A = U*D*V^T
255  SVD svdA(A);
256 
257  scalarDiagonalMatrix D(reduOrCompDim);
258  const scalarDiagonalMatrix& S = svdA.S();
259 
260  // Replace the value of vector D by max(D, 1/2), first ISAT paper
261  for (label i=0; i<reduOrCompDim; i++)
262  {
263  D[i] = max(S[i], 0.5);
264  }
265 
266  // Rebuild A with max length, tol and scale factor before QR decomposition
267  scalarRectangularMatrix Atilde(reduOrCompDim);
268 
269  // Result stored in Atilde
270  multiply(Atilde, svdA.U(), D, svdA.V().T());
271 
272  for (label i=0; i<reduOrCompDim; i++)
273  {
274  for (label j=0; j<reduOrCompDim; j++)
275  {
276  label compi = i;
277 
278  if (isMechRedActive)
279  {
280  compi = simplifiedToCompleteIndex(i);
281  }
282 
283  // SF*A/tolerance
284  // (where SF is diagonal with inverse of scale factors)
285  // SF*A is the same as dividing each line by the scale factor
286  // corresponding to the species of this line
287  Atilde(i, j) /= (tolerance*scaleFactor[compi]);
288  }
289  }
290 
291  // The object LT_ (the transpose of the Q) describe the EOA, since we have
292  // A^T B^T B A that should be factorized into L Q^T Q L^T and is set in the
293  // qrDecompose function
294  LT_ = scalarSquareMatrix(Atilde);
295 
296  qrDecompose(reduOrCompDim, LT_);
297 }

329 template<class CompType, class ThermoType>
330 bool Foam::chemPointISAT<CompType, ThermoType>::inEOA(const scalarField& phiq)
331 {
332  scalarField dphi(phiq-phi());
333  bool isMechRedActive = chemistry_.mechRed()->active();
334  label dim = (isMechRedActive) ? nActiveSpecies_ : completeSpaceSize()-2;
335  scalar epsTemp=0;
336  List<scalar> propEps(completeSpaceSize(), scalar(0));
337 
338  for (label i=0; i<completeSpaceSize()-2; i++)
339  {
340  scalar temp = 0;
341 
342  // When mechanism reduction is inactive OR on active species multiply L
343  // by dphi to get the distance in the active species direction else (for
344  // inactive species), just multiply the diagonal element and dphi
345  if
346  (
347  !(isMechRedActive)
348  ||(isMechRedActive && completeToSimplifiedIndex_[i] != -1)
349  )
350  {
351  label si=(isMechRedActive) ? completeToSimplifiedIndex_[i] : i;
352 
353  for (label j=si; j<dim; j++)// LT is upper triangular
354  {
355  label sj=(isMechRedActive) ? simplifiedToCompleteIndex_[j] : j;
356  temp += LT_(si, j)*dphi[sj];
357  }
358 
359  temp += LT_(si, nActiveSpecies_)*dphi[completeSpaceSize()-2];
360  temp += LT_(si, nActiveSpecies_+1)*dphi[completeSpaceSize()-1];
361  }
362  else
363  {
364  temp = dphi[i]/(tolerance_*scaleFactor_[i]);
365  }
366 
367  epsTemp += sqr(temp);
368 
369  if (printProportion_)
370  {
371  propEps[i] = temp;
372  }
373  }
374 
375  // Temperature
376  epsTemp +=
377  sqr
378  (
379  LT_(dim, dim)*dphi[completeSpaceSize()-2]
380  +LT_(dim, dim+1)*dphi[completeSpaceSize()-1]
381  );
382 
383  // Pressure
384  epsTemp += sqr(LT_(dim+1, dim+1)*dphi[completeSpaceSize()-1]);
385 
386  if (printProportion_)
387  {
388  propEps[completeSpaceSize()-2] =
389  sqr
390  (
391  LT_(dim, dim)*dphi[completeSpaceSize()-2]
392  + LT_(dim, dim+1)*dphi[completeSpaceSize()-1]
393  );
394 
395  propEps[completeSpaceSize()-1] =
396  sqr(LT_(dim+1, dim+1)*dphi[completeSpaceSize()-1]);
397  }
398  if (sqrt(epsTemp) > 1 + tolerance_)
399  {
400  if (printProportion_)
401  {
402  scalar max = -1;
403  label maxIndex = -1;
404  for (label i=0; i<completeSpaceSize(); i++)
405  {
406  if(max < propEps[i])
407  {
408  max = propEps[i];
409  maxIndex = i;
410  }
411  }
412  word propName;
413  if (maxIndex >= completeSpaceSize()-2)
414  {
415  if(maxIndex == completeSpaceSize()-2)
416  {
417  propName = "T";
418  }
419  else if(maxIndex == completeSpaceSize()-1)
420  {
421  propName = "p";
422  }
423  }
424  else
425  {
426  propName = chemistry_.Y()[maxIndex].name();
427  }
428  Info<< "Direction maximum impact to error in ellipsoid: "
429  << propName << endl;
430  Info<< "Proportion to the total error on the retrieve: "
431  << max/(epsTemp+SMALL) << endl;
432  }
433  return false;
434  }
435  else
436  {
437  return true;
438  }
439 }

442 template<class CompType, class ThermoType>
443 bool Foam::chemPointISAT<CompType, ThermoType>::checkSolution
444 (
445  const scalarField& phiq,
446  const scalarField& Rphiq
447 )
448 {
449  scalar eps2 = 0;
450  scalarField dR(Rphiq - Rphi());
451  scalarField dphi(phiq - phi());
452  const scalarField& scaleFactorV(scaleFactor());
453  const scalarSquareMatrix& Avar(A());
454  bool isMechRedActive = chemistry_.mechRed()->active();
455  scalar dRl = 0;
456  label dim = completeSpaceSize()-2;
457  if (isMechRedActive)
458  {
459  dim = nActiveSpecies_;
460  }
461 
462  // Since we build only the solution for the species, T and p are not
463  // included
464  for (label i=0; i<completeSpaceSize()-2; i++)
465  {
466  dRl = 0;
467  if (isMechRedActive)
468  {
469  label si = completeToSimplifiedIndex_[i];
470 
471  // If this species is active
472  if (si != -1)
473  {
474  for (label j=0; j<dim; j++)
475  {
476  label sj=simplifiedToCompleteIndex_[j];
477  dRl += Avar(si, j)*dphi[sj];
478  }
479  dRl += Avar(si, nActiveSpecies_)*dphi[completeSpaceSize()-2];
480  dRl += Avar(si, nActiveSpecies_+1)*dphi[completeSpaceSize()-1];
481  }
482  else
483  {
484  dRl = dphi[i];
485  }
486  }
487  else
488  {
489  for (label j=0; j<completeSpaceSize(); j++)
490  {
491  dRl += Avar(i, j)*dphi[j];
492  }
493  }
494  eps2 += sqr((dR[i]-dRl)/scaleFactorV[i]);
495  }
496 
497  eps2 = sqrt(eps2);
498  if (eps2 > tolerance())
499  {
500  return false;
501  }
502  else
503  {
504  // if the solution is in the ellipsoid of accuracy
505  return true;
506  }
507 }

510 template<class CompType, class ThermoType>
511 bool Foam::chemPointISAT<CompType, ThermoType>::grow(const scalarField& phiq)
512 {
513  scalarField dphi(phiq - phi());
514  label dim = completeSpaceSize();
515  label initNActiveSpecies(nActiveSpecies_);
516  bool isMechRedActive = chemistry_.mechRed()->active();
517 
518  if (isMechRedActive)
519  {
520  label activeAdded(0);
521  DynamicList<label> dimToAdd(0);
522 
523  // check if the difference of active species is lower than the maximum
524  // number of new dimensions allowed
525  for (label i=0; i<completeSpaceSize()-2; i++)
526  {
527  // first test if the current chemPoint has an inactive species
528  // corresponding to an active one in the query point
529  if
530  (
531  completeToSimplifiedIndex_[i] == -1
532  && chemistry_.completeToSimplifiedIndex()[i]!=-1
533  )
534  {
535  activeAdded++;
536  dimToAdd.append(i);
537  }
538  // then test if an active species in the current chemPoint
539  // corresponds to an inactive on the query side
540  if
541  (
542  completeToSimplifiedIndex_[i] != -1
543  && chemistry_.completeToSimplifiedIndex()[i] == -1
544  )
545  {
546  activeAdded++;
547  // we don't need to add a new dimension but we count it to have
548  // control on the difference through maxNumNewDim
549  }
550  // finally test if both points have inactive species but
551  // with a dphi!=0
552  if
553  (
554  completeToSimplifiedIndex_[i] == -1
555  && chemistry_.completeToSimplifiedIndex()[i] == -1
556  && dphi[i] != 0
557  )
558  {
559  activeAdded++;
560  dimToAdd.append(i);
561  }
562  }
563 
564  // if the number of added dimension is too large, growth fail
565  if (activeAdded > maxNumNewDim_)
566  {
567  return false;
568  }
569 
570  // the number of added dimension to the current chemPoint
571  nActiveSpecies_ += dimToAdd.size();
572  simplifiedToCompleteIndex_.setSize(nActiveSpecies_);
573  forAll(dimToAdd, i)
574  {
575  label si = nActiveSpecies_ - dimToAdd.size() + i;
576  // add the new active species
577  simplifiedToCompleteIndex_[si] = dimToAdd[i];
578  completeToSimplifiedIndex_[dimToAdd[i]] = si;
579  }
580 
581  // update LT and A :
582  //-add new column and line for the new active species
583  //-transfer last two lines of the previous matrix (p and T) to the end
584  // (change the diagonal position)
585  //-set all element of the new lines and columns to zero except diagonal
586  // (=1/(tolerance*scaleFactor))
587  if (nActiveSpecies_ > initNActiveSpecies)
588  {
589  scalarSquareMatrix LTvar = LT_; // take a copy of LT_
590  scalarSquareMatrix Avar = A_; // take a copy of A_
591  LT_ = scalarSquareMatrix(nActiveSpecies_+2, Zero);
592  A_ = scalarSquareMatrix(nActiveSpecies_+2, Zero);
593 
594  // write the initial active species
595  for (label i=0; i<initNActiveSpecies; i++)
596  {
597  for (label j=0; j<initNActiveSpecies; j++)
598  {
599  LT_(i, j) = LTvar(i, j);
600  A_(i, j) = Avar(i, j);
601  }
602  }
603 
604  // write the columns for temperature and pressure
605  for (label i=0; i<initNActiveSpecies; i++)
606  {
607  for (label j=1; j>=0; j--)
608  {
609  LT_(i, nActiveSpecies_+j)=LTvar(i, initNActiveSpecies+j);
610  A_(i, nActiveSpecies_+j)=Avar(i, initNActiveSpecies+j);
611  LT_(nActiveSpecies_+j, i)=LTvar(initNActiveSpecies+j, i);
612  A_(nActiveSpecies_+j, i)=Avar(initNActiveSpecies+j, i);
613  }
614  }
615  // end with the diagonal elements for temperature and pressure
616  LT_(nActiveSpecies_, nActiveSpecies_)=
617  LTvar(initNActiveSpecies, initNActiveSpecies);
618  A_(nActiveSpecies_, nActiveSpecies_)=
619  Avar(initNActiveSpecies, initNActiveSpecies);
620  LT_(nActiveSpecies_+1, nActiveSpecies_+1)=
621  LTvar(initNActiveSpecies+1, initNActiveSpecies+1);
622  A_(nActiveSpecies_+1, nActiveSpecies_+1)=
623  Avar(initNActiveSpecies+1, initNActiveSpecies+1);
624 
625  for (label i=initNActiveSpecies; i<nActiveSpecies_;i++)
626  {
627  LT_(i, i)=
628  1.0
629  /(tolerance_*scaleFactor_[simplifiedToCompleteIndex_[i]]);
630  A_(i, i) = 1;
631  }
632  }
633 
634  dim = nActiveSpecies_ + 2;
635  }
636 
637  // beginning of grow algorithm
638  scalarField phiTilde(dim, 0);
639  scalar normPhiTilde = 0;
640  // p' = L^T.(p-phi)
641 
642  for (label i=0; i<dim; i++)
643  {
644  for (label j=i; j<dim-2; j++)// LT is upper triangular
645  {
646  label sj = j;
647  if (isMechRedActive)
648  {
649  sj = simplifiedToCompleteIndex_[j];
650  }
651  phiTilde[i] += LT_(i, j)*dphi[sj];
652  }
653  phiTilde[i] += LT_(i, dim-2)*dphi[completeSpaceSize()-2];
654  phiTilde[i] += LT_(i, dim-1)*dphi[completeSpaceSize()-1];
655  normPhiTilde += sqr(phiTilde[i]);
656  }
657 
658  scalar invSqrNormPhiTilde = 1.0/normPhiTilde;
659  normPhiTilde = sqrt(normPhiTilde);
660 
661  // gamma = (1/|p'| - 1)/|p'|^2
662  scalar gamma = (1/normPhiTilde - 1)*invSqrNormPhiTilde;
663  scalarField u(gamma*phiTilde);
664  scalarField v(dim, 0);
665 
666  for (label i=0; i<dim; i++)
667  {
668  for (label j=0; j<=i;j++)
669  {
670  v[i] += phiTilde[j]*LT_(j, i);
671  }
672  }
673 
674  qrUpdate(LT_,dim, u, v);
675  nGrowth_++;
676 
677  return true;
678 }
