205 (
206  const label parent,
207  const label i
208 )

210  label index = -1;
211

248 void Foam::refinementHistory::freeSplitCell(const label index)
249 {
250  splitCell8& split = splitCells_[index];
251 
252  // Make sure parent does not point to me anymore.
253  if (split.parent_ >= 0)
254  {
255  autoPtr<FixedList<label, 8>>& subCellsPtr =
256  splitCells_[split.parent_].addedCellsPtr_;
257 
258  if (subCellsPtr.valid())
259  {
260  FixedList<label, 8>& subCells = subCellsPtr();
261 
262  label myPos = findIndex(subCells, index);
263 
264  if (myPos == -1)
265  {
266  FatalErrorInFunction
267  << "Problem: cannot find myself in"
268  << " parents' children" << abort(FatalError);
269  }
270  else
271  {
272  subCells[myPos] = -1;
273  }
274  }
275  }
276 
277  // Mark splitCell as free
278  split.parent_ = -2;
279 
280  // Add to cache of free splitCells
281  freeSplitCells_.append(index);

616 ) const
617 {

642 
643  if (parent >= 0)
644  {

646  }

652 void Foam::refinementHistory::distribute(const mapDistributePolyMesh& map)
653 {
654  if (!active())
655  {
656  FatalErrorInFunction
657  << "Calling distribute on inactive history" << abort(FatalError);
658  }
659 
660 
661  if (!Pstream::parRun())
662  {
663  return;
664  }
665 
666  // Remove unreferenced history.
667  compact();
668 
669  //Pout<< nl << "--BEFORE:" << endl;
670  //writeDebug();
671  //Pout<< "---------" << nl << endl;
672 
673 
674  // Distribution is only partially functional.
675  // If all 8 cells resulting from a single parent are sent across in one
676  // go it will also send across that part of the refinement history.
677  // If however e.g. first 1 and then the other 7 are sent across the
678  // history will not be reconstructed.
679 
680  // Determine clusters. This is per every entry in splitCells_ (that is
681  // a parent of some refinement) a label giving the processor it goes to
682  // if all its children are going to the same processor.
683 
684  // Per visible cell the processor it goes to.
685  labelList destination(visibleCells_.size());
686 
687  const labelListList& subCellMap = map.cellMap().subMap();
688 
689  forAll(subCellMap, procI)
690  {
691  const labelList& newToOld = subCellMap[procI];
692 
693  forAll(newToOld, i)
694  {
695  label oldCellI = newToOld[i];
696 
697  destination[oldCellI] = procI;
698  }
699  }
700 
701 //Pout<< "refinementHistory::distribute :"
702 // << " destination:" << destination << endl;
703 
704  // Per splitCell entry the processor it moves to
705  labelList splitCellProc(splitCells_.size(), -1);
706  // Per splitCell entry the number of live cells that move to that processor
707  labelList splitCellNum(splitCells_.size(), 0);
708 
709  forAll(visibleCells_, cellI)
710  {
711  label index = visibleCells_[cellI];
712 
713  if (index >= 0)
714  {
715  countProc
716  (
717  splitCells_[index].parent_,
718  destination[cellI],
719  splitCellProc,
720  splitCellNum
721  );
722  }
723  }
724 
725  //Pout<< "refinementHistory::distribute :"
726  // << " splitCellProc:" << splitCellProc << endl;
727  //
728  //Pout<< "refinementHistory::distribute :"
729  // << " splitCellNum:" << splitCellNum << endl;
730 
731 
732  // Create subsetted refinement tree consisting of all parents that
733  // move in their whole to other processor.
734  for (label procI = 0; procI < Pstream::nProcs(); procI++)
735  {
736  //Pout<< "-- Subetting for processor " << procI << endl;
737 
738  // From uncompacted to compacted splitCells.
739  labelList oldToNew(splitCells_.size(), -1);
740 
741  // Compacted splitCells. Similar to subset routine below.
742  DynamicList<splitCell8> newSplitCells(splitCells_.size());
743 
744  // Loop over all entries. Note: could recurse like countProc so only
745  // visit used entries but is probably not worth it.
746 
747  forAll(splitCells_, index)
748  {
749 // Pout<< "oldCell:" << index
750 // << " proc:" << splitCellProc[index]
751 // << " nCells:" << splitCellNum[index]
752 // << endl;
753 
754  if (splitCellProc[index] == procI && splitCellNum[index] == 8)
755  {
756  // Entry moves in its whole to procI
757  oldToNew[index] = newSplitCells.size();
758  newSplitCells.append(splitCells_[index]);
759 
760  //Pout<< "Added oldCell " << index
761  // << " info " << newSplitCells.last()
762  // << " at position " << newSplitCells.size()-1
763  // << endl;
764  }
765  }
766 
767  // Add live cells that are subsetted.
768  forAll(visibleCells_, cellI)
769  {
770  label index = visibleCells_[cellI];
771 
772  if (index >= 0 && destination[cellI] == procI)
773  {
774  label parent = splitCells_[index].parent_;
775 
776  //Pout<< "Adding refined cell " << cellI
777  // << " since moves to "
778  // << procI << " old parent:" << parent << endl;
779 
780  // Create new splitCell with parent
781  oldToNew[index] = newSplitCells.size();
782  newSplitCells.append(splitCell8(parent));
783  }
784  }
785 
786  //forAll(oldToNew, index)
787  //{
788  // Pout<< "old:" << index << " new:" << oldToNew[index]
789  // << endl;
790  //}
791 
792  newSplitCells.shrink();
793 
794  // Renumber contents of newSplitCells
795  forAll(newSplitCells, index)
796  {
797  splitCell8& split = newSplitCells[index];
798 
799  if (split.parent_ >= 0)
800  {
801  split.parent_ = oldToNew[split.parent_];
802  }
803  if (split.addedCellsPtr_.valid())
804  {
805  FixedList<label, 8>& splits = split.addedCellsPtr_();
806 
807  forAll(splits, i)
808  {
809  if (splits[i] >= 0)
810  {
811  splits[i] = oldToNew[splits[i]];
812  }
813  }
814  }
815  }
816 
817 
818  const labelList& subMap = subCellMap[procI];
819 
820  // New visible cells.
821  labelList newVisibleCells(subMap.size(), -1);
822 
823  forAll(subMap, newCellI)
824  {
825  label oldCellI = subMap[newCellI];
826 
827  label oldIndex = visibleCells_[oldCellI];
828 
829  if (oldIndex >= 0)
830  {
831  newVisibleCells[newCellI] = oldToNew[oldIndex];
832  }
833  }
834 
835  //Pout<< nl << "--Subset for domain:" << procI << endl;
836  //writeDebug(newVisibleCells, newSplitCells);
837  //Pout<< "---------" << nl << endl;
838 
839 
840  // Send to neighbours
841  OPstream toNbr(Pstream::blocking, procI);
842  toNbr << newSplitCells << newVisibleCells;
843  }
844 
845 
846  // Receive from neighbours and merge

917 void Foam::refinementHistory::compact()
918 {
919  if (debug)
920  {
921  Pout<< "refinementHistory::compact() Entering with:"
922  << " freeSplitCells_:" << freeSplitCells_.size()
923  << " splitCells_:" << splitCells_.size()
924  << " visibleCells_:" << visibleCells_.size()
925  << endl;
926 
927  // Check all free splitCells are marked as such
928  forAll(freeSplitCells_, i)
929  {
930  label index = freeSplitCells_[i];
931 
932  if (splitCells_[index].parent_ != -2)
933  {
934  FatalErrorInFunction
935  << "Problem index:" << index
936  << abort(FatalError);
937  }
938  }
939 
940  // Check none of the visible cells are marked as free
941  forAll(visibleCells_, cellI)
942  {
943  if
944  (
945  visibleCells_[cellI] >= 0
946  && splitCells_[visibleCells_[cellI]].parent_ == -2
947  )
948  {
949  FatalErrorInFunction
950  << "Problem : visible cell:" << cellI
951  << " is marked as being free." << abort(FatalError);
952  }
953  }
954  }
955 
956  DynamicList<splitCell8> newSplitCells(splitCells_.size());
957 
958  // From uncompacted to compacted splitCells.
959  labelList oldToNew(splitCells_.size(), -1);
960 
961  // Mark all used splitCell entries. These are either indexed by visibleCells
962  // or indexed from other splitCell entries.
963 
964  // Mark from visibleCells
965  forAll(visibleCells_, cellI)
966  {
967  label index = visibleCells_[cellI];
968 
969  if (index >= 0)
970  {
971  // Make sure we only mark visible indices if they either have a
972  // parent or subsplits.
973  if
974  (
975  splitCells_[index].parent_ != -1
976  || splitCells_[index].addedCellsPtr_.valid()
977  )
978  {
979  markSplit(index, oldToNew, newSplitCells);
980  }
981  }
982  }
983 
984  // Mark from splitCells
985  forAll(splitCells_, index)
986  {
987  if (splitCells_[index].parent_ == -2)
988  {
989  // freed cell.
990  }
991  else if
992  (
993  splitCells_[index].parent_ == -1
994  && splitCells_[index].addedCellsPtr_.empty()
995  )
996  {
997  // recombined cell. No need to keep since no parent and no subsplits
998  // Note that gets marked if reachable from other index!
999  }
1000  else
1001  {
1002  // Is used element.
1003  markSplit(index, oldToNew, newSplitCells);
1004  }
1005  }
1006 
1007 
1008  // Now oldToNew is fully complete and compacted elements are in
1009  // newSplitCells.
1010  // Renumber contents of newSplitCells and visibleCells.
1011  forAll(newSplitCells, index)
1012  {
1013  splitCell8& split = newSplitCells[index];
1014 
1015  if (split.parent_ >= 0)
1016  {
1017  split.parent_ = oldToNew[split.parent_];
1018  }
1019  if (split.addedCellsPtr_.valid())
1020  {
1021  FixedList<label, 8>& splits = split.addedCellsPtr_();
1022 
1023  forAll(splits, i)
1024  {
1025  if (splits[i] >= 0)
1026  {
1027  splits[i] = oldToNew[splits[i]];
1028  }
1029  }
1030  }
1031  }
1032 
1033 
1034  if (debug)
1035  {
1036  Pout<< "refinementHistory::compact : compacted splitCells from "
1037  << splitCells_.size() << " to " << newSplitCells.size() << endl;
1038  }
1039 
1040  splitCells_.transfer(newSplitCells);
1041  freeSplitCells_.clearStorage();
1042 
1043 
1044  if (debug)
1045  {
1046  Pout<< "refinementHistory::compact() NOW:"
1047  << " freeSplitCells_:" << freeSplitCells_.size()
1048  << " splitCells_:" << splitCells_.size()
1049  << " newSplitCells:" << newSplitCells.size()
1050  << " visibleCells_:" << visibleCells_.size()
1051  << endl;
1052  }
1053 
1054 
1055  // Adapt indices in visibleCells_
1056  forAll(visibleCells_, cellI)
1057  {
1058  label index = visibleCells_[cellI];
1059 
1060  if (index >= 0)
1061  {
1062  // Note that oldToNew can be -1 so it resets newVisibleCells.
1063  visibleCells_[cellI] = oldToNew[index];
1064  }
1065  else
1066  {
1067  // Keep -1 value.
1068  }
1069  }
1070 }
