88 (

100  scalar s2 = sin(qj.z());
101  scalar c2 = cos(qj.z());
102 
103  J.X.E() = tensor
104  (
105  c0*c1, s0*c1, -s1,
106  c0*s1*s2 - s0*c2, s0*s1*s2 + c0*c2, c1*s2,
107  c0*s1*c2 + s0*s2, s0*s1*c2 - c0*s2, c1*c2
108  );
109  J.X.r() = Zero;
110 
111  J.S = Zero;
112  J.S.xx() = -s1;
113  J.S.xz() = 1;
114  J.S.yx() = c1*s2;
115  J.S.yy() = c2;
116  J.S.zx() = c1*c2;
117  J.S.zy() = -s2;
118

120  J.v = J.S & qDotj;
121

125

127  + c1*c2*qDotj.x()*qDotj.z()
128  - s2*qDotj.y()*qDotj.z(),
129 
130  -s1*c2*qDotj.x()*qDotj.y()
131  - c1*s2*qDotj.x()*qDotj.z()
