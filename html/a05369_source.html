35 template<class RdeltaTType, class RhoType, class SpType, class SuType>
36 void Foam::MULES::explicitSolve
37 (
38  const RdeltaTType& rDeltaT,
39  const RhoType& rho,
40  volScalarField& psi,
41  const surfaceScalarField& phiPsi,
42  const SpType& Sp,
43  const SuType& Su
44 )
45 {
46  Info<< "MULES: Solving for " << psi.name() << endl;
47 
48  const fvMesh& mesh = psi.mesh();
49 
50  scalarField& psiIf = psi;
51  const scalarField& psi0 = psi.oldTime();
52 
53  psiIf = 0.0;
54  fvc::surfaceIntegrate(psiIf, phiPsi);
55 
56  if (mesh.moving())
57  {
58  psiIf =
59  (
60  mesh.Vsc0()().field()*rho.oldTime().field()
61  *psi0*rDeltaT/mesh.Vsc()().field()
62  + Su.field()
63  - psiIf
64  )/(rho.field()*rDeltaT - Sp.field());
65  }
66  else
67  {
68  psiIf =
69  (
70  rho.oldTime().field()*psi0*rDeltaT
71  + Su.field()
72  - psiIf
73  )/(rho.field()*rDeltaT - Sp.field());
74  }
75 
76  psi.correctBoundaryConditions();
77 }

141 template<class RdeltaTType, class RhoType, class SpType, class SuType>
142 void Foam::MULES::limiter
143 (
144  scalarField& allLambda,
145  const RdeltaTType& rDeltaT,
146  const RhoType& rho,
147  const volScalarField& psi,
148  const surfaceScalarField& phiBD,
149  const surfaceScalarField& phiCorr,
150  const SpType& Sp,
151  const SuType& Su,
152  const scalar psiMax,
153  const scalar psiMin,
154  const label nLimiterIter
155 )
156 {
157  const scalarField& psiIf = psi;
158  const volScalarField::GeometricBoundaryField& psiBf = psi.boundaryField();
159 
160  const scalarField& psi0 = psi.oldTime();
161 
162  const fvMesh& mesh = psi.mesh();
163 
164  const labelUList& owner = mesh.owner();
165  const labelUList& neighb = mesh.neighbour();
166  tmp<volScalarField::DimensionedInternalField> tVsc = mesh.Vsc();
167  const scalarField& V = tVsc();
168 
169  const scalarField& phiBDIf = phiBD;
170  const surfaceScalarField::GeometricBoundaryField& phiBDBf =
171  phiBD.boundaryField();
172 
173  const scalarField& phiCorrIf = phiCorr;
174  const surfaceScalarField::GeometricBoundaryField& phiCorrBf =
175  phiCorr.boundaryField();
176 
177  slicedSurfaceScalarField lambda
178  (
179  IOobject
180  (
181  "lambda",
182  mesh.time().timeName(),
183  mesh,
184  IOobject::NO_READ,
185  IOobject::NO_WRITE,
186  false
187  ),
188  mesh,
189  dimless,
190  allLambda,
191  false // Use slices for the couples
192  );
193 
194  scalarField& lambdaIf = lambda;
195  surfaceScalarField::GeometricBoundaryField& lambdaBf =
196  lambda.boundaryField();
197 
198  scalarField psiMaxn(psiIf.size(), psiMin);
199  scalarField psiMinn(psiIf.size(), psiMax);
200 
201  scalarField sumPhiBD(psiIf.size(), 0.0);
202 
203  scalarField sumPhip(psiIf.size(), VSMALL);
204  scalarField mSumPhim(psiIf.size(), VSMALL);
205 
206  forAll(phiCorrIf, facei)
207  {
208  label own = owner[facei];
209  label nei = neighb[facei];
210 
211  psiMaxn[own] = max(psiMaxn[own], psiIf[nei]);
212  psiMinn[own] = min(psiMinn[own], psiIf[nei]);
213 
214  psiMaxn[nei] = max(psiMaxn[nei], psiIf[own]);
215  psiMinn[nei] = min(psiMinn[nei], psiIf[own]);
216 
217  sumPhiBD[own] += phiBDIf[facei];
218  sumPhiBD[nei] -= phiBDIf[facei];
219 
220  scalar phiCorrf = phiCorrIf[facei];
221 
222  if (phiCorrf > 0.0)
223  {
224  sumPhip[own] += phiCorrf;
225  mSumPhim[nei] += phiCorrf;
226  }
227  else
228  {
229  mSumPhim[own] -= phiCorrf;
230  sumPhip[nei] -= phiCorrf;
231  }
232  }
233 
234  forAll(phiCorrBf, patchi)
235  {
236  const fvPatchScalarField& psiPf = psiBf[patchi];
237  const scalarField& phiBDPf = phiBDBf[patchi];
238  const scalarField& phiCorrPf = phiCorrBf[patchi];
239 
240  const labelList& pFaceCells = mesh.boundary()[patchi].faceCells();
241 
242  if (psiPf.coupled())
243  {
244  const scalarField psiPNf(psiPf.patchNeighbourField());
245 
246  forAll(phiCorrPf, pFacei)
247  {
248  label pfCelli = pFaceCells[pFacei];
249 
250  psiMaxn[pfCelli] = max(psiMaxn[pfCelli], psiPNf[pFacei]);
251  psiMinn[pfCelli] = min(psiMinn[pfCelli], psiPNf[pFacei]);
252  }
253  }
254  else
255  {
256  forAll(phiCorrPf, pFacei)
257  {
258  label pfCelli = pFaceCells[pFacei];
259 
260  psiMaxn[pfCelli] = max(psiMaxn[pfCelli], psiPf[pFacei]);
261  psiMinn[pfCelli] = min(psiMinn[pfCelli], psiPf[pFacei]);
262  }
263  }
264 
265  forAll(phiCorrPf, pFacei)
266  {
267  label pfCelli = pFaceCells[pFacei];
268 
269  sumPhiBD[pfCelli] += phiBDPf[pFacei];
270 
271  scalar phiCorrf = phiCorrPf[pFacei];
272 
273  if (phiCorrf > 0.0)
274  {
275  sumPhip[pfCelli] += phiCorrf;
276  }
277  else
278  {
279  mSumPhim[pfCelli] -= phiCorrf;
280  }
281  }
282  }
283 
284  psiMaxn = min(psiMaxn, psiMax);
285  psiMinn = max(psiMinn, psiMin);
286 
287  //scalar smooth = 0.5;
288  //psiMaxn = min((1.0 - smooth)*psiIf + smooth*psiMaxn, psiMax);
289  //psiMinn = max((1.0 - smooth)*psiIf + smooth*psiMinn, psiMin);
290 
291  if (mesh.moving())
292  {
293  tmp<volScalarField::DimensionedInternalField> V0 = mesh.Vsc0();
294 
295  psiMaxn =
296  V
297  *(
298  (rho.field()*rDeltaT - Sp.field())*psiMaxn
299  - Su.field()
300  )
301  - (V0().field()*rDeltaT)*rho.oldTime().field()*psi0
302  + sumPhiBD;
303 
304  psiMinn =
305  V
306  *(
307  Su.field()
308  - (rho.field()*rDeltaT - Sp.field())*psiMinn
309  )
310  + (V0().field()*rDeltaT)*rho.oldTime().field()*psi0
311  - sumPhiBD;
312  }
313  else
314  {
315  psiMaxn =
316  V
317  *(
318  (rho.field()*rDeltaT - Sp.field())*psiMaxn
319  - Su.field()
320  - (rho.oldTime().field()*rDeltaT)*psi0
321  )
322  + sumPhiBD;
323 
324  psiMinn =
325  V
326  *(
327  Su.field()
328  - (rho.field()*rDeltaT - Sp.field())*psiMinn
329  + (rho.oldTime().field()*rDeltaT)*psi0
330  )
331  - sumPhiBD;
332  }
333 
334  scalarField sumlPhip(psiIf.size());
335  scalarField mSumlPhim(psiIf.size());
336 
337  for (int j=0; j<nLimiterIter; j++)
338  {
339  sumlPhip = 0.0;
340  mSumlPhim = 0.0;
341 
342  forAll(lambdaIf, facei)
343  {
344  label own = owner[facei];
345  label nei = neighb[facei];
346 
347  scalar lambdaPhiCorrf = lambdaIf[facei]*phiCorrIf[facei];
348 
349  if (lambdaPhiCorrf > 0.0)
350  {
351  sumlPhip[own] += lambdaPhiCorrf;
352  mSumlPhim[nei] += lambdaPhiCorrf;
353  }
354  else
355  {
356  mSumlPhim[own] -= lambdaPhiCorrf;
357  sumlPhip[nei] -= lambdaPhiCorrf;
358  }
359  }
360 
361  forAll(lambdaBf, patchi)
362  {
363  scalarField& lambdaPf = lambdaBf[patchi];
364  const scalarField& phiCorrfPf = phiCorrBf[patchi];
365 
366  const labelList& pFaceCells = mesh.boundary()[patchi].faceCells();
367 
368  forAll(lambdaPf, pFacei)
369  {
370  label pfCelli = pFaceCells[pFacei];
371 
372  scalar lambdaPhiCorrf = lambdaPf[pFacei]*phiCorrfPf[pFacei];
373 
374  if (lambdaPhiCorrf > 0.0)
375  {
376  sumlPhip[pfCelli] += lambdaPhiCorrf;
377  }
378  else
379  {
380  mSumlPhim[pfCelli] -= lambdaPhiCorrf;
381  }
382  }
383  }
384 
385  forAll(sumlPhip, celli)
386  {
387  sumlPhip[celli] =
388  max(min
389  (
390  (sumlPhip[celli] + psiMaxn[celli])
391  /(mSumPhim[celli] - SMALL),
392  1.0), 0.0
393  );
394 
395  mSumlPhim[celli] =
396  max(min
397  (
398  (mSumlPhim[celli] + psiMinn[celli])
399  /(sumPhip[celli] + SMALL),
400  1.0), 0.0
401  );
402  }
403 
404  const scalarField& lambdam = sumlPhip;
405  const scalarField& lambdap = mSumlPhim;
406 
407  forAll(lambdaIf, facei)
408  {
409  if (phiCorrIf[facei] > 0.0)
410  {
411  lambdaIf[facei] = min
412  (
413  lambdaIf[facei],
414  min(lambdap[owner[facei]], lambdam[neighb[facei]])
415  );
416  }
417  else
418  {
419  lambdaIf[facei] = min
420  (
421  lambdaIf[facei],
422  min(lambdam[owner[facei]], lambdap[neighb[facei]])
423  );
424  }
425  }
426 
427 
428  forAll(lambdaBf, patchi)
429  {
430  fvsPatchScalarField& lambdaPf = lambdaBf[patchi];
431  const scalarField& phiCorrfPf = phiCorrBf[patchi];
432  const fvPatchScalarField& psiPf = psiBf[patchi];
433 
434  if (isA<wedgeFvPatch>(mesh.boundary()[patchi]))
435  {
436  lambdaPf = 0;
437  }
438  else if (psiPf.coupled())
439  {
440  const labelList& pFaceCells =
441  mesh.boundary()[patchi].faceCells();
442 
443  forAll(lambdaPf, pFacei)
444  {
445  label pfCelli = pFaceCells[pFacei];
446 
447  if (phiCorrfPf[pFacei] > 0.0)
448  {
449  lambdaPf[pFacei] =
450  min(lambdaPf[pFacei], lambdap[pfCelli]);
451  }
452  else
453  {
454  lambdaPf[pFacei] =
455  min(lambdaPf[pFacei], lambdam[pfCelli]);
456  }
457  }
458  }
459  else
460  {
461  const labelList& pFaceCells =
462  mesh.boundary()[patchi].faceCells();
463  const scalarField& phiBDPf = phiBDBf[patchi];
464  const scalarField& phiCorrPf = phiCorrBf[patchi];
465 
466  forAll(lambdaPf, pFacei)
467  {
468  // Limit outlet faces only
469  if ((phiBDPf[pFacei] + phiCorrPf[pFacei]) > SMALL*SMALL)
470  {
471  label pfCelli = pFaceCells[pFacei];
472 
473  if (phiCorrfPf[pFacei] > 0.0)
474  {
475  lambdaPf[pFacei] =
476  min(lambdaPf[pFacei], lambdap[pfCelli]);
477  }
478  else
479  {
480  lambdaPf[pFacei] =
481  min(lambdaPf[pFacei], lambdam[pfCelli]);
482  }
483  }
484  }
485  }
486  }
487 
488  syncTools::syncFaceList(mesh, allLambda, minEqOp<scalar>());
489  }

493 template<class RdeltaTType, class RhoType, class SpType, class SuType>
494 void Foam::MULES::limit
495 (
496  const RdeltaTType& rDeltaT,
497  const RhoType& rho,
498  const volScalarField& psi,
499  const surfaceScalarField& phi,
500  surfaceScalarField& phiPsi,
501  const SpType& Sp,
502  const SuType& Su,
503  const scalar psiMax,
504  const scalar psiMin,
505  const label nLimiterIter,
506  const bool returnCorr
507 )
508 {
509  const fvMesh& mesh = psi.mesh();
510 
511  surfaceScalarField phiBD(upwind<scalar>(psi.mesh(), phi).flux(psi));
512 
513  surfaceScalarField& phiCorr = phiPsi;
514  phiCorr -= phiBD;
515 
516  scalarField allLambda(mesh.nFaces(), 1.0);
517 
518  slicedSurfaceScalarField lambda
519  (
520  IOobject
521  (
522  "lambda",
523  mesh.time().timeName(),
524  mesh,
525  IOobject::NO_READ,
526  IOobject::NO_WRITE,
527  false
528  ),
529  mesh,
530  dimless,
531  allLambda,
532  false // Use slices for the couples
533  );
534 
535  limiter
536  (
537  allLambda,
538  rDeltaT,
539  rho,
540  psi,
541  phiBD,
542  phiCorr,
543  Sp,
544  Su,
545  psiMax,
546  psiMin,
547  nLimiterIter
548  );
549 
550  if (returnCorr)
551  {
552  phiCorr *= lambda;
553  }
554  else
555  {
556  phiPsi = phiBD + lambda*phiCorr;
557  }
558 }
