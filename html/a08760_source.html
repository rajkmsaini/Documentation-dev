120 (

128  }
129  else

132  const labelList& nei = faceNeighbour();

134

140

142  storage.append(own[faceI]);
143

145  if (faceI < nInternalFaces())
146  {
147  storage.append(nei[faceI]);
148  }
149  }
150

155

158  {

163  }
