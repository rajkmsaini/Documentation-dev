74 (
75  const UList<septernion>& ss,
76  const UList<scalar> w
77 )
78 {
79  septernion sa(w[0]*ss[0]);
80 
81  for (label i=1; i<ss.size(); i++)
82  {
83  sa.t() += w[i]*ss[i].t();
84 
85  // Invert quaternion if it has the opposite sign to the average
86  if ((sa.r() & ss[i].r()) > 0)
87  {
88  sa.r() += w[i]*ss[i].r();
89  }
90  else

93  }
94  }
95
