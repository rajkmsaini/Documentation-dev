114 void Foam::prghTotalHydrostaticPressureFvPatchScalarField::updateCoeffs()
115 {
116  if (updated())
117  {
118  return;
119  }
120 
121  const scalarField& rhop =
122  patch().lookupPatchField<volScalarField, scalar>(rhoName_);
123 
124  const scalarField& ph_rghp =
125  patch().lookupPatchField<volScalarField, scalar>(ph_rghName_);
126 
127  const scalarField& phip =
128  patch().lookupPatchField<surfaceScalarField, scalar>(phiName_);
129 
130  const vectorField& Up =
131  patch().lookupPatchField<volVectorField, vector>(UName_);
132 
133  operator==
134  (
135  ph_rghp
136  - 0.5*rhop*(1.0 - pos(phip))*magSqr(Up)
137  );
138 
139  fixedValueFvPatchScalarField::updateCoeffs();
140 }
