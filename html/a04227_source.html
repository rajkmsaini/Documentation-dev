89 template<class CombThermoType, class ThermoType>
90 void FSD<CombThermoType, ThermoType>::calculateSourceNorm()
91 {
92  this->singleMixturePtr_->fresCorrect();
93 
94  const label fuelI = this->singleMixturePtr_->fuelIndex();
95 
96  const volScalarField& YFuel = this->thermoPtr_->composition().Y()[fuelI];
97 
98  const volScalarField& YO2 = this->thermoPtr_->composition().Y("O2");
99 
100  const dimensionedScalar s = this->singleMixturePtr_->s();
101 
102  ft_ =
103  (s*YFuel - (YO2 - YO2OxiStream_))/(s*YFuelFuelStream_ + YO2OxiStream_);
104 
105 
106  volVectorField nft(fvc::grad(ft_));
107 
108  volScalarField mgft(mag(nft));
109 
110  surfaceVectorField SfHat(this->mesh().Sf()/this->mesh().magSf());
111 
112  volScalarField cAux(scalar(1) - ft_);
113 
114  dimensionedScalar dMgft = 1.0e-3*
115  (ft_*cAux*mgft)().weightedAverage(this->mesh().V())
116  /((ft_*cAux)().weightedAverage(this->mesh().V()) + SMALL)
117  + dimensionedScalar("ddMgft", mgft.dimensions(), SMALL);
118 
119  mgft += dMgft;
120 
121  nft /= mgft;
122 
123  const volVectorField& U = YO2.db().lookupObject<volVectorField>("U");
124 
125  const volScalarField sigma
126  (
127  (nft & nft)*fvc::div(U) - (nft & fvc::grad(U) & nft)
128  );
129 
130  reactionRateFlameArea_->correct(sigma);
131 
132  const volScalarField& omegaFuel = reactionRateFlameArea_->omega();
133 
134 
135  const scalar ftStoich =
136  YO2OxiStream_.value()
137  /(
138  s.value()*YFuelFuelStream_.value() + YO2OxiStream_.value()
139  );
140 
141  tmp<volScalarField> tPc
142  (
143  new volScalarField
144  (
145  IOobject
146  (
147  "Pc",
148  U.time().timeName(),
149  U.db(),
150  IOobject::NO_READ,
151  IOobject::NO_WRITE
152  ),
153  U.mesh(),
154  dimensionedScalar("Pc", dimless, 0)
155  )
156  );
157 
158  volScalarField& pc = tPc();
159 
160  tmp<volScalarField> tomegaFuel
161  (
162  new volScalarField
163  (
164  IOobject
165  (
166  "omegaFuelBar",
167  U.time().timeName(),
168  U.db(),
169  IOobject::NO_READ,
170  IOobject::NO_WRITE
171  ),
172  U.mesh(),
173  dimensionedScalar
174  (
175  "omegaFuelBar",
176  omegaFuel.dimensions(),
177  0
178  )
179  )
180  );
181 
182  volScalarField& omegaFuelBar = tomegaFuel();
183 
184  // Calculation of the mixture fraction variance (ftVar)
185  const compressible::LESModel& lesModel =
186  YO2.db().lookupObject<compressible::LESModel>("LESProperties");
187 
188  const volScalarField& delta = lesModel.delta();
189  const volScalarField ftVar(Cv_*sqr(delta)*sqr(mgft));
190 
191  // Thickened flame (average flame thickness for counterflow configuration
192  // is 1.5 mm)
193 
194  volScalarField deltaF
195  (
196  lesModel.delta()/dimensionedScalar("flame", dimLength, 1.5e-3)
197  );
198 
199  // Linear correlation between delta and flame thickness
200  volScalarField omegaF(max(deltaF*(4.0/3.0) + (2.0/3.0), scalar(1)));
201 
202  scalar deltaFt = 1.0/ftDim_;
203 
204  forAll(ft_, cellI)
205  {
206  if (ft_[cellI] > ftMin_ && ft_[cellI] < ftMax_)
207  {
208  scalar ftCell = ft_[cellI];
209 
210  if (ftVar[cellI] > ftVarMin_) //sub-grid beta pdf of ft_
211  {
212  scalar ftVarc = ftVar[cellI];
213  scalar a =
214  max(ftCell*(ftCell*(1.0 - ftCell)/ftVarc - 1.0), 0.0);
215  scalar b = max(a/ftCell - a, 0.0);
216 
217  for (int i=1; i<ftDim_; i++)
218  {
219  scalar ft = i*deltaFt;
220  pc[cellI] += pow(ft, a-1.0)*pow(1.0 - ft, b - 1.0)*deltaFt;
221  }
222 
223  for (int i=1; i<ftDim_; i++)
224  {
225  scalar ft = i*deltaFt;
226  omegaFuelBar[cellI] +=
227  omegaFuel[cellI]/omegaF[cellI]
228  *exp
229  (
230  -sqr(ft - ftStoich)
231  /(2.0*sqr(0.01*omegaF[cellI]))
232  )
233  *pow(ft, a - 1.0)
234  *pow(1.0 - ft, b - 1.0)
235  *deltaFt;
236  }
237  omegaFuelBar[cellI] /= max(pc[cellI], 1e-4);
238  }
239  else
240  {
241  omegaFuelBar[cellI] =
242  omegaFuel[cellI]/omegaF[cellI]
243  *exp(-sqr(ftCell - ftStoich)/(2.0*sqr(0.01*omegaF[cellI])));
244  }
245  }
246  else
247  {
248  omegaFuelBar[cellI] = 0.0;
249  }
250  }
251 
252 
253  // Combustion progress variable, c
254 
255  List<label> productsIndex(2, label(-1));
256  {
257  label i = 0;
258  forAll(this->singleMixturePtr_->specieProd(), specieI)
259  {
260  if (this->singleMixturePtr_->specieProd()[specieI] < 0)
261  {
262  productsIndex[i] = specieI;
263  i++;
264  }
265  }
266  }
267 
268 
269  // Flamelet probability of the progress c based on IFC (reuse pc)
270  scalar YprodTotal = 0;
271  forAll(productsIndex, j)
272  {
273  YprodTotal += this->singleMixturePtr_->Yprod0()[productsIndex[j]];
274  }
275 
276  forAll(ft_, cellI)
277  {
278  if (ft_[cellI] < ftStoich)
279  {
280  pc[cellI] = ft_[cellI]*(YprodTotal/ftStoich);
281  }
282  else
283  {
284  pc[cellI] = (1.0 - ft_[cellI])*(YprodTotal/(1.0 - ftStoich));
285  }
286  }
287 
288  tmp<volScalarField> tproducts
289  (
290  new volScalarField
291  (
292  IOobject
293  (
294  "products",
295  U.time().timeName(),
296  U.db(),
297  IOobject::NO_READ,
298  IOobject::NO_WRITE
299  ),
300  U.mesh(),
301  dimensionedScalar("products", dimless, 0)
302  )
303  );
304 
305  volScalarField& products = tproducts();
306 
307  forAll(productsIndex, j)
308  {
309  label specieI = productsIndex[j];
310  const volScalarField& Yp = this->thermoPtr_->composition().Y()[specieI];
311  products += Yp;
312  }
313 
314  volScalarField c
315  (
316  max(scalar(1) - products/max(pc, scalar(1e-5)), scalar(0))
317  );
318 
319  pc = min(C_*c, scalar(1));
320 
321  const volScalarField fres(this->singleMixturePtr_->fres(fuelI));
322 
323  this->wFuel_ == mgft*pc*omegaFuelBar;
324 }
