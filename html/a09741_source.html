51 
52  if (Pstream::master())
53  {
54  // Combine values into single field
55  Field<Type> allValues

57  ListListOps::combine<Field<Type>>

59  gatheredValues,
60  accessOp<Field<Type>>()

62  );

66  {

69  }
70
