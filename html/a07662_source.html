45 void* Foam::codedBase::loadLibrary
46 (
47  const fileName& libPath,
48  const string& globalFuncName,
49  const dictionary& contextDict
50 ) const
51 {
52  void* lib = 0;
53 
54  // avoid compilation by loading an existing library
55  if (!libPath.empty())
56  {
57  if (libs().open(libPath, false))
58  {
59  lib = libs().findLibrary(libPath);
60 
61  // verify the loaded version and unload if needed
62  if (lib)
63  {
64  // provision for manual execution of code after loading
65  if (dlSymFound(lib, globalFuncName))
66  {
67  loaderFunctionType function =
68  reinterpret_cast<loaderFunctionType>
69  (
70  dlSym(lib, globalFuncName)
71  );
72 
73  if (function)
74  {
75  (*function)(true); // force load
76  }
77  else
78  {
79  FatalIOErrorIn
80  (
81  "codedBase::updateLibrary()",
82  contextDict
83  ) << "Failed looking up symbol " << globalFuncName
84  << nl << "from " << libPath << exit(FatalIOError);
85  }
86  }
87  else
88  {
89  FatalIOErrorIn
90  (
91  "codedBase::loadLibrary()",
92  contextDict
93  ) << "Failed looking up symbol " << globalFuncName << nl
94  << "from " << libPath << exit(FatalIOError);
95 
96  lib = 0;
97  if (!libs().close(libPath, false))
98  {
99  FatalIOErrorIn
100  (
101  "codedBase::loadLibrary()",
102  contextDict
103  ) << "Failed unloading library "
104  << libPath
105  << exit(FatalIOError);
106  }
107  }
108  }
109  }
110  }
111 
112  return lib;
113 }

175 void Foam::codedBase::createLibrary
176 (
177  dynamicCode& dynCode,
178  const dynamicCodeContext& context
179 ) const
180 {
181  bool create =
182  Pstream::master()
183  || (regIOobject::fileModificationSkew <= 0); // not NFS
184 
185  if (create)
186  {
187  // Write files for new library
188  if (!dynCode.upToDate(context))
189  {
190  // filter with this context
191  dynCode.reset(context);
192 
193  this->prepare(dynCode, context);
194 
195  if (!dynCode.copyOrCreateFiles(true))
196  {
197  FatalIOErrorIn
198  (
199  "codedBase::createLibrary(..)",
200  context.dict()
201  ) << "Failed writing files for" << nl
202  << dynCode.libRelPath() << nl
203  << exit(FatalIOError);
204  }
205  }
206 
207  if (!dynCode.wmakeLibso())
208  {
209  FatalIOErrorIn
210  (
211  "codedBase::createLibrary(..)",
212  context.dict()
213  ) << "Failed wmake " << dynCode.libRelPath() << nl
214  << exit(FatalIOError);
215  }
216  }
217 
218 
219  // all processes must wait for compile to finish
220  if (regIOobject::fileModificationSkew > 0)
221  {
222  //- Since the library has only been compiled on the master the
223  // other nodes need to pick this library up through NFS
224  // We do this by just polling a few times using the
225  // fileModificationSkew.
226 
227  const fileName libPath = dynCode.libPath();
228 
229  off_t mySize = Foam::fileSize(libPath);
230  off_t masterSize = mySize;
231  Pstream::scatter(masterSize);
232 
233  if (debug)
234  {
235  Pout<< endl<< "on processor " << Pstream::myProcNo()
236  << " have masterSize:" << masterSize
237  << " and localSize:" << mySize
238  << endl;
239  }
240 
241 
242  if (mySize < masterSize)
243  {
244  if (debug)
245  {
246  Pout<< "Local file " << libPath
247  << " not of same size (" << mySize
248  << ") as master ("
249  << masterSize << "). Waiting for "
250  << regIOobject::fileModificationSkew
251  << " seconds." << endl;
252  }
253  Foam::sleep(regIOobject::fileModificationSkew);
254 
255  // Recheck local size
256  mySize = Foam::fileSize(libPath);
257 
258  if (mySize < masterSize)
259  {
260  FatalIOErrorIn
261  (
262  "functionEntries::codeStream::execute(..)",
263  context.dict()
264  ) << "Cannot read (NFS mounted) library " << nl
265  << libPath << nl
266  << "on processor " << Pstream::myProcNo()
267  << " detected size " << mySize
268  << " whereas master size is " << masterSize
269  << " bytes." << nl
270  << "If your case is not NFS mounted"
271  << " (so distributed) set fileModificationSkew"
272  << " to 0"
273  << exit(FatalIOError);
274  }
275  }
276 
277  if (debug)
278  {
279  Pout<< endl<< "on processor " << Pstream::myProcNo()
280  << " after waiting: have masterSize:" << masterSize
281  << " and localSize:" << mySize
282  << endl;
283  }
284  }
285  reduce(create, orOp<bool>());
286 }

289 void Foam::codedBase::updateLibrary
290 (
291  const word& redirectType
292 ) const
293 {
294  const dictionary& dict = this->codeDict();
295 
296  dynamicCode::checkSecurity
297  (
298  "codedBase::updateLibrary()",
299  dict
300  );
301 
302  dynamicCodeContext context(dict);
303 
304  // codeName: redirectType + _<sha1>
305  // codeDir : redirectType
306  dynamicCode dynCode
307  (
308  redirectType + context.sha1().str(true),
309  redirectType
310  );
311  const fileName libPath = dynCode.libPath();
312 
313 
314  // the correct library was already loaded => we are done
315  if (libs().findLibrary(libPath))
316  {
317  return;
318  }
319 
320  Info<< "Using dynamicCode for " << this->description().c_str()
321  << " at line " << dict.startLineNumber()
322  << " in " << dict.name() << endl;
323 
324 
325  // remove instantiation of fvPatchField provided by library
326  this->clearRedirect();
327 
328  // may need to unload old library
329  unloadLibrary
330  (
331  oldLibPath_,
332  dynamicCode::libraryBaseName(oldLibPath_),
333  context.dict()
334  );
335 
336  // try loading an existing library (avoid compilation when possible)
337  if (!loadLibrary(libPath, dynCode.codeName(), context.dict()))
338  {
339  createLibrary(dynCode, context);
340 
341  loadLibrary(libPath, dynCode.codeName(), context.dict());
342  }
343 
344  // retain for future reference
345  oldLibPath_ = libPath;
346 }
