98 // Calculate band of matrix
99 label getBand(const labelList& owner, const labelList& neighbour)
100 {
101  label band = 0;
102 
103  forAll(neighbour, faceI)
104  {
105  label diff = neighbour[faceI] - owner[faceI];
106

108  {
109  band = diff;
110  }
111  }
112  return band;

116 // Calculate band of matrix
117 void getBand
118 (
119  const bool calculateIntersect,
120  const label nCells,
121  const labelList& owner,
122  const labelList& neighbour,
123  label& bandwidth,
124  scalar& profile, // scalar to avoid overflow
125  scalar& sumSqrIntersect // scalar to avoid overflow
126 )
127 {
128  labelList cellBandwidth(nCells, 0);
129  scalarField nIntersect(nCells, 0.0);
130 
131  forAll(neighbour, faceI)
132  {
133  label own = owner[faceI];
134  label nei = neighbour[faceI];
135 
136  // Note: mag not necessary for correct (upper-triangular) ordering.
137  label diff = nei-own;
138  cellBandwidth[nei] = max(cellBandwidth[nei], diff);
139  }
140 
141  bandwidth = max(cellBandwidth);
142 
143  // Do not use field algebra because of conversion label to scalar
144  profile = 0.0;
145  forAll(cellBandwidth, cellI)
146  {
147  profile += 1.0*cellBandwidth[cellI];
148  }
149 
150  sumSqrIntersect = 0.0;
151  if (calculateIntersect)
152  {
153  forAll(nIntersect, cellI)
154  {
155  for (label colI = cellI-cellBandwidth[cellI]; colI <= cellI; colI++)
156  {
157  nIntersect[colI] += 1.0;
158  }
159  }
160 
161  sumSqrIntersect = sum(Foam::sqr(nIntersect));
162  }
163 }

166 // Determine upper-triangular face order
167 labelList getFaceOrder
168 (
169  const primitiveMesh& mesh,
170  const labelList& cellOrder // New to old cell
171 )
172 {
173  labelList reverseCellOrder(invert(cellOrder.size(), cellOrder));
174 
175  labelList oldToNewFace(mesh.nFaces(), -1);
176 
177  label newFaceI = 0;
178 
179  labelList nbr;
180  labelList order;
181 
182  forAll(cellOrder, newCellI)
183  {
184  label oldCellI = cellOrder[newCellI];
185 
186  const cell& cFaces = mesh.cells()[oldCellI];
187 
188  // Neighbouring cells
189  nbr.setSize(cFaces.size());
190 
191  forAll(cFaces, i)
192  {
193  label faceI = cFaces[i];
194 
195  if (mesh.isInternalFace(faceI))
196  {
197  // Internal face. Get cell on other side.
198  label nbrCellI = reverseCellOrder[mesh.faceNeighbour()[faceI]];
199  if (nbrCellI == newCellI)
200  {
201  nbrCellI = reverseCellOrder[mesh.faceOwner()[faceI]];
202  }
203 
204  if (newCellI < nbrCellI)
205  {
206  // CellI is master
207  nbr[i] = nbrCellI;
208  }
209  else
210  {
211  // nbrCell is master. Let it handle this face.
212  nbr[i] = -1;
213  }
214  }
215  else
216  {
217  // External face. Do later.
218  nbr[i] = -1;
219  }
220  }
221 
222  order.setSize(nbr.size());
223  sortedOrder(nbr, order);
224 
225  forAll(order, i)
226  {
227  label index = order[i];
228  if (nbr[index] != -1)
229  {
230  oldToNewFace[cFaces[index]] = newFaceI++;
231  }
232  }
233  }
234 
235  // Leave patch faces intact.
236  for (label faceI = newFaceI; faceI < mesh.nFaces(); faceI++)
237  {
238  oldToNewFace[faceI] = faceI;
239  }
240 
241 
242  // Check done all faces.
243  forAll(oldToNewFace, faceI)
244  {
245  if (oldToNewFace[faceI] == -1)
246  {
247  FatalErrorIn
248  (
249  "getFaceOrder"
250  "(const primitiveMesh&, const labelList&, const labelList&)"
251  ) << "Did not determine new position" << " for face " << faceI
252  << abort(FatalError);
253  }
254  }
255 
256  return invert(mesh.nFaces(), oldToNewFace);
257 }

260 // Determine face order such that inside region faces are sorted
261 // upper-triangular but inbetween region faces are handled like boundary faces.
262 labelList getRegionFaceOrder
263 (
264  const primitiveMesh& mesh,
265  const labelList& cellOrder, // New to old cell
266  const labelList& cellToRegion // Old cell to region
267 )
268 {
269  labelList reverseCellOrder(invert(cellOrder.size(), cellOrder));
270 
271  labelList oldToNewFace(mesh.nFaces(), -1);
272 
273  label newFaceI = 0;
274 
275  label prevRegion = -1;
276 
277  forAll(cellOrder, newCellI)
278  {
279  label oldCellI = cellOrder[newCellI];
280 
281  if (cellToRegion[oldCellI] != prevRegion)
282  {
283  prevRegion = cellToRegion[oldCellI];
284  }
285 
286  const cell& cFaces = mesh.cells()[oldCellI];
287 
288  SortableList<label> nbr(cFaces.size());
289 
290  forAll(cFaces, i)
291  {
292  label faceI = cFaces[i];
293 
294  if (mesh.isInternalFace(faceI))
295  {
296  // Internal face. Get cell on other side.
297  label nbrCellI = reverseCellOrder[mesh.faceNeighbour()[faceI]];
298  if (nbrCellI == newCellI)
299  {
300  nbrCellI = reverseCellOrder[mesh.faceOwner()[faceI]];
301  }
302 
303  if (cellToRegion[oldCellI] != cellToRegion[cellOrder[nbrCellI]])
304  {
305  // Treat like external face. Do later.
306  nbr[i] = -1;
307  }
308  else if (newCellI < nbrCellI)
309  {
310  // CellI is master
311  nbr[i] = nbrCellI;
312  }
313  else
314  {
315  // nbrCell is master. Let it handle this face.
316  nbr[i] = -1;
317  }
318  }
319  else
320  {
321  // External face. Do later.
322  nbr[i] = -1;
323  }
324  }
325 
326  nbr.sort();
327 
328  forAll(nbr, i)
329  {
330  if (nbr[i] != -1)
331  {
332  oldToNewFace[cFaces[nbr.indices()[i]]] = newFaceI++;
333  }
334  }
335  }
336 
337  // Do region interfaces
338  label nRegions = max(cellToRegion)+1;
339  {
340  // Sort in increasing region
341  SortableList<label> sortKey(mesh.nFaces(), labelMax);
342 
343  for (label faceI = 0; faceI < mesh.nInternalFaces(); faceI++)
344  {
345  label ownRegion = cellToRegion[mesh.faceOwner()[faceI]];
346  label neiRegion = cellToRegion[mesh.faceNeighbour()[faceI]];
347 
348  if (ownRegion != neiRegion)
349  {
350  sortKey[faceI] =
351  min(ownRegion, neiRegion)*nRegions
352  +max(ownRegion, neiRegion);
353  }
354  }
355  sortKey.sort();
356 
357  // Extract.
358  label prevKey = -1;
359  forAll(sortKey, i)
360  {
361  label key = sortKey[i];
362 
363  if (key == labelMax)
364  {
365  break;
366  }
367 
368  if (prevKey != key)
369  {
370  prevKey = key;
371  }
372 
373  oldToNewFace[sortKey.indices()[i]] = newFaceI++;
374  }
375  }
376 
377  // Leave patch faces intact.
378  for (label faceI = newFaceI; faceI < mesh.nFaces(); faceI++)
379  {
380  oldToNewFace[faceI] = faceI;
381  }
382 
383 
384  // Check done all faces.
385  forAll(oldToNewFace, faceI)
386  {
387  if (oldToNewFace[faceI] == -1)
388  {
389  FatalErrorIn
390  (
391  "getRegionFaceOrder"
392  "(const primitveMesh&, const labelList&, const labelList&)"
393  ) << "Did not determine new position"
394  << " for face " << faceI
395  << abort(FatalError);
396  }
397  }
398 
399  return invert(mesh.nFaces(), oldToNewFace);
400 }

403 // cellOrder: old cell for every new cell
404 // faceOrder: old face for every new face. Ordering of boundary faces not
405 // changed.
406 autoPtr<mapPolyMesh> reorderMesh
407 (
408  polyMesh& mesh,
409  const labelList& cellOrder,
410  const labelList& faceOrder
411 )
412 {
413  labelList reverseCellOrder(invert(cellOrder.size(), cellOrder));
414  labelList reverseFaceOrder(invert(faceOrder.size(), faceOrder));
415 
416  faceList newFaces(reorder(reverseFaceOrder, mesh.faces()));
417  labelList newOwner
418  (
419  renumber
420  (
421  reverseCellOrder,
422  reorder(reverseFaceOrder, mesh.faceOwner())
423  )
424  );
425  labelList newNeighbour
426  (
427  renumber
428  (
429  reverseCellOrder,
430  reorder(reverseFaceOrder, mesh.faceNeighbour())
431  )
432  );
433 
434  // Check if any faces need swapping.
435  labelHashSet flipFaceFlux(newOwner.size());
436  forAll(newNeighbour, faceI)
437  {
438  label own = newOwner[faceI];
439  label nei = newNeighbour[faceI];
440 
441  if (nei < own)
442  {
443  newFaces[faceI].flip();
444  Swap(newOwner[faceI], newNeighbour[faceI]);
445  flipFaceFlux.insert(faceI);
446  }
447  }
448 
449  const polyBoundaryMesh& patches = mesh.boundaryMesh();
450  labelList patchSizes(patches.size());
451  labelList patchStarts(patches.size());
452  labelList oldPatchNMeshPoints(patches.size());
453  labelListList patchPointMap(patches.size());
454 
455  forAll(patches, patchI)
456  {
457  patchSizes[patchI] = patches[patchI].size();
458  patchStarts[patchI] = patches[patchI].start();
459  oldPatchNMeshPoints[patchI] = patches[patchI].nPoints();
460  patchPointMap[patchI] = identity(patches[patchI].nPoints());
461  }
462 
463  mesh.resetPrimitives
464  (
465  Xfer<pointField>::null(),
466  xferMove(newFaces),
467  xferMove(newOwner),
468  xferMove(newNeighbour),
469  patchSizes,
470  patchStarts,
471  true
472  );
473 
474 
475  // Re-do the faceZones
476  {
477  faceZoneMesh& faceZones = mesh.faceZones();
478  faceZones.clearAddressing();
479  forAll(faceZones, zoneI)
480  {
481  faceZone& fZone = faceZones[zoneI];
482  labelList newAddressing(fZone.size());
483  boolList newFlipMap(fZone.size());
484  forAll(fZone, i)
485  {
486  label oldFaceI = fZone[i];
487  newAddressing[i] = reverseFaceOrder[oldFaceI];
488  if (flipFaceFlux.found(newAddressing[i]))
489  {
490  newFlipMap[i] = !fZone.flipMap()[i];
491  }
492  else
493  {
494  newFlipMap[i] = fZone.flipMap()[i];
495  }
496  }
497  labelList newToOld;
498  sortedOrder(newAddressing, newToOld);
499  fZone.resetAddressing
500  (
501  UIndirectList<label>(newAddressing, newToOld)(),
502  UIndirectList<bool>(newFlipMap, newToOld)()
503  );
504  }
505  }
506  // Re-do the cellZones
507  {
508  cellZoneMesh& cellZones = mesh.cellZones();
509  cellZones.clearAddressing();
510  forAll(cellZones, zoneI)
511  {
512  cellZones[zoneI] = UIndirectList<label>
513  (
514  reverseCellOrder,
515  cellZones[zoneI]
516  )();
517  Foam::sort(cellZones[zoneI]);
518  }
519  }
520 
521 
522  return autoPtr<mapPolyMesh>
523  (
524  new mapPolyMesh
525  (
526  mesh, // const polyMesh& mesh,
527  mesh.nPoints(), // nOldPoints,
528  mesh.nFaces(), // nOldFaces,
529  mesh.nCells(), // nOldCells,
530  identity(mesh.nPoints()), // pointMap,
531  List<objectMap>(0), // pointsFromPoints,
532  faceOrder, // faceMap,
533  List<objectMap>(0), // facesFromPoints,
534  List<objectMap>(0), // facesFromEdges,
535  List<objectMap>(0), // facesFromFaces,
536  cellOrder, // cellMap,
537  List<objectMap>(0), // cellsFromPoints,
538  List<objectMap>(0), // cellsFromEdges,
539  List<objectMap>(0), // cellsFromFaces,
540  List<objectMap>(0), // cellsFromCells,
541  identity(mesh.nPoints()), // reversePointMap,
542  reverseFaceOrder, // reverseFaceMap,
543  reverseCellOrder, // reverseCellMap,
544  flipFaceFlux, // flipFaceFlux,
545  patchPointMap, // patchPointMap,
546  labelListList(0), // pointZoneMap,
547  labelListList(0), // faceZonePointMap,
548  labelListList(0), // faceZoneFaceMap,
549  labelListList(0), // cellZoneMap,
550  pointField(0), // preMotionPoints,
551  patchStarts, // oldPatchStarts,
552  oldPatchNMeshPoints, // oldPatchNMeshPoints
553  autoPtr<scalarField>() // oldCellVolumes
554  )
555  );
556 }
