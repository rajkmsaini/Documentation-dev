73 scalar restrainedHarmonicSpring::energy(const vector r) const
74 {
75  scalar magR = mag(r);
76 
77  if (magR < rR_)
78  {
79  return 0.5 * springConstant_ * magSqr(r);
80  }
81  else
82  {
83  return 0.5 * springConstant_ * rR_ * rR_
84  + springConstant_ * rR_ * (magR - rR_);
85  }
86 }

89 vector restrainedHarmonicSpring::force(const vector r) const
90 {
91  scalar magR = mag(r);
92 
93  if (magR < rR_)
94  {
95  return -springConstant_ * r;
96  }
97  else
98  {
99  return -springConstant_ * rR_ * r / magR;
