185 template<class Type>
186 Type Foam::PolynomialEntry<Type>::integrate
187 (
188  const scalar x1,
189  const scalar x2
190 ) const
191 {
192  Type intx(pTraits<Type>::zero);
193 
194  if (canIntegrate_)
195  {
196  forAll(coeffs_, i)
197  {
198  intx += cmptMultiply
199  (
200  cmptDivide
201  (
202  coeffs_[i].first(),
203  coeffs_[i].second() + pTraits<Type>::one
204  ),
205  cmptPow
206  (
207  pTraits<Type>::one*x2,
208  coeffs_[i].second() + pTraits<Type>::one
209  )
210  - cmptPow
211  (
212  pTraits<Type>::one*x1,
213  coeffs_[i].second() + pTraits<Type>::one
214  )
215  );
216  }
217  }
218 
219  return intx;
220 }
