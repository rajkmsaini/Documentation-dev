46 inline bool Foam::pimpleControl::correct()
47 {
48  corrPISO_++;
49 
50  if (debug)
51  {
52  Info<< algorithmName_ << " correct: corrPISO = " << corrPISO_ << endl;
53  }
54 
55  if (corrPISO_ <= nCorrPISO_)
56  {
57  return true;

59  else
60  {
61  corrPISO_ = 0;
