83 Foam::rigidBodyMeshMotion::rigidBodyMeshMotion
84 (
85  const polyMesh& mesh,
86  const IOdictionary& dict
87 )
88 :
89  displacementMotionSolver(mesh, dict, typeName),
90  model_
91  (
92  coeffDict(),
93  IOobject
94  (
95  "rigidBodyMotionState",
96  mesh.time().timeName(),
97  "uniform",
98  mesh
99  ).headerOk()
100  ? IOdictionary
101  (
102  IOobject
103  (
104  "rigidBodyMotionState",
105  mesh.time().timeName(),
106  "uniform",
107  mesh,
108  IOobject::READ_IF_PRESENT,
109  IOobject::NO_WRITE,
110  false
111  )
112  )
113  : coeffDict()
114  ),
115  test_(coeffDict().lookupOrDefault<Switch>("test", false)),
116  rhoInf_(1.0),
117  rhoName_(coeffDict().lookupOrDefault<word>("rho", "rho")),
118  curTimeIndex_(-1)
119 {
120  if (rhoName_ == "rhoInf")
121  {
122  rhoInf_ = readScalar(coeffDict().lookup("rhoInf"));
123  }
124 
125  const dictionary& bodiesDict = coeffDict().subDict("bodies");
126 
127  forAllConstIter(IDLList<entry>, bodiesDict, iter)
128  {
129  const dictionary& bodyDict = iter().dict();
130 
131  if (bodyDict.found("patches"))
132  {
133  const label bodyID = model_.bodyID(iter().keyword());
134 
135  if (bodyID == -1)
136  {
137  FatalErrorInFunction
138  << "Body " << iter().keyword()
139  << " has been merged with another body"
140  " and cannot be assigned a set of patches"
141  << exit(FatalError);
142  }
143 
144  bodyMeshes_.append
145  (
146  new bodyMesh
147  (
148  mesh,
149  iter().keyword(),
150  bodyID,
151  bodyDict
152  )
153  );
154  }
155  }
156 
157  // Calculate scaling factor everywhere for each meshed body
158  forAll(bodyMeshes_, bi)
159  {
160  const pointMesh& pMesh = pointMesh::New(mesh);
161 
162  pointPatchDist pDist(pMesh, bodyMeshes_[bi].patchSet_, points0());
163 
164  pointScalarField& scale = bodyMeshes_[bi].weight_;
165 
166  // Scaling: 1 up to di then linear down to 0 at do away from patches
167  scale.primitiveFieldRef() =
168  min
169  (
170  max
171  (
172  (bodyMeshes_[bi].do_ - pDist.primitiveField())
173  /(bodyMeshes_[bi].do_ - bodyMeshes_[bi].di_),
174  scalar(0)
175  ),
176  scalar(1)
177  );
178 
179  // Convert the scale function to a cosine
180  scale.primitiveFieldRef() =
181  min
182  (
183  max
184  (
185  0.5
186  - 0.5
187  *cos(scale.primitiveField()
188  *Foam::constant::mathematical::pi),
189  scalar(0)
190  ),
191  scalar(1)
192  );
193 
194  pointConstraints::New(pMesh).constrain(scale);
195  //scale.write();
196  }
197 }

215 void Foam::rigidBodyMeshMotion::solve()
216 {
217  const Time& t = mesh().time();
218 
219  if (mesh().nPoints() != points0().size())
220  {
221  FatalErrorInFunction
222  << "The number of points in the mesh seems to have changed." << endl
223  << "In constant/polyMesh there are " << points0().size()
224  << " points; in the current mesh there are " << mesh().nPoints()
225  << " points." << exit(FatalError);
226  }
227 
228  // Store the motion state at the beginning of the time-step
229  if (curTimeIndex_ != this->db().time().timeIndex())
230  {
231  model_.newTime();
232  curTimeIndex_ = this->db().time().timeIndex();
233  }
234 
235  if (db().foundObject<uniformDimensionedVectorField>("g"))
236  {
237  model_.g() =
238  db().lookupObject<uniformDimensionedVectorField>("g").value();
239  }
240 
241  if (test_)
242  {
243  label nIter(readLabel(coeffDict().lookup("nIter")));
244 
245  for (label i=0; i<nIter; i++)
246  {
247  model_.solve
248  (
249  t.deltaTValue(),
250  scalarField(model_.nDoF(), Zero),
251  Field<spatialVector>(model_.nBodies(), Zero)
252  );
253  }
254  }
255  else
256  {
257  Field<spatialVector> fx(model_.nBodies(), Zero);
258 
259  forAll(bodyMeshes_, bi)
260  {
261  const label bodyID = bodyMeshes_[bi].bodyID_;
262 
263  dictionary forcesDict;
264  forcesDict.add("type", functionObjects::forces::typeName);
265  forcesDict.add("patches", bodyMeshes_[bi].patches_);
266  forcesDict.add("rhoInf", rhoInf_);
267  forcesDict.add("rho", rhoName_);
268  forcesDict.add("CofR", vector::zero);
269 
270  functionObjects::forces f("forces", db(), forcesDict);
271  f.calcForcesMoment();
272 
273  fx[bodyID] = spatialVector(f.momentEff(), f.forceEff());
274  }
275 
276  model_.solve
277  (
278  t.deltaTValue(),
279  scalarField(model_.nDoF(), Zero),
280  fx
281  );
282  }
283 
284  if (Pstream::master() && model_.report())
285  {
286  forAll(bodyMeshes_, bi)
287  {
288  model_.status(bodyMeshes_[bi].bodyID_);
289  }
290  }
291 
292  // Update the displacements
293  if (bodyMeshes_.size() == 1)
294  {
295  pointDisplacement_.primitiveFieldRef() = model_.transformPoints
296  (
297  bodyMeshes_[0].bodyID_,
298  bodyMeshes_[0].weight_,
299  points0()
300  ) - points0();
301  }
302  else
303  {
304  labelList bodyIDs(bodyMeshes_.size());
305  List<const scalarField*> weights(bodyMeshes_.size());
306  forAll(bodyIDs, bi)
307  {
308  bodyIDs[bi] = bodyMeshes_[bi].bodyID_;
309  weights[bi] = &bodyMeshes_[bi].weight_;
310  }
311 
312  pointDisplacement_.primitiveFieldRef() =
313  model_.transformPoints(bodyIDs, weights, points0()) - points0();
314  }
315 
316  // Displacement has changed. Update boundary conditions
317  pointConstraints::New
318  (
319  pointDisplacement_.mesh()
320  ).constrainDisplacement(pointDisplacement_);
321 }
