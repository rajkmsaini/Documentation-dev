54 Foam::tmp<Foam::Field<Type>> Foam::fieldValues::faceSource::getFieldValues
55 (
56  const word& fieldName,
57  const bool mustGet,
58  const bool applyOrientation
59 ) const
60 {
61  typedef GeometricField<Type, fvsPatchField, surfaceMesh> sf;
62  typedef GeometricField<Type, fvPatchField, volMesh> vf;
63 
64  if (source_ != stSampledSurface && obr_.foundObject<sf>(fieldName))
65  {
66  return filterField(obr_.lookupObject<sf>(fieldName), applyOrientation);
67  }
68  else if (obr_.foundObject<vf>(fieldName))
69  {
70  const vf& fld = obr_.lookupObject<vf>(fieldName);
71 
72  if (surfacePtr_.valid())
73  {
74  if (surfacePtr_().interpolate())
75  {
76  const interpolationCellPoint<Type> interp(fld);
77  tmp<Field<Type>> tintFld(surfacePtr_().interpolate(interp));
78  const Field<Type>& intFld = tintFld();
79 
80  // Average
81  const faceList& faces = surfacePtr_().faces();
82  tmp<Field<Type>> tavg
83  (
84  new Field<Type>(faces.size(), pTraits<Type>::zero)
85  );
86  Field<Type>& avg = tavg.ref();
87 
88  forAll(faces, faceI)
89  {
90  const face& f = faces[faceI];
91  forAll(f, fp)
92  {
93  avg[faceI] += intFld[f[fp]];
94  }
95  avg[faceI] /= f.size();
96  }
97 
98  return tavg;
99  }
100  else
101  {
102  return surfacePtr_().sample(fld);
103  }
104  }
105  else
106  {
107  return filterField(fld, applyOrientation);
108  }
109  }
110 
111  if (mustGet)
112  {
113  FatalErrorInFunction
114  << "Field " << fieldName << " not found in database"
115  << abort(FatalError);
116  }
117 
118  return tmp<Field<Type>>(new Field<Type>(0));
119 }

122 template<class Type>
123 Type Foam::fieldValues::faceSource::processSameTypeValues
124 (
125  const Field<Type>& values,
126  const vectorField& Sf,
127  const scalarField& weightField
128 ) const
129 {
130  Type result = pTraits<Type>::zero;
131  switch (operation_)
132  {
133  case opSum:
134  {
135  result = sum(values);
136  break;
137  }
138  case opSumMag:
139  {
140  result = sum(cmptMag(values));
141  break;
142  }
143  case opSumDirection:
144  {
145  FatalErrorInFunction
146  << "Operation " << operationTypeNames_[operation_]
147  << " not available for values of type "
148  << pTraits<Type>::typeName
149  << exit(FatalError);
150 
151  result = pTraits<Type>::zero;
152  break;
153  }
154  case opSumDirectionBalance:
155  {
156  FatalErrorInFunction
157  << "Operation " << operationTypeNames_[operation_]
158  << " not available for values of type "
159  << pTraits<Type>::typeName
160  << exit(FatalError);
161 
162  result = pTraits<Type>::zero;
163  break;
164  }
165  case opAverage:
166  {
167  result = sum(values)/values.size();
168  break;
169  }
170  case opWeightedAverage:
171  {
172  if (weightField.size())
173  {
174  result = sum(weightField*values)/sum(weightField);
175  }
176  else
177  {
178  result = sum(values)/values.size();
179  }
180  break;
181  }
182  case opAreaAverage:
183  {
184  const scalarField magSf(mag(Sf));
185 
186  result = sum(magSf*values)/sum(magSf);
187  break;
188  }
189  case opWeightedAreaAverage:
190  {
191  const scalarField magSf(mag(Sf));
192 
193  if (weightField.size())
194  {
195  result = sum(weightField*magSf*values)/sum(magSf*weightField);
196  }
197  else
198  {
199  result = sum(magSf*values)/sum(magSf);
200  }
201  break;
202  }
203  case opAreaIntegrate:
204  {
205  const scalarField magSf(mag(Sf));
206 
207  result = sum(magSf*values);
208  break;
209  }
210  case opMin:
211  {
212  result = min(values);
213  break;
214  }
215  case opMax:
216  {
217  result = max(values);
218  break;
219  }
220  case opCoV:
221  {
222  const scalarField magSf(mag(Sf));
223 
224  Type meanValue = sum(values*magSf)/sum(magSf);
225 
226  const label nComp = pTraits<Type>::nComponents;
227 
228  for (direction d=0; d<nComp; ++d)
229  {
230  scalarField vals(values.component(d));
231  scalar mean = component(meanValue, d);
232  scalar& res = setComponent(result, d);
233 
234  res = sqrt(sum(magSf*sqr(vals - mean))/sum(magSf))/mean;
235  }
236 
237  break;
238  }
239  default:
240  {
241  // Do nothing
242  }
243  }
244 
245  return result;
246 }

266 (
267  const word& fieldName,
268  const scalarField& weightField,
269  const bool orient
270 )
271 {
272  const bool ok = validField<Type>(fieldName);
273 
274  if (ok)
275  {
276  Field<Type> values(getFieldValues<Type>(fieldName, true, orient));
277 
278  vectorField Sf;
279  if (surfacePtr_.valid())
280  {
281  // Get oriented Sf
282  Sf = surfacePtr_().Sf();
283  }
284  else
285  {
286  // Get oriented Sf
287  Sf = filterField(mesh().Sf(), true);
288  }
289 
290  // Combine onto master
291  combineFields(values);
292  combineFields(Sf);
293 
294  // Write raw values on surface if specified
295  if (surfaceWriterPtr_.valid())
296  {
297  faceList faces;
298  pointField points;
299 
300  if (surfacePtr_.valid())
301  {
302  combineSurfaceGeometry(faces, points);
303  }
304  else
305  {
306  combineMeshGeometry(faces, points);
307  }
308 
309  if (Pstream::master())
310  {
311  fileName outputDir =
312  baseFileDir()/name_/"surface"/obr_.time().timeName();
313 
314  surfaceWriterPtr_->write
315  (
316  outputDir,
317  word(sourceTypeNames_[source_]) + "_" + sourceName_,
318  points,
319  faces,
320  fieldName,
321  values,
322  false
323  );
324  }
325  }
326 
327 
328  // Apply scale factor
329  values *= scaleFactor_;
330 
331  if (Pstream::master())
332  {
333  Type result = processValues(values, Sf, weightField);
334 
335  // Add to result dictionary, over-writing any previous entry
336  resultDict_.add(fieldName, result, true);
337 
338  file()<< tab << result;
339 
340  if (log_) Info<< " " << operationTypeNames_[operation_]
341  << "(" << sourceName_ << ") of " << fieldName
342  << " = " << result << endl;
343  }
344  }
345 
346  return ok;
347 }

350 template<class Type>
351 Foam::tmp<Foam::Field<Type>> Foam::fieldValues::faceSource::filterField
352 (
353  const GeometricField<Type, fvPatchField, volMesh>& field,
354  const bool applyOrientation
355 ) const
356 {
357  tmp<Field<Type>> tvalues(new Field<Type>(faceId_.size()));
358  Field<Type>& values = tvalues.ref();
359 
360  forAll(values, i)
361  {
362  label faceI = faceId_[i];
363  label patchI = facePatchId_[i];
364  if (patchI >= 0)
365  {
366  values[i] = field.boundaryField()[patchI][faceI];
367  }
368  else
369  {
370  FatalErrorInFunction
371  << type() << " " << name_ << ": "
372  << sourceTypeNames_[source_] << "(" << sourceName_ << "):"
373  << nl
374  << " Unable to process internal faces for volume field "
375  << field.name() << nl << abort(FatalError);
376  }
377  }
378 
379  if (applyOrientation)
380  {
381  forAll(values, i)
382  {
383  values[i] *= faceSign_[i];
384  }
385  }
386 
387  return tvalues;
388 }
