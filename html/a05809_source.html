110 void Foam::fv::option::setCellSet()
111 {
112  switch (selectionMode_)
113  {
114  case smPoints:
115  {
116  Info<< indent << "- selecting cells using points" << endl;
117 
118  labelHashSet selectedCells;
119 
120  forAll(points_, i)
121  {
122  label cellI = mesh_.findCell(points_[i]);
123  if (cellI >= 0)
124  {
125  selectedCells.insert(cellI);
126  }
127 
128  label globalCellI = returnReduce(cellI, maxOp<label>());
129  if (globalCellI < 0)
130  {
131  WarningIn("option::setCellSet()")
132  << "Unable to find owner cell for point " << points_[i]
133  << endl;
134  }
135 
136  }
137 
138  cells_ = selectedCells.toc();
139 
140  break;
141  }
142  case smCellSet:
143  {
144  Info<< indent
145  << "- selecting cells using cellSet " << cellSetName_ << endl;
146 
147  cellSet selectedCells(mesh_, cellSetName_);
148  cells_ = selectedCells.toc();
149 
150  break;
151  }
152  case smCellZone:
153  {
154  Info<< indent
155  << "- selecting cells using cellZone " << cellSetName_ << endl;
156 
157  label zoneID = mesh_.cellZones().findZoneID(cellSetName_);
158  if (zoneID == -1)
159  {
160  FatalErrorIn("option::setCellIds()")
161  << "Cannot find cellZone " << cellSetName_ << endl
162  << "Valid cellZones are " << mesh_.cellZones().names()
163  << exit(FatalError);
164  }
165  cells_ = mesh_.cellZones()[zoneID];
166 
167  break;
168  }
169  case smMapRegion:
170  {
171  if (active_ && master_)
172  {
173  Info<< indent << "- selecting inter region mapping" << endl;
174 
175  const fvMesh& nbrMesh =
176  mesh_.time().lookupObject<fvMesh>(nbrRegionName_);
177 
178  if (mesh_.name() == nbrMesh.name())
179  {
180  FatalErrorIn("option::setCellIds()")
181  << "Inter-region model selected, but local and "
182  << "neighbour regions are the same: " << nl
183  << " local region: " << mesh_.name() << nl
184  << " secondary region: " << nbrMesh.name() << nl
185  << exit(FatalError);
186  }
187 
188  if (mesh_.bounds().overlaps(nbrMesh.bounds()))
189  {
190  meshInterpPtr_.reset
191  (
192  new meshToMesh
193  (
194  mesh_,
195  nbrMesh,
196  meshToMesh::interpolationMethodNames_.read
197  (
198  dict_.lookup("interpolationMethod")
199  ),
200  false // not interpolating patches
201  )
202  );
203  }
204  else
205  {
206  FatalErrorIn("option::setCellSet()")
207  << "regions " << mesh_.name() << " and "
208  << nbrMesh.name() << " do not intersect"
209  << exit(FatalError);
210  }
211 
212  V_ = meshInterpPtr_->V();
213  }
214  break;
215  }
216  case smAll:
217  {
218  Info<< indent << "- selecting all cells" << endl;
219  cells_ = identity(mesh_.nCells());
220 
221  break;
222  }
223  default:
224  {
225  FatalErrorIn("option::setCellSet()")
226  << "Unknown selectionMode "
227  << selectionModeTypeNames_[selectionMode_]
228  << ". Valid selectionMode types are" << selectionModeTypeNames_
229  << exit(FatalError);
230  }
231  }
232 
233  // Set volume information
234  if (selectionMode_ != smMapRegion)
235  {
236  V_ = 0.0;
237  forAll(cells_, i)
238  {
239  V_ += mesh_.V()[cells_[i]];
240  }
241  reduce(V_, sumOp<scalar>());
242 
243  Info<< indent
244  << "- selected " << returnReduce(cells_.size(), sumOp<label>())
245  << " cell(s) with volume " << V_ << nl << endl;
246  }
247 }

252 Foam::fv::option::option
253 (
254  const word& name,
255  const word& modelType,
256  const dictionary& dict,
257  const fvMesh& mesh,
258  const bool master
259 )
260 :
261  name_(name),

288  Info<< indent << "- applying source for all time" << endl;
289  }
290 
291  setSelection(dict_);
292 
293  setCellSet();
294 
295  Info<< decrIndent;

301 Foam::autoPtr<Foam::fv::option> Foam::fv::option::New
302 (
303  const word& name,
304  const dictionary& coeffs,
305  const fvMesh& mesh
306 )
307 {
308  word modelType(coeffs.lookup("type"));
309 
310  Info<< indent
311  << "Selecting finite volume options model type " << modelType << endl;
312 
313  dictionaryConstructorTable::iterator cstrIter =
314  dictionaryConstructorTablePtr_->find(modelType);
315 
316  if (cstrIter == dictionaryConstructorTablePtr_->end())
317  {
318  FatalErrorIn
319  (
320  "option::New(const word&, const dictionary&, const fvMesh&)"

322  << "Valid model types are:" << nl
323  << dictionaryConstructorTablePtr_->sortedToc()
324  << exit(FatalError);
325  }
326 
327  return autoPtr<option>(cstrIter()(name, modelType, coeffs, mesh));
328 }
