57 template<class Type>
58 typename Foam::IntegrationScheme<Type>::integrationResult
59 Foam::Analytical<Type>::integrate
60 (
61  const Type& phi,
62  const scalar dt,
63  const Type& alphaBeta,
64  const scalar beta
65 ) const
66 {
67  typename IntegrationScheme<Type>::integrationResult retValue;
68 
69  const scalar expTerm = exp(min(50, -beta*dt));
70 
71  if (beta > ROOTVSMALL)
72  {
73  const Type alpha = alphaBeta/beta;
74  retValue.average() = alpha + (phi - alpha)*(1 - expTerm)/(beta*dt);
75  retValue.value() = alpha + (phi - alpha)*expTerm;
76  }
77  else
78  {
79  retValue.average() = phi;
80  retValue.value() = phi;
81  }
82 
83 
84  return retValue;
85 }
