66 Foam::tmp<Foam::volScalarField>
67 Foam::dragModels::TomiyamaAnalytic::CdRe() const
68 {
69  volScalarField Eo(max(pair_.Eo(), residualEo_));
70  volScalarField E(max(pair_.E(), residualE_));
71 
72  volScalarField OmEsq(max(scalar(1) - sqr(E), sqr(residualE_)));
73  volScalarField rtOmEsq(sqrt(OmEsq));
74 
75  volScalarField F(max(asin(rtOmEsq) - E*rtOmEsq, residualE_)/OmEsq);
76 
77  return
78  (8.0/3.0)
79  *Eo
80  /(
81  Eo*pow(E, 2.0/3.0)/OmEsq
82  + 16*pow(E, 4.0/3.0)
83  )
84  /sqr(F)
85  *max(pair_.Re(), residualRe_);
86 }
