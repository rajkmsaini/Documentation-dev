101 bool Foam::functionObjects::fieldExpression::read(const dictionary& dict)
102 {
103  fvMeshFunctionObject::read(dict);
104 
105  if (fieldName_.empty() || dict.found("field"))
106  {
107  dict.lookup("field") >> fieldName_;
108  }
109 
110  if (dict.found("result"))
111  {
112  dict.lookup("result") >> resultName_;
113  }
114 
115  return true;
