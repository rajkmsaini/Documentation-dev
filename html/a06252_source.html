66 void Foam::skewCorrectionVectors::calcSkewCorrectionVectors()
67 {
68  if (debug)
69  {
70  InfoInFunction << "Calculating skew correction vectors" << endl;
71  }
72 
73  // Set local references to mesh data
74  const volVectorField& C = mesh_.C();
75  const surfaceVectorField& Cf = mesh_.Cf();
76  const surfaceVectorField& Sf = mesh_.Sf();
77 
78  const labelUList& owner = mesh_.owner();
79  const labelUList& neighbour = mesh_.neighbour();
80 
81  forAll(owner, facei)
82  {
83  label own = owner[facei];
84  label nei = neighbour[facei];
85 
86  vector d = C[nei] - C[own];
87  vector Cpf = Cf[facei] - C[own];
88 
89  skewCorrectionVectors_[facei] =
90  Cpf - ((Sf[facei] & Cpf)/(Sf[facei] & d))*d;
91  }
92 
93 
94  forAll(skewCorrectionVectors_.boundaryField(), patchI)
95  {
96  fvsPatchVectorField& patchSkewCorrVecs =
97  skewCorrectionVectors_.boundaryField()[patchI];
98 
99  if (!patchSkewCorrVecs.coupled())
100  {
101  patchSkewCorrVecs = Zero;
102  }
103  else
104  {
105  const fvPatch& p = patchSkewCorrVecs.patch();
106  const labelUList& faceCells = p.faceCells();
107  const vectorField& patchFaceCentres = Cf.boundaryField()[patchI];
108  const vectorField& patchSf = Sf.boundaryField()[patchI];
109  const vectorField patchD(p.delta());
110 
111  forAll(p, patchFaceI)
112  {
113  vector Cpf =
114  patchFaceCentres[patchFaceI] - C[faceCells[patchFaceI]];
115 
116  patchSkewCorrVecs[patchFaceI] =
117  Cpf
118  - (
119  (patchSf[patchFaceI] & Cpf)/
120  (patchSf[patchFaceI] & patchD[patchFaceI])
121  )*patchD[patchFaceI];
122  }
123  }
124  }
125 
126  scalar skewCoeff = 0.0;
127 
128  if (Sf.internalField().size())
129  {
130  skewCoeff =
131  max(mag(skewCorrectionVectors_)*mesh_.deltaCoeffs()).value();
132  }
133 
134  if (debug)
135  {
136  InfoInFunction << "skew coefficient = " << skewCoeff << endl;
137  }
138 
139  if (skewCoeff < 1e-5)
140  {
141  skew_ = false;
142  }
143  else
144  {
145  skew_ = true;
146  }
147 
148  if (debug)
149  {
150  Info<< " Finished constructing skew correction vectors" << endl;
151  }
152 }
