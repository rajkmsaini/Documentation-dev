122 template<class CompType, class SolidThermo>
123 Foam::tmp<Foam::volScalarField>
124 Foam::solidChemistryModel<CompType, SolidThermo>::Sh() const
125 {
126  tmp<volScalarField> tSh
127  (
128  new volScalarField
129  (
130  IOobject
131  (
132  "Sh",
133  this->mesh_.time().timeName(),
134  this->mesh_,
135  IOobject::NO_READ,
136  IOobject::AUTO_WRITE,
137  false
138  ),
139  this->mesh_,
140  dimensionedScalar("zero", dimEnergy/dimTime/dimVolume, 0.0),
141  zeroGradientFvPatchScalarField::typeName
142  )
143  );
144 
145  if (this->chemistry_)
146  {
147  scalarField& Sh = tSh();
148 
149  forAll(Ys_, i)
150  {
151  forAll(Sh, cellI)
152  {
153  scalar hf = solidThermo_[i].Hc();
154  Sh[cellI] -= hf*RRs_[i][cellI];
155  }
156  }
157  }
158 
159  return tSh;
160 }
