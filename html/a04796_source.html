55 Foam::functionObjects::streamLine::wallPatch() const
56 {
57  const polyBoundaryMesh& patches = mesh_.boundaryMesh();
58 
59  label nFaces = 0;
60 
61  forAll(patches, patchi)
62  {
63  if (isA<wallPolyPatch>(patches[patchi]))
64  {
65  nFaces += patches[patchi].size();

67  }
68 
69  labelList addressing(nFaces);
70 
71  nFaces = 0;
72 
73  forAll(patches, patchi)

84  }

87  (
88  new indirectPrimitivePatch
89  (
90  IndirectList<face>
91  (
92  mesh_.faces(),
93  addressing
94  ),
95  mesh_.points()
96  )

101 void Foam::functionObjects::streamLine::track()
102 {
103  IDLList<streamLineParticle> initialParticles;
104  streamLineParticleCloud particles
105  (
106  mesh_,
107  cloudName_,
108  initialParticles
109  );
110 
111  const sampledSet& seedPoints = sampledSetPtr_();
112 
113  forAll(seedPoints, i)
114  {
115  particles.addParticle
116  (
117  new streamLineParticle
118  (
119  mesh_,
120  seedPoints[i],
121  seedPoints.cells()[i],
122  lifeTime_
123  )
124  );
125  }
126 
127  label nSeeds = returnReduce(particles.size(), sumOp<label>());
128 
129  Info << " seeded " << nSeeds << " particles" << endl;
130 
131  // Read or lookup fields
132  PtrList<volScalarField> vsFlds;
133  PtrList<interpolation<scalar>> vsInterp;
134  PtrList<volVectorField> vvFlds;
135  PtrList<interpolation<vector>> vvInterp;
136 
137  label UIndex = -1;
138 
139  label nScalar = 0;
140  label nVector = 0;
141 
142  forAll(fields_, i)
143  {
144  if (mesh_.foundObject<volScalarField>(fields_[i]))
145  {
146  nScalar++;
147  }
148  else if (mesh_.foundObject<volVectorField>(fields_[i]))
149  {
150  nVector++;
151  }
152  else
153  {
154  FatalErrorInFunction
155  << "Cannot find field " << fields_[i] << nl
156  << "Valid scalar fields are:"
157  << mesh_.names(volScalarField::typeName) << nl
158  << "Valid vector fields are:"
159  << mesh_.names(volVectorField::typeName)
160  << exit(FatalError);
161  }
162  }
163  vsInterp.setSize(nScalar);
164  nScalar = 0;
165  vvInterp.setSize(nVector);
166  nVector = 0;
167 
168  forAll(fields_, i)
169  {
170  if (mesh_.foundObject<volScalarField>(fields_[i]))
171  {
172  const volScalarField& f = mesh_.lookupObject<volScalarField>
173  (
174  fields_[i]
175  );
176  vsInterp.set
177  (
178  nScalar++,
179  interpolation<scalar>::New
180  (
181  interpolationScheme_,
182  f
183  )
184  );
185  }
186  else if (mesh_.foundObject<volVectorField>(fields_[i]))
187  {
188  const volVectorField& f = mesh_.lookupObject<volVectorField>
189  (
190  fields_[i]
191  );
192 
193  if (f.name() == UName_)
194  {
195  UIndex = nVector;
196  }
197 
198  vvInterp.set
199  (
200  nVector++,
201  interpolation<vector>::New
202  (
203  interpolationScheme_,
204  f
205  )
206  );
207  }
208  }
209 
210  // Store the names
211  scalarNames_.setSize(vsInterp.size());
212  forAll(vsInterp, i)
213  {
214  scalarNames_[i] = vsInterp[i].psi().name();
215  }
216  vectorNames_.setSize(vvInterp.size());
217  forAll(vvInterp, i)
218  {
219  vectorNames_[i] = vvInterp[i].psi().name();
220  }
221 
222  // Check that we know the index of U in the interpolators.
223 
224  if (UIndex == -1)
225  {
226  FatalErrorInFunction
227  << "Cannot find field to move particles with : " << UName_ << nl
228  << "This field has to be present in the sampled fields " << fields_
229  << " and in the objectRegistry."
230  << exit(FatalError);
231  }
232 
233  // Sampled data
234  // ~~~~~~~~~~~~
235 
236  // Size to maximum expected sizes.
237  allTracks_.clear();
238  allTracks_.setCapacity(nSeeds);
239  allScalars_.setSize(vsInterp.size());
240  forAll(allScalars_, i)
241  {
242  allScalars_[i].clear();
243  allScalars_[i].setCapacity(nSeeds);
244  }
245  allVectors_.setSize(vvInterp.size());
246  forAll(allVectors_, i)
247  {
248  allVectors_[i].clear();
249  allVectors_[i].setCapacity(nSeeds);
250  }
251 
252 
253  // Additional particle info
254  streamLineParticle::trackingData td
255  (
256  particles,
257  vsInterp,
258  vvInterp,
259  UIndex, // index of U in vvInterp
260  trackForward_, // track in +u direction?
261  nSubCycle_, // automatic track control:step through cells in steps?
262  trackLength_, // fixed track length
263 
264  allTracks_,
265  allScalars_,
266  allVectors_
267  );
268 
269 
270  // Set very large dt. Note: cannot use GREAT since 1/GREAT is SMALL
271  // which is a trigger value for the tracking...
272  const scalar trackTime = Foam::sqrt(GREAT);
273 
274  // Track
275  particles.move(td, trackTime);
276 }

304 bool Foam::functionObjects::streamLine::read(const dictionary& dict)
305 {
306  if (dict != dict_)
307  {
308  dict_ = dict;
309  }
310 
311  Info<< type() << " " << name() << ":" << nl;
312 
313  dict.lookup("fields") >> fields_;
314  if (dict.found("U"))
315  {
316  dict.lookup("U") >> UName_;
317  }
318  else
319  {
320  UName_ = "U";
321  if (dict.found("U"))
322  {
323  IOWarningInFunction(dict)
324  << "Using deprecated entry \"U\"."
325  << " Please use \"UName\" instead."
326  << endl;
327  dict.lookup("U") >> UName_;
328  }
329  }
330 
331  if (findIndex(fields_, UName_) == -1)
332  {
333  FatalIOErrorInFunction(dict)
334  << "Velocity field for tracking " << UName_
335  << " should be present in the list of fields " << fields_
336  << exit(FatalIOError);
337  }
338 
339 
340  dict.lookup("trackForward") >> trackForward_;
341  dict.lookup("lifeTime") >> lifeTime_;
342  if (lifeTime_ < 1)
343  {
344  FatalErrorInFunction
345  << "Illegal value " << lifeTime_ << " for lifeTime"
346  << exit(FatalError);
347  }
348 
349 
350  bool subCycling = dict.found("nSubCycle");
351  bool fixedLength = dict.found("trackLength");
352 
353  if (subCycling && fixedLength)
354  {
355  FatalIOErrorInFunction(dict)
356  << "Cannot both specify automatic time stepping (through '"
357  << "nSubCycle' specification) and fixed track length (through '"
358  << "trackLength')"
359  << exit(FatalIOError);
360  }
361 
362 
363  nSubCycle_ = 1;
364  if (dict.readIfPresent("nSubCycle", nSubCycle_))
365  {
366  trackLength_ = VGREAT;
367  if (nSubCycle_ < 1)
368  {
369  nSubCycle_ = 1;
370  }
371  Info<< " automatic track length specified through"
372  << " number of sub cycles : " << nSubCycle_ << nl << endl;
373  }
374  else
375  {
376  dict.lookup("trackLength") >> trackLength_;
377 
378  Info<< " fixed track length specified : "
379  << trackLength_ << nl << endl;
380  }
381 
382 
383  interpolationScheme_ = dict.lookupOrDefault
384  (
385  "interpolationScheme",
386  interpolationCellPoint<scalar>::typeName
387  );
388 
389  cloudName_ = dict.lookupOrDefault<word>("cloudName", "streamLine");
390  dict.lookup("seedSampleSet") >> seedSet_;
391 
392  meshSearchPtr_.reset(new meshSearch(mesh_));
393 
394  const dictionary& coeffsDict = dict.subDict(seedSet_ + "Coeffs");
395  sampledSetPtr_ = sampledSet::New
396  (
397  seedSet_,
398  mesh_,
399  meshSearchPtr_(),
400  coeffsDict
401  );
402  coeffsDict.lookup("axis") >> sampledSetAxis_;
403 
404  scalarFormatterPtr_ = writer<scalar>::New(dict.lookup("setFormat"));
405  vectorFormatterPtr_ = writer<vector>::New(dict.lookup("setFormat"));
406 
407  return true;
408 }

417 bool Foam::functionObjects::streamLine::write()
418 {
419  Info<< type() << " " << name() << " write:" << nl;
420 
421  const Time& runTime = obr_.time();
422 
423  // Do all injection and tracking
424  track();
425 
426 
427  if (Pstream::parRun())
428  {
429  // Append slave tracks to master ones
430  // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
431 
432  globalIndex globalTrackIDs(allTracks_.size());
433 
434  // Construct a distribution map to pull all to the master.
435  labelListList sendMap(Pstream::nProcs());
436  labelListList recvMap(Pstream::nProcs());
437 
438  if (Pstream::master())
439  {
440  // Master: receive all. My own first, then consecutive
441  // processors.
442  label trackI = 0;
443 
444  forAll(recvMap, proci)
445  {
446  labelList& fromProc = recvMap[proci];
447  fromProc.setSize(globalTrackIDs.localSize(proci));
448  forAll(fromProc, i)
449  {
450  fromProc[i] = trackI++;
451  }
452  }
453  }
454 
455  labelList& toMaster = sendMap[0];
456  toMaster.setSize(globalTrackIDs.localSize());
457  forAll(toMaster, i)
458  {
459  toMaster[i] = i;
460  }
461 
462  const mapDistribute distMap
463  (
464  globalTrackIDs.size(),
465  sendMap.xfer(),
466  recvMap.xfer()
467  );
468 
469 
470  // Distribute the track positions. Note: use scheduled comms
471  // to prevent buffering.
472  mapDistributeBase::distribute
473  (
474  Pstream::scheduled,
475  distMap.schedule(),
476  distMap.constructSize(),
477  distMap.subMap(),
478  false,
479  distMap.constructMap(),
480  false,
481  allTracks_,
482  flipOp()
483  );
484 
485  // Distribute the scalars
486  forAll(allScalars_, scalarI)
487  {
488  allScalars_[scalarI].shrink();
489  mapDistributeBase::distribute
490  (
491  Pstream::scheduled,
492  distMap.schedule(),
493  distMap.constructSize(),
494  distMap.subMap(),
495  false,
496  distMap.constructMap(),
497  false,
498  allScalars_[scalarI],
499  flipOp()
500  );
501  allScalars_[scalarI].setCapacity(allScalars_[scalarI].size());
502  }
503  // Distribute the vectors
504  forAll(allVectors_, vectorI)
505  {
506  allVectors_[vectorI].shrink();
507  mapDistributeBase::distribute
508  (
509  Pstream::scheduled,
510  distMap.schedule(),
511  distMap.constructSize(),
512  distMap.subMap(),
513  false,
514  distMap.constructMap(),
515  false,
516  allVectors_[vectorI],
517  flipOp()
518  );
519  allVectors_[vectorI].setCapacity(allVectors_[vectorI].size());
520  }
521  }
522 
523 
524  label n = 0;
525  forAll(allTracks_, trackI)
526  {
527  n += allTracks_[trackI].size();
528  }
529 
530  Info<< " Tracks:" << allTracks_.size() << nl
531  << " Total samples:" << n
532  << endl;
533 
534 
535  // Massage into form suitable for writers
536  // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
537 
538  if (Pstream::master() && allTracks_.size())
539  {
540  // Make output directory
541 
542  fileName vtkPath
543  (
544  Pstream::parRun()
545  ? runTime.path()/".."/"postProcessing"/"sets"/name()
546  : runTime.path()/"postProcessing"/"sets"/name()
547  );
548  if (mesh_.name() != fvMesh::defaultRegion)
549  {
550  vtkPath = vtkPath/mesh_.name();
551  }
552  vtkPath = vtkPath/mesh_.time().timeName();
553 
554  mkDir(vtkPath);
555 
556  // Convert track positions
557 
558  PtrList<coordSet> tracks(allTracks_.size());
559  forAll(allTracks_, trackI)
560  {
561  tracks.set
562  (
563  trackI,
564  new coordSet
565  (
566  "track" + Foam::name(trackI),
567  sampledSetAxis_ //"xyz"
568  )
569  );
570  tracks[trackI].transfer(allTracks_[trackI]);
571  }
572 
573  // Convert scalar values
574 
575  if (allScalars_.size() > 0)
576  {
577  List<List<scalarField>> scalarValues(allScalars_.size());
578 
579  forAll(allScalars_, scalarI)
580  {
581  DynamicList<scalarList>& allTrackVals =
582  allScalars_[scalarI];
583  scalarValues[scalarI].setSize(allTrackVals.size());
584 
585  forAll(allTrackVals, trackI)
586  {
587  scalarList& trackVals = allTrackVals[trackI];
588  scalarValues[scalarI][trackI].transfer(trackVals);
589  }
590  }
591 
592  fileName vtkFile
593  (
594  vtkPath
595  / scalarFormatterPtr_().getFileName
596  (
597  tracks[0],
598  scalarNames_
599  )
600  );
601 
602  Info<< " Writing data to " << vtkFile.path() << endl;
603 
604  scalarFormatterPtr_().write
605  (
606  true, // writeTracks
607  tracks,
608  scalarNames_,
609  scalarValues,
610  OFstream(vtkFile)()
611  );
612  }
613 
614  // Convert vector values
615 
616  if (allVectors_.size() > 0)
617  {
618  List<List<vectorField>> vectorValues(allVectors_.size());
619 
620  forAll(allVectors_, vectorI)
621  {
622  DynamicList<vectorList>& allTrackVals =
623  allVectors_[vectorI];
624  vectorValues[vectorI].setSize(allTrackVals.size());
625 
626  forAll(allTrackVals, trackI)
627  {
628  vectorList& trackVals = allTrackVals[trackI];
629  vectorValues[vectorI][trackI].transfer(trackVals);
630  }
631  }
632 
633  fileName vtkFile
634  (
635  vtkPath
636  / vectorFormatterPtr_().getFileName
637  (
638  tracks[0],
639  vectorNames_
640  )
641  );
642 
643  vectorFormatterPtr_().write
644  (
645  true, // writeTracks
646  tracks,
647  vectorNames_,
648  vectorValues,
649  OFstream(vtkFile)()
650  );
651  }
652  }
653 
654  return true;
655 }
