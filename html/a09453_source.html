58 void Foam::functionObjects::wallShearStress::calcShearStress
59 (
60  const fvMesh& mesh,
61  const volSymmTensorField& Reff,
62  volVectorField& shearStress
63 )
64 {
65  forAllConstIter(labelHashSet, patchSet_, iter)
66  {
67  label patchi = iter.key();
68  const polyPatch& pp = mesh.boundaryMesh()[patchi];
69 
70  vectorField& ssp = shearStress.boundaryFieldRef()[patchi];
71  const vectorField& Sfp = mesh.Sf().boundaryField()[patchi];
72  const scalarField& magSfp = mesh.magSf().boundaryField()[patchi];
73  const symmTensorField& Reffp = Reff.boundaryField()[patchi];
74 
75  ssp = (-Sfp/magSfp) & Reffp;
76 
77  vector minSsp = gMin(ssp);
78  vector maxSsp = gMax(ssp);
79 
80  if (Pstream::master())
81  {
82  file() << mesh.time().value()
83  << token::TAB << pp.name()
84  << token::TAB << minSsp
85  << token::TAB << maxSsp
86  << endl;
87  }
88 
89  if (log_) Info<< " min/max(" << pp.name() << ") = "
90  << minSsp << ", " << maxSsp << endl;
91  }
92 }

162 void Foam::functionObjects::wallShearStress::read(const dictionary& dict)
163 {
164  log_ = dict.lookupOrDefault<Switch>("log", true);
165 
166  const fvMesh& mesh = refCast<const fvMesh>(obr_);
167  const polyBoundaryMesh& pbm = mesh.boundaryMesh();
168 
169  patchSet_ =
170  mesh.boundaryMesh().patchSet
171  (
172  wordReList(dict.lookupOrDefault("patches", wordReList()))
173  );
174 
175  Info<< type() << " " << name_ << ":" << nl;
176 
177  if (patchSet_.empty())
178  {
179  forAll(pbm, patchi)
180  {
181  if (isA<wallPolyPatch>(pbm[patchi]))
182  {
183  patchSet_.insert(patchi);
184  }
185  }
186 
187  Info<< " processing all wall patches" << nl << endl;
188  }
189  else
190  {
191  Info<< " processing wall patches: " << nl;
192  labelHashSet filteredPatchSet;
193  forAllConstIter(labelHashSet, patchSet_, iter)
194  {
195  label patchi = iter.key();
196  if (isA<wallPolyPatch>(pbm[patchi]))
197  {
198  filteredPatchSet.insert(patchi);
199  Info<< " " << pbm[patchi].name() << endl;
200  }
201  else
202  {
203  WarningInFunction
204  << "Requested wall shear stress on non-wall boundary "
205  << "type patch: " << pbm[patchi].name() << endl;
206  }
207  }
208 
209  Info<< endl;
210 
211  patchSet_ = filteredPatchSet;
212  }
213 }

216 void Foam::functionObjects::wallShearStress::execute()
217 {
218  typedef compressible::turbulenceModel cmpModel;
219  typedef incompressible::turbulenceModel icoModel;
220 
221  functionObjectFiles::write();
222 
223  const fvMesh& mesh = refCast<const fvMesh>(obr_);
224 
225  volVectorField& wallShearStress =
226  const_cast<volVectorField&>
227  (
228  mesh.lookupObject<volVectorField>(type())
229  );
230 
231  if (log_) Info<< type() << " " << name_ << " output:" << nl;
232 
233 
234  tmp<volSymmTensorField> Reff;
235  if (mesh.foundObject<cmpModel>(turbulenceModel::propertiesName))
236  {
237  const cmpModel& model =
238  mesh.lookupObject<cmpModel>(turbulenceModel::propertiesName);
239 
240  Reff = model.devRhoReff();
241  }
242  else if (mesh.foundObject<icoModel>(turbulenceModel::propertiesName))
243  {
244  const icoModel& model =
245  mesh.lookupObject<icoModel>(turbulenceModel::propertiesName);
246 
247  Reff = model.devReff();
248  }
249  else
250  {
251  FatalErrorInFunction
252  << "Unable to find turbulence model in the "
253  << "database" << exit(FatalError);
254  }
255 
256  calcShearStress(mesh, Reff(), wallShearStress);
257 }
