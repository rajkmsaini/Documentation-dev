56 bool Foam::tolerances::read()
57 {
58  if (regIOobject::read())
59  {
60  const word toleranceSetName(lookup("toleranceSet"));
61  const dictionary& toleranceSet(subDict(toleranceSetName));
62 
63  if (toleranceSet.found("relaxationFactors"))
64  {
65  relaxationFactors_ = toleranceSet.subDict("relaxationFactors");
66  }
67 
68  if (toleranceSet.found("solverTolerances"))
69  {
70  solverTolerances_ = toleranceSet.subDict("solverTolerances");
71  }
72 
73  if (toleranceSet.found("solverRelativeTolerances"))
74  {
75  solverRelativeTolerances_ =
76  toleranceSet.subDict("solverRelativeTolerances");
77  }
78 
79  return true;
80  }
81  else
82  {
83  return false;
84  }
85 }
