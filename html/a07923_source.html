412 {
413  label i = 0;
414  while (i < f.size() && !f[i].size()) i++;
415 
416  if (i < f.size())
417  {
418  Type Max(max(f[i]));

421  {
422  if (f[j].size())
423  {
424  Max = max(max(f[j]), Max);
425  }
426  }
427 
428  return Max;
