70 template<class Type>
71 Type Foam::fieldValues::cellSource::processValues
72 (
73  const Field<Type>& values,
74  const scalarField& V,
75  const scalarField& weightField
76 ) const
77 {
78  Type result = pTraits<Type>::zero;
79  switch (operation_)
80  {
81  case opSum:
82  {
83  result = sum(values);
84  break;
85  }
86  case opSumMag:
87  {
88  result = sum(cmptMag(values));
89  break;
90  }
91  case opAverage:
92  {
93  result = sum(values)/values.size();
94  break;
95  }
96  case opWeightedAverage:
97  {
98  result = sum(weightField*values)/sum(weightField);
99  break;
100  }
101  case opVolAverage:
102  {
103  result = sum(V*values)/sum(V);
104  break;
105  }
106  case opWeightedVolAverage:
107  {
108  result = sum(weightField*V*values)/sum(weightField*V);
109  break;
110  }
111  case opVolIntegrate:
112  {
113  result = sum(V*values);
114  break;
115  }
116  case opMin:
117  {
118  result = min(values);
119  break;
120  }
121  case opMax:
122  {
123  result = max(values);
124  break;
125  }
126  case opCoV:
127  {
128  Type meanValue = sum(values*V)/sum(V);
129 
130  const label nComp = pTraits<Type>::nComponents;
131 
132  for (direction d=0; d<nComp; ++d)
133  {
134  scalarField vals(values.component(d));
135  scalar mean = component(meanValue, d);
136  scalar& res = setComponent(result, d);
137 
138  res = sqrt(sum(V*sqr(vals - mean))/sum(V))/mean;
139  }
140 
141  break;
142  }
143  default:
144  {
145  // Do nothing
146  }
147  }
148 
149  return result;
150 }

155 template<class Type>
156 bool Foam::fieldValues::cellSource::writeValues(const word& fieldName)
157 {
158  const bool ok = validField<Type>(fieldName);
159 
160  if (ok)
161  {
162  Field<Type> values(setFieldValues<Type>(fieldName));
163  scalarField V(filterField(mesh().V()));
164  scalarField weightField(values.size(), 1.0);
165 
166  if (weightFieldName_ != "none")
167  {
168  weightField = setFieldValues<scalar>(weightFieldName_, true);
169  }
170 
171  // Combine onto master
172  combineFields(values);
173  combineFields(V);
174  combineFields(weightField);
175 
176  if (Pstream::master())
177  {
178  Type result = processValues(values, V, weightField);
179 
180  // Add to result dictionary, over-writing any previous entry
181  resultDict_.add(fieldName, result, true);
182 
183  if (valueOutput_)
184  {
185  IOField<Type>
186  (
187  IOobject
188  (
189  fieldName + "_" + sourceTypeNames_[source_] + "-"
190  + sourceName_,
191  obr_.time().timeName(),
192  obr_,
193  IOobject::NO_READ,
194  IOobject::NO_WRITE
195  ),
196  weightField*values
197  ).write();
198  }
199 
200 
201  file()<< tab << result;
202 
203  if (log_) Info<< " " << operationTypeNames_[operation_]
204  << "(" << sourceName_ << ") of " << fieldName
205  << " = " << result << endl;
206  }
207  }
208 
209  return ok;
210 }
