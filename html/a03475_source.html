63 Foam::tmp<Foam::volScalarField> Foam::dragModels::SyamlalOBrien::CdRe() const
64 {
65  volScalarField alpha2
66  (
67  max(scalar(1) - pair_.dispersed(), pair_.continuous().residualAlpha())
68  );
69 
70  volScalarField A(pow(alpha2, 4.14));
71  volScalarField B
72  (
73  neg(alpha2 - 0.85)*(0.8*pow(alpha2, 1.28))
74  + pos(alpha2 - 0.85)*(pow(alpha2, 2.65))
75  );
76  volScalarField Re(pair_.Re());
77  volScalarField Vr
78  (
79  0.5
80  *(
81  A - 0.06*Re + sqrt(sqr(0.06*Re) + 0.12*Re*(2.0*B - A) + sqr(A))
82  )
83  );
84  volScalarField CdsRe(sqr(0.63*sqrt(Re) + 4.8*sqrt(Vr)));
85 
86  return
87  CdsRe
88  *max(pair_.continuous(), pair_.continuous().residualAlpha())
89  /sqr(Vr);
