33 Foam::labelList Foam::fvMeshAdder::calcPatchMap
34 (
35  const label oldStart,
36  const label oldSize,
37  const labelList& oldToNew,
38  const polyPatch& newPatch,
39  const label unmappedValue
40 )
41 {
42  labelList newToOld(newPatch.size(), unmappedValue);
43 
44  label newStart = newPatch.start();
45  label newSize = newPatch.size();
46 
47  for (label i = 0; i < oldSize; i++)
48  {
49  label newFacei = oldToNew[oldStart+i];
50 
51  if (newFacei >= newStart && newFacei < newStart+newSize)
52  {
53  newToOld[newFacei-newStart] = i;
54  }
55  }
56  return newToOld;
57 }
