69 inline Foam::scalar Foam::ArrheniusReactionRate::operator()
70 (
71  const scalar p,
72  const scalar T,
73  const scalarField&
74 ) const
75 {
76  scalar ak = A_;
77 
78  if (mag(beta_) > VSMALL)
79  {
80  ak *= pow(T, beta_);
81  }
82 
83  if (mag(Ta_) > VSMALL)
84  {
85  ak *= exp(-Ta_/T);
86  }
87 
88  return ak;
89 }
