115 template<class CompType, class SolidThermo>
116 Foam::tmp<Foam::volScalarField>
117 Foam::solidChemistryModel<CompType, SolidThermo>::Sh() const
118 {
119  tmp<volScalarField> tSh
120  (
121  new volScalarField
122  (
123  IOobject
124  (
125  "Sh",
126  this->mesh_.time().timeName(),
127  this->mesh_,
128  IOobject::NO_READ,
129  IOobject::AUTO_WRITE,
130  false
131  ),
132  this->mesh_,
133  dimensionedScalar("zero", dimEnergy/dimTime/dimVolume, 0.0)
134  )
135  );
136 
137  if (this->chemistry_)
138  {
139  scalarField& Sh = tSh.ref();
140 
141  forAll(Ys_, i)
142  {
143  forAll(Sh, celli)
144  {
145  scalar hf = solidThermo_[i].Hc();
146  Sh[celli] -= hf*RRs_[i][celli];
147  }
148  }
149  }
150 
151  return tSh;
152 }
