36 void Foam::fanFvPatchField<Foam::scalar>::calcFanJump()
37 {
38  if (this->cyclicPatch().owner())
39  {
40  const surfaceScalarField& phi =
41  db().lookupObject<surfaceScalarField>(phiName_);
42 
43  const fvsPatchField<scalar>& phip =
44  patch().patchField<surfaceScalarField, scalar>(phi);
45 
46  scalarField Un(max(phip/patch().magSf(), scalar(0)));
47 
48  if (phi.dimensions() == dimDensity*dimVelocity*dimArea)
49  {
50  Un /= patch().lookupPatchField<volScalarField, scalar>(rhoName_);
51  }
52 
53  this->jump_ = max(this->jumpTable_->value(Un), scalar(0));
54  }
55 }

61 Foam::fanFvPatchField<Foam::scalar>::fanFvPatchField
62 (
63  const fvPatch& p,
64  const DimensionedField<scalar, volMesh>& iF,
65  const dictionary& dict
66 )
67 :
68  uniformJumpFvPatchField<scalar>(p, iF),

70  rhoName_(dict.lookupOrDefault<word>("rho", "rho"))
71 {
72  if (this->cyclicPatch().owner())
73  {
74  if (dict.found("f"))
75  {
76  // Backwards compatibility
77  Istream& is = dict.lookup("f");
78  is.format(IOstream::ASCII);
79  scalarList f(is);
80 
81  label nPows = 0;
82  forAll(f, powI)
83  {
84  if (mag(f[powI]) > VSMALL)
85  {
86  nPows++;
87  }
88  }
89  List<Tuple2<scalar, scalar>> coeffs(nPows);
90  nPows = 0;
91  forAll(f, powI)
92  {
93  if (mag(f[powI]) > VSMALL)
94  {
95  coeffs[nPows++] = Tuple2<scalar, scalar>(f[powI], powI);
96  }
97  }
98 
99  this->jumpTable_.reset
100  (
101  new Function1Types::Polynomial<scalar>("jumpTable", coeffs)
102  );
103  }
104  else
105  {
106  // Generic input constructed from dictionary
107  this->jumpTable_ = Function1<scalar>::New("jumpTable", dict);
108  }
109  }
110 
111  if (dict.found("value"))
112  {
113  fvPatchScalarField::operator=
114  (
115  scalarField("value", dict, p.size())
116  );
117  }
118  else
119  {
120  this->evaluate(Pstream::blocking);
121  }
122 }
