112 inline Foam::vector Foam::extendedFeatureEdgeMesh::edgeDirection
113 (
114  label edgeI,
115  label ptI
116 ) const
117 {
118  const edge& e = edges()[edgeI];
119 
120  if (ptI == e.start())
121  {
122  return edgeDirections()[edgeI];
123  }
124  else if (ptI == e.end())
125  {
126  return -edgeDirections()[edgeI];
127  }
128  else
129  {
130  FatalErrorInFunction
131  << "Requested ptI " << ptI << " is not a point on the requested "
132  << "edgeI " << edgeI << ". edgeI start and end: "
133  << e.start() << " " << e.end()
134  << exit(FatalError);
135 
136  return vector::zero;
137  }
138 }
