86 template<class CloudType>
87 void Foam::PackingModels::Implicit<CloudType>::cacheFields(const bool store)
88 {
89  PackingModel<CloudType>::cacheFields(store);
90 
91  if (store)
92  {
93  const fvMesh& mesh = this->owner().mesh();
94  const dimensionedScalar deltaT = this->owner().db().time().deltaT();
95  const word& cloudName = this->owner().name();
96 
97  const dimensionedVector& g = this->owner().g();
98  const volScalarField& rhoc = this->owner().rho();
99 
100  const AveragingMethod<scalar>& rhoAverage =
101  mesh.lookupObject<AveragingMethod<scalar> >
102  (
103  cloudName + ":rhoAverage"
104  );
105  const AveragingMethod<scalar>& uSqrAverage =
106  mesh.lookupObject<AveragingMethod<scalar> >
107  (
108  cloudName + ":uSqrAverage"
109  );
110 
111 
112  // Property fields
113  // ~~~~~~~~~~~~~~~
114 
115  // volume fraction field
116  alpha_ = max(this->owner().theta(), alphaMin_);
117  alpha_.correctBoundaryConditions();
118 
119  // average density
120  volScalarField rho
121  (
122  IOobject
123  (
124  cloudName + ":rho",
125  this->owner().db().time().timeName(),
126  mesh,
127  IOobject::NO_READ,
128  IOobject::NO_WRITE
129  ),
130  mesh,
131  dimensionedScalar("zero", dimDensity, 0),
132  zeroGradientFvPatchField<scalar>::typeName
133  );
134  rho.internalField() = max(rhoAverage.internalField(), rhoMin_);
135  rho.correctBoundaryConditions();
136 
137  //Info << " x: " << mesh.C().internalField().component(2) << endl;
138  //Info << " alpha: " << alpha_.internalField() << endl;
139  //Info << "alphaOld: " << alpha_.oldTime().internalField() << endl;
140  //Info << " rho: " << rho.internalField() << endl;
141  //Info << endl;
142 
143 
144  // Stress field
145  // ~~~~~~~~~~~~
146 
147  // stress derivative wrt volume fraction
148  volScalarField tauPrime
149  (
150  IOobject
151  (
152  cloudName + ":tauPrime",
153  this->owner().db().time().timeName(),
154  mesh,
155  IOobject::NO_READ,
156  IOobject::NO_WRITE
157  ),
158  mesh,
159  dimensionedScalar("zero", dimPressure, 0),
160  zeroGradientFvPatchField<scalar>::typeName
161  );
162 
163  tauPrime.internalField() =
164  this->particleStressModel_->dTaudTheta
165  (
166  alpha_.internalField(),
167  rho.internalField(),
168  uSqrAverage.internalField()
169  )();
170 
171  tauPrime.correctBoundaryConditions();
172 
173 
174  // Implicit solution for the volume fraction
175  // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
176 
177  surfaceScalarField
178  tauPrimeByRhoAf
179  (
180  "tauPrimeByRhoAf",
181  fvc::interpolate(deltaT*tauPrime/rho)
182  );
183 
184  fvScalarMatrix alphaEqn
185  (
186  fvm::ddt(alpha_)
187  - fvc::ddt(alpha_)
188  - fvm::laplacian(tauPrimeByRhoAf, alpha_)
189  );
190 
191  if (applyGravity_)
192  {
193  surfaceScalarField
194  phiGByA
195  (
196  "phiGByA",
197  deltaT*(g & mesh.Sf())*fvc::interpolate(1.0 - rhoc/rho)
198  );
199 
200  alphaEqn += fvm::div(phiGByA, alpha_);
201  }
202 
203  alphaEqn.solve();
204 
205 
206  // Generate correction fields
207  // ~~~~~~~~~~~~~~~~~
208 
209  // correction volumetric flux
210  phiCorrect_ = tmp<surfaceScalarField>
211  (
212  new surfaceScalarField
213  (
214  cloudName + ":phiCorrect",
215  alphaEqn.flux()/fvc::interpolate(alpha_)
216  )
217  );
218 
219  // correction velocity
220  uCorrect_ = tmp<volVectorField>
221  (
222  new volVectorField
223  (
224  cloudName + ":uCorrect",
225  fvc::reconstruct(phiCorrect_())
226  )
227 
228  );
229  uCorrect_->correctBoundaryConditions();
230 
231  //Info << endl;
232  //Info << " alpha: " << alpha_.internalField() << endl;
233  //Info << "phiCorrect: " << phiCorrect_->internalField() << endl;
234  //Info << " uCorrect: " << uCorrect_->internalField() << endl;
235  //Info << endl;
236  }
237  else
238  {
239  alpha_.oldTime();
240  phiCorrect_.clear();
241  uCorrect_.clear();
242  }
243 }

246 template<class CloudType>
247 Foam::vector Foam::PackingModels::Implicit<CloudType>::velocityCorrection
248 (
249  typename CloudType::parcelType& p,
250  const scalar deltaT
251 ) const
252 {
253  const fvMesh& mesh = this->owner().mesh();
254 
255  // containing tetrahedron and parcel coordinates within
256  const label cellI = p.cell();
257  const label faceI = p.tetFace();
258  const tetIndices tetIs(cellI, faceI, p.tetPt(), mesh);
259  List<scalar> tetCoordinates(4);
260  tetIs.tet(mesh).barycentric(p.position(), tetCoordinates);
261 
262  // cell velocity
263  const vector U = uCorrect_()[cellI];
264 
265  // face geometry
266  vector nHat = mesh.faces()[faceI].normal(mesh.points());
267  const scalar nMag = mag(nHat);
268  nHat /= nMag;
269 
270  // get face flux
271  scalar phi;
272  const label patchI = mesh.boundaryMesh().whichPatch(faceI);
273  if (patchI == -1)
274  {
275  phi = phiCorrect_()[faceI];
276  }
277  else
278  {
279  phi =
280  phiCorrect_().boundaryField()[patchI]
281  [
282  mesh.boundaryMesh()[patchI].whichFace(faceI)
283  ];
284  }
285 
286  // interpolant equal to 1 at the cell centre and 0 at the face
287  const scalar t = tetCoordinates[0];
288 
289  // the normal component of the velocity correction is interpolated linearly
290  // the tangential component is equal to that at the cell centre
291  return U + (1.0 - t)*nHat*(phi/nMag - (U & nHat));
292 }
