99 bool Foam::functionObjects::writeObjects::read(const dictionary& dict)
100 {
101  if (dict.found("field"))
102  {
103  objectNames_.setSize(1);
104  dict.lookup("field") >> objectNames_[0];
105  }
106  else if (dict.found("fields"))
107  {
108  dict.lookup("fields") >> objectNames_;
109  }
110  else
111  {
112  dict.lookup("objects") >> objectNames_;
113  }
114 
115  if (dict.found("writeOption"))
116  {
117  writeOption_ = writeOptionNames_.read(dict.lookup("writeOption"));
118  }
119  else
120  {
121  writeOption_ = ANY_WRITE;
122  }
123 
124  return true;
125 }

134 bool Foam::functionObjects::writeObjects::write()
135 {
136  Info<< type() << " " << name() << " write:" << nl;
137 
138  if (!obr_.time().writeTime())
139  {
140  obr_.time().writeTimeDict();
141  }
142 
143  DynamicList<word> allNames(obr_.toc().size());
144  forAll(objectNames_, i)
145  {
146  wordList names(obr_.names<regIOobject>(objectNames_[i]));
147 
148  if (names.size())
149  {
150  allNames.append(names);
151  }
152  else
153  {
154  WarningInFunction
155  << "Object " << objectNames_[i] << " not found in "
156  << "database. Available objects:" << nl << obr_.sortedToc()
157  << endl;
158  }
159  }
160 
161  forAll(allNames, i)
162  {
163  regIOobject& obj = const_cast<regIOobject&>
164  (
165  obr_.lookupObject<regIOobject>(allNames[i])
166  );
167 
168  switch (writeOption_)
169  {
170  case AUTO_WRITE:
171  {
172  if(obj.writeOpt() != IOobject::AUTO_WRITE)
173  {
174  continue;
175  }
176 
177  break;
178  }
179  case NO_WRITE:
180  {
181  if(obj.writeOpt() != IOobject::NO_WRITE)
182  {
183  continue;
184  }
185 
186  break;
187  }
188  case ANY_WRITE:
189  {
190  break;
191  }
192  default:
193  {
194  FatalErrorInFunction
195  << "Unknown writeOption "
196  << writeOptionNames_[writeOption_]
197  << ". Valid writeOption types are" << writeOptionNames_
198  << exit(FatalError);
199  }
200  }
201 
202  if
203  (
204  obj.writeOpt() == IOobject::AUTO_WRITE
205  && obr_.time().writeTime()
206  )
207  {
208  Info<< " automatically written object " << obj.name() << endl;
209  }
210  else
211  {
212  Info<< " writing object " << obj.name() << endl;
213 
214  obj.write();
215  }
216  }
217 
218  return true;
219 }
