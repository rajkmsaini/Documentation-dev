43 calcLocalPointOrder() const

45  // Note: Cannot use bandCompressing as point-point addressing does

48

52  << "calcLocalPointOrder() : "
53  << "calculating local point order"
54  << endl;
55  }
56 
57  if (localPointOrderPtr_)
58  {
59  // it is considered an error to attempt to recalculate
60  // if already allocated
61  FatalErrorInFunction
62  << "local point order already calculated"
63  << abort(FatalError);
64  }
65 
66  const List<Face>& lf = localFaces();
67 
68  const labelListList& ff = faceFaces();
69 
70  boolList visitedFace(lf.size(), false);
71 
72  localPointOrderPtr_ = new labelList(meshPoints().size(), -1);
73 
74  labelList& pointOrder = *localPointOrderPtr_;
75 
76  boolList visitedPoint(pointOrder.size(), false);
77 
78  label nPoints = 0;
79 
80  forAll(lf, facei)
81  {
82  if (!visitedFace[facei])
83  {
84  SLList<label> faceOrder(facei);
85 
86  do
87  {
88  const label curFace = faceOrder.first();
89 
90  faceOrder.removeHead();
91 
92  if (!visitedFace[curFace])
93  {
94  visitedFace[curFace] = true;
95 
96  const labelList& curPoints = lf[curFace];
97 
98  // mark points
99  forAll(curPoints, pointi)
100  {
101  if (!visitedPoint[curPoints[pointi]])
102  {
103  visitedPoint[curPoints[pointi]] = true;
104 
105  pointOrder[nPoints] = curPoints[pointi];
106 
107  nPoints++;
108  }
109  }
110 
111  // add face neighbours to the list
112  const labelList& nbrs = ff[curFace];
113 
114  forAll(nbrs, nbrI)
115  {
116  if (!visitedFace[nbrs[nbrI]])
117  {
118  faceOrder.append(nbrs[nbrI]);
119  }
120  }
121  }
122  } while (faceOrder.size());
123  }
124  }
125 
126  if (debug)

128  Pout<< "PrimitivePatch<Face, FaceList, PointField, PointType>::"
129  << "calcLocalPointOrder() "
130  << "finished calculating local point order"
131  << endl;
132  }
133 }
