34 template<class T>
35 T Foam::ReadHex(ISstream& is)
36 {
37  // Takes into account that 'a' (or 'A') is 10
38  static const int alphaOffset = toupper('A') - 10;
39  // Takes into account that '0' is 0
40  static const int zeroOffset = int('0');
41 
42  char c = 0;
43 
44  // Get next non-whitespace character
45  while (is.get(c) && isspace(c))
46  {}
47 
48  T result = 0;
49  do
50  {
51  if (isspace(c) || c == 0) break;
52 
53  if (!isxdigit(c))
54  {
55  FatalIOErrorInFunction(is)
56  << "Illegal hex digit: '" << c << "'"
57  << exit(FatalIOError);
58  }
59 
60  result <<= 4;
61 
62  if (isdigit(c))
63  {
64  result += int(c) - zeroOffset;
65  }
66  else
67  {
68  result += toupper(c) - alphaOffset;
69  }
70  } while (is.get(c));
71 
72  return result;
73 }
