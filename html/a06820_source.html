70  const scalarField& y0,
71  const scalarField& dydx0,
72  const scalar dx,
73  scalarField& y
74 ) const
75 {
76  // Calculate error estimate from the change in state:
77  forAll(err_, i)

81 
82  // Update the state

86  }
