31 template<class Type>
32 bool Foam::functionObjects::randomise::calcRandomised()
33 {
34  typedef GeometricField<Type, fvPatchField, volMesh> VolFieldType;
35 
36  if (foundObject<VolFieldType>(fieldName_))
37  {
38  const VolFieldType& field = lookupObject<VolFieldType>(fieldName_);
39 
40  resultName_ = fieldName_ + "Random";
41 
42  tmp<VolFieldType> rfieldt(new VolFieldType(field));
43  VolFieldType& rfield = rfieldt.ref();
44 
45  Random rand(1234567);
46 
47  forAll(field, celli)
48  {
49  Type rndPert;
50  rand.randomise(rndPert);
51  rndPert = 2.0*rndPert - pTraits<Type>::one;
52  rndPert /= mag(rndPert);
53  rfield[celli] += magPerturbation_*rndPert;
54  }
55 
56  return store(resultName_, rfieldt);
57  }
58  else
59  {
60  return false;
61  }
62 }
