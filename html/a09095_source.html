246 Foam::fileName Foam::cwd()
247 {
248  label pathLengthLimit = POSIX::pathLengthChunk;
249  List<char> path(pathLengthLimit);
250 
251  // Resize path if getcwd fails with an ERANGE error
252  while(pathLengthLimit == path.size())
253  {
254  if (::getcwd(path.data(), path.size()))
255  {
256  return path.data();
257  }
258  else if(errno == ERANGE)
259  {
260  // Increment path length upto the pathLengthMax limit
261  if
262  (
263  (pathLengthLimit += POSIX::pathLengthChunk)
264  >= POSIX::pathLengthMax
265  )
266  {
267  FatalErrorInFunction
268  << "Attempt to increase path length beyond limit of "
269  << POSIX::pathLengthMax
270  << exit(FatalError);
271  }
272 
273  path.setSize(pathLengthLimit);
274  }
275  else
276  {
277  break;
278  }
279  }
280 
281  FatalErrorInFunction
282  << "Couldn't get the current working directory"
283  << exit(FatalError);
284 
285  return fileName::null;
286 }

444 bool Foam::mkDir(const fileName& pathName, mode_t mode)
445 {
446  // empty names are meaningless
447  if (pathName.empty())
448  {
449  return false;
450  }
451 
452  // Construct instance path directory if does not exist
453  if (::mkdir(pathName.c_str(), mode) == 0)
454  {
455  // Directory made OK so return true
456  return true;
457  }
458  else
459  {
460  switch (errno)
461  {
462  case EPERM:
463  {
464  FatalErrorInFunction
465  << "The filesystem containing " << pathName
466  << " does not support the creation of directories."
467  << exit(FatalError);
468 
469  return false;
470  }
471 
472  case EEXIST:
473  {
474  // Directory already exists so simply return true
475  return true;
476  }
477 
478  case EFAULT:
479  {
480  FatalErrorInFunction
481  << "" << pathName
482  << " points outside your accessible address space."
483  << exit(FatalError);
484 
485  return false;
486  }
487 
488  case EACCES:
489  {
490  FatalErrorInFunction
491  << "The parent directory does not allow write "
492  "permission to the process,"<< nl
493  << "or one of the directories in " << pathName
494  << " did not allow search (execute) permission."
495  << exit(FatalError);
496 
497  return false;
498  }
499 
500  case ENAMETOOLONG:
501  {
502  FatalErrorInFunction
503  << "" << pathName << " is too long."
504  << exit(FatalError);
505 
506  return false;
507  }
508 
509  case ENOENT:
510  {
511  // Part of the path does not exist so try to create it
512  if (pathName.path().size() && mkDir(pathName.path(), mode))
513  {
514  return mkDir(pathName, mode);
515  }
516  else
517  {
518  FatalErrorInFunction
519  << "Couldn't create directory " << pathName
520  << exit(FatalError);
521 
522  return false;
523  }
524  }
525 
526  case ENOTDIR:
527  {
528  FatalErrorInFunction
529  << "A component used as a directory in " << pathName
530  << " is not, in fact, a directory."
531  << exit(FatalError);
532 
533  return false;
534  }
535 
536  case ENOMEM:
537  {
538  FatalErrorInFunction
539  << "Insufficient kernel memory was available to make "
540  "directory " << pathName << '.'
541  << exit(FatalError);
542 
543  return false;
544  }
545 
546  case EROFS:
547  {
548  FatalErrorInFunction
549  << "" << pathName
550  << " refers to a file on a read-only filesystem."
551  << exit(FatalError);
552 
553  return false;
554  }
555 
556  case ELOOP:
557  {
558  FatalErrorInFunction
559  << "Too many symbolic links were encountered in resolving "
560  << pathName << '.'
561  << exit(FatalError);
562 
563  return false;
564  }
565 
566  case ENOSPC:
567  {
568  FatalErrorInFunction
569  << "The device containing " << pathName
570  << " has no room for the new directory or "
571  << "the user's disk quota is exhausted."
572  << exit(FatalError);
573 
574  return false;
575  }
576 
577  default:
578  {
579  FatalErrorInFunction
580  << "Couldn't create directory " << pathName
581  << exit(FatalError);
582 
583  return false;
584  }
585  }
586  }
587 }

675 Foam::fileNameList Foam::readDir
676 (
677  const fileName& directory,
678  const fileName::Type type,
679  const bool filtergz
680 )
681 {
682  // Initial filename list size
683  // also used as increment if initial size found to be insufficient
684  static const int maxNnames = 100;
685 
686  if (POSIX::debug)
687  {
688  InfoInFunction
689  << "reading directory " << directory << endl;
690  }
691 
692  // Setup empty string list MAXTVALUES long
693  fileNameList dirEntries(maxNnames);
694 
695  // Pointers to the directory entries
696  DIR *source;
697  struct dirent *list;
698 
699  // Temporary variables and counters
700  label nEntries = 0;
701 
702  // Attempt to open directory and set the structure pointer
703  if ((source = ::opendir(directory.c_str())) == NULL)
704  {
705  dirEntries.setSize(0);
706 
707  if (POSIX::debug)
708  {
709  InfoInFunction
710  << "cannot open directory " << directory << endl;
711  }
712  }
713  else
714  {
715  // Read and parse all the entries in the directory
716  while ((list = ::readdir(source)) != NULL)
717  {
718  fileName fName(list->d_name);
719 
720  // ignore files begining with ., i.e. '.', '..' and '.*'
721  if (fName.size() && fName[0] != '.')
722  {
723  word fExt = fName.ext();
724 
725  if
726  (
727  (type == fileName::DIRECTORY)
728  ||
729  (
730  type == fileName::FILE
731  && fName[fName.size()-1] != '~'
732  && fExt != "bak"
733  && fExt != "BAK"
734  && fExt != "old"
735  && fExt != "save"
736  )
737  )
738  {
739  if ((directory/fName).type() == type)
740  {
741  if (nEntries >= dirEntries.size())
742  {
743  dirEntries.setSize(dirEntries.size() + maxNnames);
744  }
745 
746  if (filtergz && fExt == "gz")
747  {
748  dirEntries[nEntries++] = fName.lessExt();
749  }
750  else
751  {
752  dirEntries[nEntries++] = fName;
753  }
754  }
755  }
756  }
757  }
758 
759  // Reset the length of the entries list
760  dirEntries.setSize(nEntries);
761 
762  ::closedir(source);
763  }
764 
765  return dirEntries;
766 }

769 bool Foam::cp(const fileName& src, const fileName& dest)
770 {
771  // Make sure source exists.
772  if (!exists(src))
773  {
774  return false;
775  }
776 
777  fileName destFile(dest);
778 
779  // Check type of source file.
780  if (src.type() == fileName::FILE)
781  {
782  // If dest is a directory, create the destination file name.
783  if (destFile.type() == fileName::DIRECTORY)
784  {
785  destFile = destFile/src.name();
786  }
787 
788  // Make sure the destination directory exists.
789  if (!isDir(destFile.path()) && !mkDir(destFile.path()))
790  {
791  return false;
792  }
793 
794  // Open and check streams.
795  std::ifstream srcStream(src.c_str());
796  if (!srcStream)
797  {
798  return false;
799  }
800 
801  std::ofstream destStream(destFile.c_str());
802  if (!destStream)
803  {
804  return false;
805  }
806 
807  // Copy character data.
808  char ch;
809  while (srcStream.get(ch))
810  {
811  destStream.put(ch);
812  }
813 
814  // Final check.
815  if (!srcStream.eof() || !destStream)
816  {
817  return false;
818  }
819  }
820  else if (src.type() == fileName::DIRECTORY)
821  {
822  // If dest is a directory, create the destination file name.
823  if (destFile.type() == fileName::DIRECTORY)
824  {
825  destFile = destFile/src.component(src.components().size() -1);
826  }
827 
828  // Make sure the destination directory exists.
829  if (!isDir(destFile) && !mkDir(destFile))
830  {
831  return false;
832  }
833 
834  // Copy files
835  fileNameList contents = readDir(src, fileName::FILE, false);
836  forAll(contents, i)
837  {
838  if (POSIX::debug)
839  {
840  InfoInFunction
841  << "Copying : " << src/contents[i]
842  << " to " << destFile/contents[i] << endl;
843  }
844 
845  // File to file.
846  cp(src/contents[i], destFile/contents[i]);
847  }
848 
849  // Copy sub directories.
850  fileNameList subdirs = readDir(src, fileName::DIRECTORY);
851  forAll(subdirs, i)
852  {
853  if (POSIX::debug)
854  {
855  InfoInFunction
856  << "Copying : " << src/subdirs[i]
857  << " to " << destFile << endl;
858  }
859 
860  // Dir to Dir.
861  cp(src/subdirs[i], destFile);
862  }
863  }
864 
865  return true;
866 }
