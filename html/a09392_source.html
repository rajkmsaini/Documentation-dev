108 void Foam::Q::execute()
109 {
110  if (active_)
111  {
112  const fvMesh& mesh = refCast<const fvMesh>(obr_);
113 
114  const volVectorField& U =
115  mesh.lookupObject<volVectorField>(UName_);
116 
117  const volTensorField gradU(fvc::grad(U));
118 
119  volScalarField& Q =
120  const_cast<volScalarField&>
121  (
122  mesh.lookupObject<volScalarField>(type())
123  );
124 
125  Q = 0.5*(sqr(tr(gradU)) - tr(((gradU) & (gradU))));
126  }
127 }
