99 void Foam::energyJumpAMIFvPatchScalarField::updateCoeffs()
100 {
101  if (this->updated())
102  {
103  return;
104  }
105 
106  if (this->cyclicAMIPatch().owner())
107  {
108  const basicThermo& thermo = basicThermo::lookupThermo(*this);
109  label patchID = patch().index();
110 
111  const scalarField& pp = thermo.p().boundaryField()[patchID];
112  const fixedJumpAMIFvPatchScalarField& TbPatch =
113  refCast<const fixedJumpAMIFvPatchScalarField>
114  (
115  thermo.T().boundaryField()[patchID]
116  );
117 
118  fixedJumpAMIFvPatchScalarField& Tbp =
119  const_cast<fixedJumpAMIFvPatchScalarField&>(TbPatch);
120 
121  // force update of jump
122  Tbp.updateCoeffs();
123 
124  const labelUList& faceCells = this->patch().faceCells();
125 
126  jump_ = thermo.he(pp, Tbp.jump(), faceCells);
127  }

130 }
