128 template<class Type>
129 void Foam::cyclicAMIPointPatchField<Type>::swapAddSeparated
130 (
131  const Pstream::commsTypes,
132  Field<Type>& pField
133 ) const
134 {
135  if (cyclicAMIPatch_.cyclicAMIPatch().owner())
136  {
137  // We inplace modify pField. To prevent the other side (which gets
138  // evaluated at a later date) using already changed values we do
139  // all swaps on the side that gets evaluated first.
140 
141  // Get neighbouring pointPatch
142  const cyclicAMIPointPatch& nbrPatch = cyclicAMIPatch_.neighbPatch();
143 
144  // Get neighbouring pointPatchField
145  const GeometricField<Type, pointPatchField, pointMesh>& fld =
146  refCast<const GeometricField<Type, pointPatchField, pointMesh> >
147  (
148  this->dimensionedInternalField()
149  );
150 
151  const cyclicAMIPointPatchField<Type>& nbr =
152  refCast<const cyclicAMIPointPatchField<Type> >
153  (
154  fld.boundaryField()[nbrPatch.index()]
155  );
156 
157 
158  Field<Type> ptFld(this->patchInternalField(pField));
159  Field<Type> nbrPtFld(nbr.patchInternalField(pField));
160 
161 
162  if (doTransform())
163  {
164  const tensor& forwardT = this->forwardT()[0];
165  const tensor& reverseT = this->reverseT()[0];
166 
167  transform(ptFld, reverseT, ptFld);
168  transform(nbrPtFld, forwardT, nbrPtFld);
169  }
170 
171  // convert point field to face field, AMI interpolate, then
172  // face back to point
173  {
174  // add neighbour side contribution to owner
175  Field<Type> nbrFcFld(nbrPpi().pointToFaceInterpolate(nbrPtFld));
176 
177  // interpolate to owner
178  if (cyclicAMIPatch_.cyclicAMIPatch().applyLowWeightCorrection())
179  {
180  Field<Type> fcFld(ppi().pointToFaceInterpolate(ptFld));
181 
182  nbrFcFld =
183  cyclicAMIPatch_.cyclicAMIPatch().interpolate
184  (
185  nbrFcFld,
186  fcFld
187  );
188  }
189  else
190  {
191  nbrFcFld =
192  cyclicAMIPatch_.cyclicAMIPatch().interpolate(nbrFcFld);
193  }
194 
195  // add to internal field
196  this->addToInternalField
197  (
198  pField,
199  ppi().faceToPointInterpolate(nbrFcFld)()
200  );
201  }
202 
203  {
204  // add owner side contribution to neighbour
205  Field<Type> fcFld(ppi().pointToFaceInterpolate(ptFld));
206 
207  // interpolate to neighbour
208  if (cyclicAMIPatch_.cyclicAMIPatch().applyLowWeightCorrection())
209  {
210  Field<Type> nbrFcFld(nbrPpi().pointToFaceInterpolate(nbrPtFld));
211 
212  fcFld =
213  cyclicAMIPatch_.cyclicAMIPatch().neighbPatch().interpolate
214  (
215  fcFld,
216  nbrFcFld
217  );
218  }
219  else
220  {
221  fcFld =
222  cyclicAMIPatch_.cyclicAMIPatch().neighbPatch().interpolate
223  (
224  fcFld
225  );
226  }
227 
228  // add to internal field
229  nbr.addToInternalField
230  (
231  pField,
232  nbrPpi().faceToPointInterpolate(fcFld)()
233  );
234  }
235  }
236 }
