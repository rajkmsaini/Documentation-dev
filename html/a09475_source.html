60 void Foam::functionObjects::wallShearStress::calcShearStress
61 (
62  const fvMesh& mesh,
63  const volSymmTensorField& Reff,
64  volVectorField& shearStress
65 )
66 {
67  forAllConstIter(labelHashSet, patchSet_, iter)
68  {
69  label patchi = iter.key();
70  const polyPatch& pp = mesh.boundaryMesh()[patchi];
71 
72  vectorField& ssp = shearStress.boundaryFieldRef()[patchi];
73  const vectorField& Sfp = mesh.Sf().boundaryField()[patchi];
74  const scalarField& magSfp = mesh.magSf().boundaryField()[patchi];
75  const symmTensorField& Reffp = Reff.boundaryField()[patchi];
76 
77  ssp = (-Sfp/magSfp) & Reffp;
78 
79  vector minSsp = gMin(ssp);
80  vector maxSsp = gMax(ssp);
81 
82  if (Pstream::master())
83  {
84  file() << mesh.time().value()
85  << token::TAB << pp.name()
86  << token::TAB << minSsp
87  << token::TAB << maxSsp
88  << endl;
89  }
90 
91  if (log_) Info<< " min/max(" << pp.name() << ") = "
92  << minSsp << ", " << maxSsp << endl;
93  }
94 }

154 bool Foam::functionObjects::wallShearStress::read(const dictionary& dict)
155 {
156  writeFiles::read(dict);
157 
158  const fvMesh& mesh = refCast<const fvMesh>(obr_);
159  const polyBoundaryMesh& pbm = mesh.boundaryMesh();
160 
161  patchSet_ =
162  mesh.boundaryMesh().patchSet
163  (
164  wordReList(dict.lookupOrDefault("patches", wordReList()))
165  );
166 
167  Info<< type() << " " << name() << ":" << nl;
168 
169  if (patchSet_.empty())
170  {
171  forAll(pbm, patchi)
172  {
173  if (isA<wallPolyPatch>(pbm[patchi]))
174  {
175  patchSet_.insert(patchi);
176  }
177  }
178 
179  Info<< " processing all wall patches" << nl << endl;
180  }
181  else
182  {
183  Info<< " processing wall patches: " << nl;
184  labelHashSet filteredPatchSet;
185  forAllConstIter(labelHashSet, patchSet_, iter)
186  {
187  label patchi = iter.key();
188  if (isA<wallPolyPatch>(pbm[patchi]))
189  {
190  filteredPatchSet.insert(patchi);
191  Info<< " " << pbm[patchi].name() << endl;
192  }
193  else
194  {
195  WarningInFunction
196  << "Requested wall shear stress on non-wall boundary "
197  << "type patch: " << pbm[patchi].name() << endl;
198  }
199  }
200 
201  Info<< endl;
202 
203  patchSet_ = filteredPatchSet;
204  }
205 
206  return true;
207 }

210 bool Foam::functionObjects::wallShearStress::execute(const bool postProcess)
211 {
212  typedef compressible::turbulenceModel cmpModel;
213  typedef incompressible::turbulenceModel icoModel;
214 
215  writeFiles::write();
216 
217  const fvMesh& mesh = refCast<const fvMesh>(obr_);
218 
219  volVectorField& wallShearStress =
220  const_cast<volVectorField&>
221  (
222  mesh.lookupObject<volVectorField>(type())
223  );
224 
225  if (log_) Info<< type() << " " << name() << " output:" << nl;
226 
227 
228  tmp<volSymmTensorField> Reff;
229  if (mesh.foundObject<cmpModel>(turbulenceModel::propertiesName))
230  {
231  const cmpModel& model =
232  mesh.lookupObject<cmpModel>(turbulenceModel::propertiesName);
233 
234  Reff = model.devRhoReff();
235  }
236  else if (mesh.foundObject<icoModel>(turbulenceModel::propertiesName))
237  {
238  const icoModel& model =
239  mesh.lookupObject<icoModel>(turbulenceModel::propertiesName);
240 
241  Reff = model.devReff();
242  }
243  else
244  {
245  FatalErrorInFunction
246  << "Unable to find turbulence model in the "
247  << "database" << exit(FatalError);
248  }
249 
250  calcShearStress(mesh, Reff(), wallShearStress);
251 
252  return true;
253 }
