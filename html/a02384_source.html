98 void Foam::rotatingTotalPressureFvPatchScalarField::updateCoeffs()
99 {
100  if (updated())
101  {
102  return;
103  }
104 
105  const scalar t = this->db().time().timeOutputValue();
106  const vector om = omega_->value(t);
107 
108  vector axisHat = om/mag(om);
109  tmp<vectorField> rotationVelocity =
110  om ^ (patch().Cf() - axisHat*(axisHat & patch().Cf()));
111 
112  const vectorField Up
113  (
114  patch().lookupPatchField<volVectorField, vector>(UName())
115  + rotationVelocity
116  );
117 
118  totalPressureFvPatchScalarField::updateCoeffs(p0(), Up);
119 }
