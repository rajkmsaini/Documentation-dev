310 inline Scalar sqrtSumSqr(const Scalar a, const Scalar b)
311 {
312  Scalar maga = mag(a);
313  Scalar magb = mag(b);
314 
315  if (maga > magb)
316  {
317  return maga*sqrt(1.0 + sqr(magb/maga));
318  }
319  else
320  {
321  return magb < ScalarVSMALL ? 0.0 : magb*sqrt(1.0 + sqr(maga/magb));
322  }
323 }
