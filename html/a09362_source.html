72 template<class Type>
73 void Foam::blendingFactor::calc()
74 {
75  typedef GeometricField<Type, fvPatchField, volMesh> fieldType;
76 
77  if (!obr_.foundObject<fieldType>(fieldName_))
78  {
79  return;
80  }
81 
82  const fvMesh& mesh = refCast<const fvMesh>(obr_);
83 
84  const fieldType& field = mesh.lookupObject<fieldType>(fieldName_);
85 
86  const word divScheme("div(" + phiName_ + ',' + fieldName_ + ')');
87  ITstream& its = mesh.divScheme(divScheme);
88 
89  const surfaceScalarField& phi =
90  mesh.lookupObject<surfaceScalarField>(phiName_);
91 
92  tmp<fv::convectionScheme<Type>> cs =
93  fv::convectionScheme<Type>::New(mesh, phi, its);
94 
95  const fv::gaussConvectionScheme<Type>& gcs =
96  refCast<const fv::gaussConvectionScheme<Type>>(cs());
97 
98  const surfaceInterpolationScheme<Type>& interpScheme =
99  gcs.interpScheme();
100 
101  if (!isA<blendedSchemeBase<Type>>(interpScheme))
102  {
103  FatalErrorInFunction
104  << interpScheme.typeName << " is not a blended scheme"
105  << exit(FatalError);
106  }
107 
108  // retrieve the face-based blending factor
109  const blendedSchemeBase<Type>& blendedScheme =
110  refCast<const blendedSchemeBase<Type>>(interpScheme);
111  const surfaceScalarField factorf(blendedScheme.blendingFactor(field));
112 
113  // convert into vol field whose values represent the local face maxima
114  volScalarField& factor = this->factor(field);
115  factor = fvc::cellReduce(factorf, maxEqOp<scalar>());
116  factor.correctBoundaryConditions();
117 }
