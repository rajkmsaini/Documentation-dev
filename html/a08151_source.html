100 Foam::Istream& Foam::UIPstream::read(token& t)
101 {
102  // Return the put back token if it exists
103  if (Istream::getBack(t))
104  {
105  return *this;
106  }
107 
108  char c;
109 
110  // return on error
111  if (!read(c))
112  {
113  t.setBad();
114  return *this;
115  }
116 
117  // Set the line number of this token to the current stream line number
118  t.lineNumber() = lineNumber();
119 
120  // Analyse input starting with this character.
121  switch (c)
122  {
123  // Punctuation
124  case token::END_STATEMENT :
125  case token::BEGIN_LIST :
126  case token::END_LIST :
127  case token::BEGIN_SQR :
128  case token::END_SQR :
129  case token::BEGIN_BLOCK :
130  case token::END_BLOCK :
131  case token::COLON :
132  case token::COMMA :
133  case token::ASSIGN :
134  case token::ADD :
135  case token::SUBTRACT :
136  case token::MULTIPLY :
137  case token::DIVIDE :
138  {
139  t = token::punctuationToken(c);
140  return *this;
141  }
142 
143  // Word
144  case token::WORD :
145  {
146  word* pval = new word;
147  if (read(*pval))
148  {
149  if (token::compound::isCompound(*pval))
150  {
151  t = token::compound::New(*pval, *this).ptr();
152  delete pval;
153  }
154  else
155  {
156  t = pval;
157  }
158  }
159  else
160  {
161  delete pval;
162  t.setBad();
163  }
164  return *this;
165  }
166 
167  // String
168  case token::VERBATIMSTRING :
169  {
170  // Recurse to read actual string
171  read(t);
172  t.type() = token::VERBATIMSTRING;
173  return *this;
174  }
175  case token::VARIABLE :
176  {
177  // Recurse to read actual string
178  read(t);
179  t.type() = token::VARIABLE;
180  return *this;
181  }
182  case token::STRING :
183  {
184  string* pval = new string;
185  if (read(*pval))
186  {
187  t = pval;
188  if (c == token::VERBATIMSTRING)
189  {
190  t.type() = token::VERBATIMSTRING;
191  }
192  }
193  else
194  {
195  delete pval;
196  t.setBad();
197  }
198  return *this;
199  }
200 
201  // Label
202  case token::LABEL :
203  {
204  label val;
205  if (read(val))
206  {
207  t = val;
208  }
209  else
210  {
211  t.setBad();
212  }
213  return *this;
214  }
215 
216  // floatScalar
217  case token::FLOAT_SCALAR :
218  {
219  floatScalar val;
220  if (read(val))
221  {
222  t = val;
223  }
224  else
225  {
226  t.setBad();
227  }
228  return *this;
229  }
230 
231  // doubleScalar
232  case token::DOUBLE_SCALAR :
233  {
234  doubleScalar val;
235  if (read(val))
236  {
237  t = val;
238  }
239  else
240  {
241  t.setBad();
242  }
243  return *this;
244  }
245 
246  // Character (returned as a single character word) or error
247  default:
248  {
249  if (isalpha(c))
250  {
251  t = word(c);
252  return *this;
253  }
254 
255  setBad();
256  t.setBad();
257 
258  return *this;
259  }
260  }
261 }
