37 )

44  nReconstructed_(0)
45 {
46  forAll(procMeshes_, proci)

48  const fvMesh& procMesh = procMeshes_[proci];

53  || boundaryProcAddressing[proci].size() != procMesh.boundary().size()
54  )

63  << "boundaryProcAddressing : "
