27 Description
28  Mesh and field preparation utility for PDR type simulations.
29 
30  Reads
31  - cellSet giving blockedCells
32  - faceSets giving blockedFaces and the patch they should go into
33 
34  NOTE: To avoid exposing wrong fields values faceSets should include
35  faces contained in the blockedCells cellset.
36 
37  - coupledFaces reads coupledFacesSet to introduces mixe-coupled baffles
38 
39  Subsets out the blocked cells and splits the blockedFaces and updates
40  fields.
41 
42  The face splitting is done by duplicating the faces. No duplication of
43  points for now (so checkMesh will show a lot of 'duplicate face' messages)

526 // Wrapper around find patch. Also makes sure same patch in parallel.
527 label findPatch(const polyBoundaryMesh& patches, const word& patchName)
528 {
529  label patchI = patches.findPatchID(patchName);
530 
531  if (patchI == -1)
532  {
533  FatalErrorInFunction
534  << "Illegal patch " << patchName
535  << nl << "Valid patches are " << patches.names()
536  << exit(FatalError);
537  }
538 
539  // Check same patch for all procs
540  {
541  label newPatch = patchI;
542  reduce(newPatch, minOp<label>());
543 
544  if (newPatch != patchI)
545  {
546  FatalErrorInFunction
547  << "Patch " << patchName
548  << " should have the same patch index on all processors." << nl
549  << "On my processor it has index " << patchI
550  << " ; on some other processor it has index " << newPatch
551  << exit(FatalError);
552  }
553  }
554  return patchI;
555 }
