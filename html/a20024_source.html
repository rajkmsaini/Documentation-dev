43 int main(int argc, char *argv[])
44 {
45  argList::validArgs.append("CHEMKINFile");
46  argList::validArgs.append("CHEMKINThermodynamicsFile");
47  argList::validArgs.append("CHEMKINTransport");
48  argList::validArgs.append("FOAMChemistryFile");
49  argList::validArgs.append("FOAMThermodynamicsFile");
50 
51  argList::addBoolOption
52  (
53  "newFormat",
54  "read Chemkin thermo file in new format"
55  );
56 
57  argList args(argc, argv);
58 
59  bool newFormat = args.optionFound("newFormat");
60 
61  speciesTable species;
62 
63  chemkinReader cr(species, args[1], args[3], args[2], newFormat);
64 
65 
66  OFstream reactionsFile(args[4]);
67  reactionsFile
68  << "elements" << cr.elementNames() << token::END_STATEMENT << nl << nl;
69  reactionsFile
70  << "species" << cr.species() << token::END_STATEMENT << nl << nl;
71  cr.reactions().write(reactionsFile);
72 
73 
74  // Temporary hack to splice the specie composition data into the thermo file
75  // pending complete integration into the thermodynamics structure
76 
77  OStringStream os;
78  cr.speciesThermo().write(os);
79  dictionary thermoDict(IStringStream(os.str())());
80 
81  wordList speciesList(thermoDict.toc());
82 
83  // Add elements
84  forAll(speciesList, si)
85  {
86  dictionary elementsDict("elements");
87  forAll(cr.specieComposition()[speciesList[si]], ei)
88  {
89  elementsDict.add
90  (
91  cr.specieComposition()[speciesList[si]][ei].name(),
92  cr.specieComposition()[speciesList[si]][ei].nAtoms()
93  );
94  }
95 
96  thermoDict.subDict(speciesList[si]).add("elements", elementsDict);
97  }
98 
99  thermoDict.write(OFstream(args[5])(), false);
100 
101 
102  Info<< "End\n" << endl;
103 
104  return 0;
105 }
