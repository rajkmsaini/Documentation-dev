132 {

134  {

140 
141  label oldSize = this->size();
142 
143  if (newSize == 0)
144  {
145  clear();
146  }

196 template<class T>
197 void Foam::PtrList<T>::reorder(const labelUList& oldToNew)
198 {
199  if (oldToNew.size() != this->size())
200  {
201  FatalErrorInFunction
202  << "Size of map (" << oldToNew.size()
203  << ") not equal to list size (" << this->size()
204  << ") for type " << typeid(T).name()
205  << abort(FatalError);
206  }
207 
208  List<T*> newPtrs_(this->ptrs_.size(), reinterpret_cast<T*>(0));
209 
210  forAll(*this, i)
211  {
212  label newI = oldToNew[i];
213 
214  if (newI < 0 || newI >= this->size())
215  {
216  FatalErrorInFunction
217  << "Illegal index " << newI << nl
218  << "Valid indices are 0.." << this->size()-1
219  << " for type " << typeid(T).name()
220  << abort(FatalError);
221  }
222 
223  if (newPtrs_[newI])
224  {
225  FatalErrorInFunction
226  << "reorder map is not unique; element " << newI
227  << " already set for type " << typeid(T).name()
228  << abort(FatalError);
229  }
230  newPtrs_[newI] = this->ptrs_[i];
231  }
232 
233  forAll(newPtrs_, i)
234  {
235  if (!newPtrs_[i])
236  {
237  FatalErrorInFunction
238  << "Element " << i << " not set after reordering with type "
239  << typeid(T).name() << nl << abort(FatalError);
240  }
241  }
242 
243  this->ptrs_.transfer(newPtrs_);
244 }

249 template<class T>
250 void Foam::PtrList<T>::operator=(const PtrList<T>& a)
251 {
252  if (this == &a)
253  {
254  FatalErrorInFunction
255  << "attempted assignment to self for type " << typeid(T).name()
256  << abort(FatalError);
257  }
258 
259  if (this->size() == 0)
260  {
261  setSize(a.size());
262 
263  forAll(*this, i)
264  {
265  this->ptrs_[i] = (a[i]).clone().ptr();
266  }
267  }
268  else if (a.size() == this->size())
269  {
270  forAll(*this, i)
271  {
272  (*this)[i] = a[i];
273  }
274  }
275  else
276  {
277  FatalErrorInFunction
278  << "bad size: " << a.size()
279  << " for type " << typeid(T).name()
280  << abort(FatalError);
281  }
282 }
