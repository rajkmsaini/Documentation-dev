252 Foam::wordList Foam::polyPatch::constraintTypes()
253 {
254  wordList cTypes(dictionaryConstructorTablePtr_->size());
255 
256  label i = 0;
257 
258  for
259  (
260  dictionaryConstructorTable::iterator cstrIter =
261  dictionaryConstructorTablePtr_->begin();
262  cstrIter != dictionaryConstructorTablePtr_->end();
263  ++cstrIter
264  )
265  {
266  if (constraintType(cstrIter.key()))
267  {
268  cTypes[i++] = cstrIter.key();
269  }
270  }
271 
272  cTypes.setSize(i);
273 
274  return cTypes;

296 Foam::tmp<Foam::vectorField> Foam::polyPatch::faceCellCentres() const
297 {
298  tmp<vectorField> tcc(new vectorField(size()));
299  vectorField& cc = tcc.ref();
300 
301  // get reference to global cell centres
302  const vectorField& gcc = boundaryMesh_.mesh().cellCentres();
303 
304  const labelUList& faceCells = this->faceCells();
305 
306  forAll(faceCells, facei)
307  {
308  cc[facei] = gcc[faceCells[facei]];
309  }
310 
311  return tcc;
312 }
