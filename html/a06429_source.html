43 void Foam::functionObjects::forceCoeffs::writeFileHeader(const label i)
44 {
45  if (i == 0)
46  {
47  // force coeff data
48 
49  writeHeader(file(i), "Force coefficients");
50  writeHeaderValue(file(i), "liftDir", liftDir_);
51  writeHeaderValue(file(i), "dragDir", dragDir_);
52  writeHeaderValue(file(i), "pitchAxis", pitchAxis_);
53  writeHeaderValue(file(i), "magUInf", magUInf_);
54  writeHeaderValue(file(i), "lRef", lRef_);
55  writeHeaderValue(file(i), "Aref", Aref_);
56  writeHeaderValue(file(i), "CofR", coordSys_.origin());
57  writeCommented(file(i), "Time");
58  writeTabbed(file(i), "Cm");
59  writeTabbed(file(i), "Cd");
60  writeTabbed(file(i), "Cl");
61  writeTabbed(file(i), "Cl(f)");
62  writeTabbed(file(i), "Cl(r)");
63  file(i)
64  << tab << "Cm" << tab << "Cd" << tab << "Cl" << tab << "Cl(f)"
65  << tab << "Cl(r)";
66  }
67  else if (i == 1)
68  {
69  // bin coeff data
70 
71  writeHeader(file(i), "Force coefficient bins");
72  writeHeaderValue(file(i), "bins", nBin_);
73  writeHeaderValue(file(i), "start", binMin_);
74  writeHeaderValue(file(i), "delta", binDx_);
75  writeHeaderValue(file(i), "direction", binDir_);
76 
77  vectorField binPoints(nBin_);
78  writeCommented(file(i), "x co-ords :");
79  forAll(binPoints, pointi)
80  {
81  binPoints[pointi] = (binMin_ + (pointi + 1)*binDx_)*binDir_;
82  file(i) << tab << binPoints[pointi].x();
83  }
84  file(i) << nl;
85 
86  writeCommented(file(i), "y co-ords :");
87  forAll(binPoints, pointi)
88  {
89  file(i) << tab << binPoints[pointi].y();
90  }
91  file(i) << nl;
92 
93  writeCommented(file(i), "z co-ords :");
94  forAll(binPoints, pointi)
95  {
96  file(i) << tab << binPoints[pointi].z();
97  }
98  file(i) << nl;
99 
100  writeCommented(file(i), "Time");
101 
102  for (label j = 0; j < nBin_; j++)
103  {
104  const word jn('(' + Foam::name(j) + ')');
105  writeTabbed(file(i), "Cm" + jn);
106  writeTabbed(file(i), "Cd" + jn);
107  writeTabbed(file(i), "Cl" + jn);
108  }
109  }
110  else
111  {
112  FatalErrorInFunction

115  }

176 bool Foam::functionObjects::forceCoeffs::write()
177 {
178  forces::calcForcesMoment();
179 
180  if (Pstream::master())
181  {
182  writeFiles::write();
183 
184  scalar pDyn = 0.5*rhoRef_*magUInf_*magUInf_;
185 
186  Field<vector> totForce(force_[0] + force_[1] + force_[2]);
187  Field<vector> totMoment(moment_[0] + moment_[1] + moment_[2]);
188 
189  List<Field<scalar>> coeffs(3);
190  coeffs[0].setSize(nBin_);
191  coeffs[1].setSize(nBin_);
192  coeffs[2].setSize(nBin_);
193 
194  // lift, drag and moment
195  coeffs[0] = (totForce & liftDir_)/(Aref_*pDyn);
196  coeffs[1] = (totForce & dragDir_)/(Aref_*pDyn);
197  coeffs[2] = (totMoment & pitchAxis_)/(Aref_*lRef_*pDyn);
198 
199  scalar Cl = sum(coeffs[0]);
200  scalar Cd = sum(coeffs[1]);
201  scalar Cm = sum(coeffs[2]);
202 
203  scalar Clf = Cl/2.0 + Cm;
204  scalar Clr = Cl/2.0 - Cm;
205 
206  writeTime(file(0));
207  file(0)
208  << tab << Cm << tab << Cd
209  << tab << Cl << tab << Clf << tab << Clr << endl;
210 
211  Log << type() << " " << name() << " write:" << nl
212  << " Cm = " << Cm << nl
213  << " Cd = " << Cd << nl
214  << " Cl = " << Cl << nl
215  << " Cl(f) = " << Clf << nl
216  << " Cl(r) = " << Clr << endl;
217 
218  if (nBin_ > 1)
219  {
220  if (binCumulative_)
221  {
222  for (label i = 1; i < coeffs[0].size(); i++)
223  {
224  coeffs[0][i] += coeffs[0][i-1];
225  coeffs[1][i] += coeffs[1][i-1];
226  coeffs[2][i] += coeffs[2][i-1];
227  }
228  }
229 
230  writeTime(file(1));
231 
232  forAll(coeffs[0], i)
233  {
234  file(1)
235  << tab << coeffs[2][i]
236  << tab << coeffs[1][i]
237  << tab << coeffs[0][i];
238  }
239 
240  file(1) << endl;
241  }
242 
243  Log << endl;
244  }
245 
246  return true;
247 }
